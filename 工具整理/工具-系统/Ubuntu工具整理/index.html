    1.oh-my-zsh终端工具¶

    1.1.安装¶

    # 安装zsh
sudo apt-get install zsh
# 查看shell
cat /etc/shells
echo $SHELL

# 安装 oh-my-zsh
sh -c "$(curl -fsSL https://raw.githubusercontent.com/robbyrussell/oh-my-zsh/master/tools/install.sh)"
# or

wget https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh
./install.sh

# 修改
chsh -s $(which zsh)

    1.2.插件¶

    • 安装
    # 更快地切换目录
sudo apt install autojump
# 快速访问文件或目录
sudo apt install fasd

# 补全提示
git clone https://github.com/zsh-users/zsh-autosuggestions ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-autosuggestions

# 高亮
git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting
    • 配置
    plugins=(
  git
  extract
  autojump
  zsh-autosuggestions
  zsh-syntax-highlighting
)

    1.3.字体¶

    • 安装
    mkdir ~/.local/share/fonts

# 方法1
sudo apt install fonts-powerline
# or
git clone https://github.com/powerline/fonts.git --depth=1
cd fonts
./install.sh

# 方法2
# 下载 符号字体 和 字体配置文件
wget https://github.com/powerline/powerline/raw/develop/font/PowerlineSymbols.otf
wget https://github.com/powerline/powerline/raw/develop/font/10-powerline-symbols.conf
mv PowerlineSymbols.otf ~/.local/share/fonts/
fc-cache -vf ~/.local/share/fonts/
mv 10-powerline-symbols.conf ~/.config/fontconfig/conf.d/

# 刷新
fc-cache -f -v
fc-cache -vf ~/.local/share/fonts/
    • 配置
    # 字体配置管理
sudo apt install fontconfig -y

fc-list | grep "<name-of-font>"  # 查看
    • 使用
    setfont
sudo dpkg-reconfigure console-setup

# Alt + F2 输入 gnome-tweaks 打开界面进行 字体 设置
sudo apt install gnome-tweaks

    3.Go 环境搭建¶

    3.1.go安装¶

    # 下载
wget https://golang.google.cn/dl/go1.18.3.linux-amd64.tar.gz
# 解压
tar -C /usr/local -xzf go1.18.3.linux-amd64.tar.gz
# 添加环境变量
echo -e "GOROOT=/usr/local/go\nPATH=$PATH:$GOROOT/bin" >>  /etc/profile
export GOPATH=/home/yaojusheng/GOPATH

    3.2.依赖工具¶

    go install golang.org/x/tools/go/buildutil

go env -w GOPROXY=https://goproxy.cn,direct # 修改代理
go env   # 查看修改结果

    3.3.版本打包¶

    go build [-o output] [-i] [build flags] [packages]

# 安装 goreleaser
sudo apt install snap
snap install goreleaser --classic
# 或者
apt install goreleaser

goreleaser release --skip-publish --rm-dist

    3.4.其他命令¶

    • go get：安装第三方包
    • go clean：移除当前源码包和关联源码包里面编译生成的文件
    • go fmt：格式化代码
    • go run: 编译并运行Go程序
    • go mode 命令:
    go mod download    # 下载依赖的module到本地cache（默认为$GOPATH/pkg/mod目录）
go mod edit        # 编辑go.mod文件
go mod graph       # 打印模块依赖图
go mod init        # 初始化当前文件夹, 创建go.mod文件
go mod tidy        # 增加缺少的module，删除无用的module
go mod vendor      # 将依赖复制到vendor下
go mod verify      # 校验依赖
go mod why         # 解释为什么需要依赖
    • go list 命令
    go list -m -f '{{if not .Indirect}}{{.Path}}{{end}}' all # 查看项目所有依赖包

    参考: golang编译程序以及部署

export GTK_IM_MODULE=fcitx
export QT_IM_MODULE=fcitx
export XMODIFIERS=@im=fcitx

    9.2.lazydocker¶

    # 设置版本
LAZYDOCKER_VERSION=$(curl -s "https://api.github.com/repos/jesseduffield/lazydocker/releases/latest" | grep -Po '"tag_name": "v\K[0-9.]+')
# 下载
curl -Lo lazydocker.tar.gz "https://github.com/jesseduffield/lazydocker/releases/latest/download/lazydocker_${LAZYDOCKER_VERSION}_Linux_x86_64.tar.gz"
# 解压
mkdir lazydocker-temp
tar xf lazydocker.tar.gz -C lazydocker-temp
sudo mv lazydocker-temp/lazydocker /usr/local/bin
# 删除
rm -rf lazydocker.tar.gz
rm -rf lazydocker-temp
# 查看
lazydocker --version
