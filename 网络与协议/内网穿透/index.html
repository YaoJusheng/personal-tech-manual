    1. 内网穿透概述¶

    内网穿透是一种网络技术，允许外部网络访问位于NAT或防火墙后面的内网服务。它解决了内网设备无法直接从公网访问的问题。

    1.1 应用场景¶

    • 远程办公: 访问公司内网资源
    • IoT设备管理: 远程控制智能设备
    • 开发调试: 外网访问本地开发服务
    • 游戏联机: 突破NAT限制
    • 文件共享: 远程访问家庭NAS

    1.2 技术原理¶

    内网穿透主要通过以下方式实现：

    • 端口映射: 路由器端口转发
    • 反向代理: 通过中间服务器转发
    • 隧道技术: 建立加密通道
    • P2P打洞: 直接建立点对点连接

    2. 主流穿透技术¶

    2.1 frp (Fast Reverse Proxy)¶

    frp是一个高性能的反向代理应用，支持TCP、UDP、HTTP、HTTPS等协议。

    服务端配置 (frps.ini)¶

    [common]
# 服务端监听端口
bind_port = 7000

# 面板端口
dashboard_port = 7500
dashboard_user = admin
dashboard_pwd = admin123

# 认证token
token = your_token_here

# 日志配置
log_file = ./frps.log
log_level = info
log_max_days = 3

# 允许的端口范围
allow_ports = 2000-3000,3001,3003,4000-50000

# 最大连接池大小
max_pool_count = 5

# 心跳超时时间
heartbeat_timeout = 90

# 用户连接超时时间
user_conn_timeout = 10

    客户端配置 (frpc.ini)¶

    [common]
# 服务端地址和端口
server_addr = your_server_ip
server_port = 7000

# 认证token
token = your_token_here

# 日志配置
log_file = ./frpc.log
log_level = info
log_max_days = 3

# 心跳间隔
heartbeat_interval = 30
heartbeat_timeout = 90

# HTTP代理配置
[web]
type = http
local_ip = 127.0.0.1
local_port = 8080
custom_domains = your_domain.com

# HTTPS代理配置
[web_https]
type = https
local_ip = 127.0.0.1
local_port = 8443
custom_domains = your_domain.com

# TCP代理配置
[ssh]
type = tcp
local_ip = 127.0.0.1
local_port = 22
remote_port = 6000

# UDP代理配置
[dns]
type = udp
local_ip = 127.0.0.1
local_port = 53
remote_port = 6053

# 文件服务器
[file_server]
type = tcp
local_ip = 127.0.0.1
local_port = 80
remote_port = 8080

    2.2 ngrok¶

    ngrok是一个商业化的内网穿透服务，提供免费和付费版本。

    基础使用¶

    # 安装ngrok
wget https://bin.equinox.io/c/4VmDzA7iaHb/ngrok-stable-linux-amd64.zip
unzip ngrok-stable-linux-amd64.zip

# 配置authtoken
./ngrok authtoken YOUR_AUTH_TOKEN

# 穿透HTTP服务
./ngrok http 8080

# 穿透TCP服务
./ngrok tcp 22

# 自定义域名
./ngrok http -hostname=myapp.ngrok.io 8080

    配置文件 (ngrok.yml)¶

    version: "2"
authtoken: YOUR_AUTH_TOKEN

tunnels:
  web:
    proto: http
    addr: 8080
    hostname: myapp.ngrok.io

  ssh:
    proto: tcp
    addr: 22

  api:
    proto: http
    addr: 3000
    auth: "user:password"

  secure:
    proto: http
    addr: 8443
    schemes:
      - https

    2.3 ZeroTier¶

    ZeroTier创建虚拟网络，让设备像在同一局域网中通信。

    安装和配置¶

    # 安装ZeroTier
curl -s https://install.zerotier.com | sudo bash

# 加入网络
sudo zerotier-cli join NETWORK_ID

# 查看状态
sudo zerotier-cli status

# 列出网络
sudo zerotier-cli listnetworks

# 离开网络
sudo zerotier-cli leave NETWORK_ID

    网络管理¶

    // ZeroTier Central API 示例
const networkId = 'your_network_id';
const apiToken = 'your_api_token';

// 获取网络信息
fetch(`https://my.zerotier.com/api/network/${networkId}`, {
    headers: {
        'Authorization': `Bearer ${apiToken}`

// 授权成员

    2.4 WireGuard VPN¶

    WireGuard是现代VPN协议，性能优异且配置简单。

    服务端配置¶

    # /etc/wireguard/wg0.conf
[Interface]
PrivateKey = SERVER_PRIVATE_KEY
Address = 10.0.0.1/24
ListenPort = 51820
PostUp = iptables -A FORWARD -i %i -j ACCEPT; iptables -A FORWARD -o %i -j ACCEPT; iptables -t nat -A POSTROUTING -o eth0 -j MASQUERADE
PostDown = iptables -D FORWARD -i %i -j ACCEPT; iptables -D FORWARD -o %i -j ACCEPT; iptables -t nat -D POSTROUTING -o eth0 -j MASQUERADE

# 客户端1
[Peer]
PublicKey = CLIENT1_PUBLIC_KEY
AllowedIPs = 10.0.0.2/32

# 客户端2
[Peer]
PublicKey = CLIENT2_PUBLIC_KEY
AllowedIPs = 10.0.0.3/32

    客户端配置¶

    # client.conf
[Interface]

[Peer]

Endpoint = SERVER_IP:51820

    管理脚本¶

    #!/bin/bash
# wireguard-manager.sh

# 生成密钥对
generate_keys() {
    private_key=$(wg genkey)
    public_key=$(echo "$private_key" | wg pubkey)
    echo "Private Key: $private_key"
    echo "Public Key: $public_key"
}

# 启动WireGuard
start_wg() {
    sudo wg-quick up wg0
    echo "WireGuard started"
}

# 停止WireGuard
stop_wg() {
    sudo wg-quick down wg0
    echo "WireGuard stopped"
}

# 查看状态
status_wg() {
    sudo wg show
}

case "$1" in
    generate)
        generate_keys
        ;;
    start)
        start_wg
        ;;
    stop)
        stop_wg
        ;;
    status)
        status_wg
        ;;
    *)
        echo "Usage: $0 {generate|start|stop|status}"
        exit 1
        ;;
esac

    3. 自建穿透服务¶

    3.1 简单HTTP隧道¶

    # tunnel_server.py
import asyncio
import websockets
import json
from aiohttp import web, ClientSession
import logging

class TunnelServer:
    def __init__(self):
        self.tunnels = {}
        self.clients = {}

    async def handle_websocket(self, websocket, path):
        """处理WebSocket连接"""
        try:
            async for message in websocket:
                data = json.loads(message)
                await self.process_message(websocket, data)
        except websockets.exceptions.ConnectionClosed:
            await self.cleanup_client(websocket)
        except Exception as e:
            logging.error(f"WebSocket error: {e}")

    async def process_message(self, websocket, data):
        """处理客户端消息"""
        msg_type = data.get('type')

        if msg_type == 'register':
            await self.register_tunnel(websocket, data)
        elif msg_type == 'response':
            await self.forward_response(data)

    async def register_tunnel(self, websocket, data):
        """注册隧道"""
        tunnel_id = data.get('tunnel_id')
        subdomain = data.get('subdomain')

        self.tunnels[subdomain] = {
            'tunnel_id': tunnel_id,
            'websocket': websocket,
            'local_port': data.get('local_port')
        }

        self.clients[websocket] = {
            'tunnel_id': tunnel_id,
            'subdomain': subdomain
        }

        await websocket.send(json.dumps({
            'type': 'registered',
            'subdomain': subdomain,
            'url': f'https://{subdomain}.tunnel.example.com'
        }))

    async def handle_http_request(self, request):
        """处理HTTP请求"""
        host = request.headers.get('Host', '')
        subdomain = host.split('.')[0]

        if subdomain not in self.tunnels:
            return web.Response(text='Tunnel not found', status=404)

        tunnel = self.tunnels[subdomain]

        # 转发请求到客户端
        request_data = {
            'type': 'request',
            'method': request.method,
            'path': request.path_qs,
            'headers': dict(request.headers),
            'body': await request.read()
        }

        try:
            await tunnel['websocket'].send(json.dumps(request_data))
            # 等待响应 (实际实现中需要更复杂的响应匹配机制)
            return web.Response(text='Request forwarded', status=200)
        except Exception as e:
            return web.Response(text='Tunnel error', status=500)

    async def cleanup_client(self, websocket):
        """清理断开的客户端"""
        if websocket in self.clients:
            client_info = self.clients[websocket]
            subdomain = client_info['subdomain']

            if subdomain in self.tunnels:
                del self.tunnels[subdomain]
            del self.clients[websocket]

# 启动服务器
async def main():
    server = TunnelServer()

    # WebSocket服务器
    ws_server = websockets.serve(
        server.handle_websocket,
        'localhost',
        8765
    )

    # HTTP服务器
    app = web.Application()
    app.router.add_route('*', '/{path:.*}', server.handle_http_request)

    http_server = web.AppRunner(app)
    await http_server.setup()
    site = web.TCPSite(http_server, 'localhost', 8080)
    await site.start()

    await asyncio.gather(ws_server)

if __name__ == '__main__':
    asyncio.run(main())

    3.2 隧道客户端¶

    # tunnel_client.py
import asyncio
import websockets
import json
import aiohttp
from aiohttp import web
import uuid

class TunnelClient:
    def __init__(self, server_url, local_port, subdomain=None):
        self.server_url = server_url
        self.local_port = local_port
        self.subdomain = subdomain or str(uuid.uuid4())[:8]
        self.tunnel_id = str(uuid.uuid4())

    async def connect(self):
        """连接到隧道服务器"""
        uri = f"ws://{self.server_url}:8765"

        async with websockets.connect(uri) as websocket:
            # 注册隧道
            await self.register_tunnel(websocket)

            # 处理服务器消息
            async for message in websocket:
                data = json.loads(message)
                await self.handle_message(data)

    async def register_tunnel(self, websocket):
        """注册隧道"""
        register_data = {
            'type': 'register',
            'tunnel_id': self.tunnel_id,
            'subdomain': self.subdomain,
            'local_port': self.local_port
        }

        await websocket.send(json.dumps(register_data))

    async def handle_message(self, data):
        """处理服务器消息"""
        msg_type = data.get('type')

        if msg_type == 'registered':
            print(f"Tunnel registered: {data['url']}")
        elif msg_type == 'request':

    7.2 日志分析¶

    # log_analyzer.py
import re
from collections import Counter
from datetime import datetime, timedelta

class TunnelLogAnalyzer:
    def __init__(self, log_file):
        self.log_file = log_file

    def analyze_connections(self):
        """分析连接统计"""
        connections = []

        with open(self.log_file, 'r') as f:
            for line in f:
                # 解析连接日志
                match = re.search(r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}).*new connection from (\d+\.\d+\.\d+\.\d+)', line)
                if match:
                    timestamp = datetime.strptime(match.group(1), '%Y-%m-%d %H:%M:%S')
                    ip = match.group(2)
                    connections.append((timestamp, ip))

        return connections

    def get_top_ips(self, limit=10):
        """获取访问最多的IP"""
        connections = self.analyze_connections()
        ips = [conn[1] for conn in connections]
        return Counter(ips).most_common(limit)

    def get_hourly_stats(self):
        """获取每小时连接统计"""
        connections = self.analyze_connections()
        hourly = Counter()

        for timestamp, _ in connections:
            hour_key = timestamp.strftime('%Y-%m-%d %H:00')
            hourly[hour_key] += 1

        return dict(hourly)

# 使用示例
analyzer = TunnelLogAnalyzer('/var/log/frps.log')
print("Top IPs:", analyzer.get_top_ips())
print("Hourly stats:", analyzer.get_hourly_stats())
