    1. TUS 协议简介¶

    TUS (Tus Resumable Upload Protocol) 是一个开放的协议，用于可恢复的文件上传。它解决了网络不稳定环境下大文件上传中断的问题，支持断点续传功能。

    1.1 核心特性¶

    • 可恢复性: 支持断点续传，网络中断后可继续上传
    • 跨平台: 支持多种编程语言和平台
    • HTTP兼容: 基于标准HTTP协议
    • 扩展性: 支持多种扩展功能
    • 高效性: 减少重复上传，节省带宽

    1.2 协议版本¶

    • TUS 1.0.0: 当前稳定版本
    • TUS 1.1.0: 开发中的下一版本

    2. 协议工作流程¶

    2.1 基本流程¶

    1. 创建上传 (POST)
2. 获取上传信息 (HEAD)
3. 上传数据块 (PATCH)
4. 完成上传

    2.2 HTTP 方法说明¶

    方法 用途 必需
    POST 创建新的上传 是
    HEAD 获取上传偏移量 是
    PATCH 上传数据块 是
    DELETE 删除上传 可选
    OPTIONS 获取服务器能力 可选

    3. 核心扩展¶

    3.1 Creation 扩展¶

    用于创建新的上传会话。

    POST /files HTTP/1.1
Host: tus.example.org
Content-Length: 0
Upload-Length: 11821906
Tus-Resumable: 1.0.0

HTTP/1.1 201 Created
Location: https://tus.example.org/files/24e533e02ec3bc40c387f1a0e460e216
Tus-Resumable: 1.0.0

    3.2 Head 请求¶

    获取当前上传状态。

    HEAD /files/24e533e02ec3bc40c387f1a0e460e216 HTTP/1.1
Host: tus.example.org
Tus-Resumable: 1.0.0

HTTP/1.1 200 OK
Upload-Offset: 5242880
Upload-Length: 11821906
Tus-Resumable: 1.0.0

    3.3 Patch 请求¶

    上传数据块。

    PATCH /files/24e533e02ec3bc40c387f1a0e460e216 HTTP/1.1
Host: tus.example.org
Content-Type: application/offset+octet-stream
Content-Length: 1048576
Upload-Offset: 5242880
Tus-Resumable: 1.0.0

[1048576 bytes]

HTTP/1.1 204 No Content
Upload-Offset: 6291456
Tus-Resumable: 1.0.0

    4. 客户端实现¶

    4.1 JavaScript 客户端¶

    class TusClient {
    constructor(file, options = {}) {
        this.file = file;
        this.endpoint = options.endpoint || '/files/';
        this.chunkSize = options.chunkSize || 1024 * 1024; // 1MB
        this.uploadUrl = null;
        this.offset = 0;
        this.onProgress = options.onProgress || (() => {});
        this.onSuccess = options.onSuccess || (() => {});
        this.onError = options.onError || (() => {});
    }

    async upload() {
        try {
            // 创建上传
            await this.createUpload();

            // 检查现有进度
            await this.getOffset();

            // 上传文件块
            await this.uploadChunks();

            this.onSuccess(this.uploadUrl);
        } catch (error) {
            this.onError(error);
        }
    }

    async createUpload() {
        const response = await fetch(this.endpoint, {
            method: 'POST',
            headers: {
                'Tus-Resumable': '1.0.0',
                'Upload-Length': this.file.size.toString(),
                'Upload-Metadata': this.encodeMetadata({
                    filename: this.file.name,
                    filetype: this.file.type
                })
            }
        });

        if (!response.ok) {
            throw new Error(`创建上传失败: ${response.status}`);
        }

        this.uploadUrl = response.headers.get('Location');
        if (!this.uploadUrl) {
            throw new Error('服务器未返回上传URL');
        }
    }

    async getOffset() {
        const response = await fetch(this.uploadUrl, {
            method: 'HEAD',
            headers: {
                'Tus-Resumable': '1.0.0'
            }
        });

        if (!response.ok) {
            throw new Error(`获取偏移量失败: ${response.status}`);
        }

        this.offset = parseInt(response.headers.get('Upload-Offset') || '0');
    }

    async uploadChunks() {
        while (this.offset < this.file.size) {
            const chunk = this.file.slice(this.offset, this.offset + this.chunkSize);

            const response = await fetch(this.uploadUrl, {
                method: 'PATCH',
                headers: {
                    'Tus-Resumable': '1.0.0',
                    'Content-Type': 'application/offset+octet-stream',
                    'Upload-Offset': this.offset.toString()
                },
                body: chunk
            });

            if (!response.ok) {
                throw new Error(`上传块失败: ${response.status}`);
            }

            this.offset = parseInt(response.headers.get('Upload-Offset'));
            this.onProgress(this.offset / this.file.size);
        }
    }

    encodeMetadata(metadata) {
        return Object.entries(metadata)
            .map(([key, value]) => `${key} ${btoa(value)}`)
            .join(',');
    }

    async resume() {
        if (!this.uploadUrl) {
            throw new Error('没有可恢复的上传');
        }

        await this.getOffset();
        await this.uploadChunks();
        this.onSuccess(this.uploadUrl);
    }
}

// 使用示例
const fileInput = document.getElementById('file-input');
fileInput.addEventListener('change', (event) => {
    const file = event.target.files[0];
    if (!file) return;

    const uploader = new TusClient(file, {

        },

            console.log(`上传完成: ${url}`);
        },

            console.error('上传失败:', error);
        }
    });

    uploader.upload();
});

    4.2 Python 客户端¶

    import requests
import os
import base64
from typing import Optional, Callable

class TusClient:
    def __init__(self, file_path: str, endpoint: str, chunk_size: int = 1024*1024):
        self.file_path = file_path
        self.endpoint = endpoint
        self.chunk_size = chunk_size
        self.upload_url: Optional[str] = None
        self.offset = 0
        self.file_size = os.path.getsize(file_path)

    def upload(self,
               on_progress: Optional[Callable[[float], None]] = None,
               on_success: Optional[Callable[[str], None]] = None,
               on_error: Optional[Callable[[Exception], None]] = None):
        try:
            # 创建上传
            self._create_upload()

            # 获取当前偏移量
            self._get_offset()

            # 上传文件块
            self._upload_chunks(on_progress)

            if on_success:
                on_success(self.upload_url)

        except Exception as e:
            if on_error:
                on_error(e)
            else:
                raise

    def _create_upload(self):
        headers = {
            'Tus-Resumable': '1.0.0',
            'Upload-Length': str(self.file_size),
            'Upload-Metadata': self._encode_metadata({
                'filename': os.path.basename(self.file_path),
                'filetype': 'application/octet-stream'
            })
        }

        response = requests.post(self.endpoint, headers=headers)
        response.raise_for_status()

        self.upload_url = response.headers.get('Location')
        if not self.upload_url:
            raise ValueError('服务器未返回上传URL')

    def _get_offset(self):
        headers = {'Tus-Resumable': '1.0.0'}
        response = requests.head(self.upload_url, headers=headers)
        response.raise_for_status()

        self.offset = int(response.headers.get('Upload-Offset', 0))

    def _upload_chunks(self, on_progress: Optional[Callable[[float], None]] = None):
        with open(self.file_path, 'rb') as f:
            f.seek(self.offset)

            while self.offset < self.file_size:
                chunk = f.read(self.chunk_size)
                if not chunk:
                    break

                headers = {
                    'Tus-Resumable': '1.0.0',
                    'Content-Type': 'application/offset+octet-stream',
                    'Upload-Offset': str(self.offset)
                }

                response = requests.patch(
                    self.upload_url,
                    headers=headers,
                    data=chunk
                )
                response.raise_for_status()

                self.offset = int(response.headers.get('Upload-Offset'))

                if on_progress:
                    progress = self.offset / self.file_size
                    on_progress(progress)

    def _encode_metadata(self, metadata: dict) -> str:
        encoded = []
        for key, value in metadata.items():
            encoded_value = base64.b64encode(value.encode()).decode()
            encoded.append(f'{key} {encoded_value}')
        return ','.join(encoded)

    def resume(self):
        """恢复中断的上传"""
        if not self.upload_url:
            raise ValueError('没有可恢复的上传')

        self._get_offset()
        self._upload_chunks()

# 使用示例
def progress_callback(progress):
    print(f"上传进度: {progress * 100:.2f}%")

def success_callback(url):
    print(f"上传完成: {url}")

def error_callback(error):
    print(f"上传失败: {error}")

uploader = TusClient(
    file_path='/path/to/large/file.zip',
    endpoint='https://tusd.tusdemo.net/files/',
    chunk_size=2 * 1024 * 1024  # 2MB
)

uploader.upload(
    on_progress=progress_callback,
    on_success=success_callback,
    on_error=error_callback
)

    5.1 Node.js 服务端¶

    const express = require('express');
const multer = require('multer');
const fs = require('fs').promises;
const path = require('path');
const crypto = require('crypto');

class TusServer {
    constructor(uploadDir = './uploads') {
        this.uploadDir = uploadDir;
        this.app = express();
        this.setupMiddleware();
        this.setupRoutes();
    }

    setupMiddleware() {
        this.app.use(express.raw({ type: 'application/offset+octet-stream' }));

        // CORS 支持
        this.app.use((req, res, next) => {
            res.header('Access-Control-Allow-Origin', '*');
            res.header('Access-Control-Allow-Methods', 'POST, GET, HEAD, PATCH, DELETE, OPTIONS');
            res.header('Access-Control-Allow-Headers', 'Tus-Resumable, Upload-Length, Upload-Offset, Content-Type, Upload-Metadata');
            res.header('Access-Control-Expose-Headers', 'Tus-Resumable, Upload-Offset, Location');

            if (req.method === 'OPTIONS') {
                return res.sendStatus(200);
            }
            next();
        });

        // TUS 版本检查
        this.app.use((req, res, next) => {
            const tusResumable = req.headers['tus-resumable'];
            if (tusResumable && tusResumable !== '1.0.0') {
                return res.status(412).json({ error: '不支持的TUS版本' });
            }
            res.header('Tus-Resumable', '1.0.0');
            next();
        });
    }

    setupRoutes() {
        // 创建上传
        this.app.post('/files', async (req, res) => {
            try {
                const uploadLength = parseInt(req.headers['upload-length']);
                if (!uploadLength) {
                    return res.status(400).json({ error: '缺少Upload-Length头' });
                }

                const uploadId = crypto.randomUUID();
                const uploadPath = path.join(this.uploadDir, uploadId);

                // 创建上传信息文件
                const uploadInfo = {
                    id: uploadId,
                    size: uploadLength,
                    offset: 0,
                    metadata: this.parseMetadata(req.headers['upload-metadata']),
                    createdAt: new Date().toISOString()
                };

                await fs.writeFile(`${uploadPath}.info`, JSON.stringify(uploadInfo));
                await fs.writeFile(uploadPath, '');

                res.status(201)
                   .header('Location', `/files/${uploadId}`)
                   .send();

            } catch (error) {
                console.error('创建上传失败:', error);
                res.status(500).json({ error: '内部服务器错误' });
            }
        });

        // 获取上传信息
        this.app.head('/files/:id', async (req, res) => {
            try {
                const uploadId = req.params.id;
                const infoPath = path.join(this.uploadDir, `${uploadId}.info`);

                const infoData = await fs.readFile(infoPath, 'utf8');
                const uploadInfo = JSON.parse(infoData);

                res.header('Upload-Offset', uploadInfo.offset.toString())
                   .header('Upload-Length', uploadInfo.size.toString())
                   .send();

            } catch (error) {
                if (error.code === 'ENOENT') {
                    return res.status(404).json({ error: '上传不存在' });
                }
                console.error('获取上传信息失败:', error);
                res.status(500).json({ error: '内部服务器错误' });
            }
        });

        // 上传数据块
        this.app.patch('/files/:id', async (req, res) => {
            try {
                const uploadId = req.params.id;
                const uploadOffset = parseInt(req.headers['upload-offset']);
                const infoPath = path.join(this.uploadDir, `${uploadId}.info`);
                const uploadPath = path.join(this.uploadDir, uploadId);

                // 读取上传信息
                const infoData = await fs.readFile(infoPath, 'utf8');
                const uploadInfo = JSON.parse(infoData);

                // 验证偏移量
                if (uploadOffset !== uploadInfo.offset) {
                    return res.status(409).json({
                        error: `偏移量不匹配，期望: ${uploadInfo.offset}, 实际: ${uploadOffset}`
                    });
                }

                // 写入数据
                const fileHandle = await fs.open(uploadPath, 'r+');
                await fileHandle.write(req.body, 0, req.body.length, uploadOffset);
                await fileHandle.close();

                // 更新偏移量
                uploadInfo.offset += req.body.length;
                await fs.writeFile(infoPath, JSON.stringify(uploadInfo));

                res.status(204)
                   .header('Upload-Offset', uploadInfo.offset.toString())
                   .send();

            } catch (error) {
                console.error('上传数据块失败:', error);
                res.status(500).json({ error: '内部服务器错误' });
            }
        });

        // 删除上传
        this.app.delete('/files/:id', async (req, res) => {
            try {
                const uploadId = req.params.id;
                const infoPath = path.join(this.uploadDir, `${uploadId}.info`);
                const uploadPath = path.join(this.uploadDir, uploadId);

                await Promise.all([
                    fs.unlink(infoPath).catch(() => {}),
                    fs.unlink(uploadPath).catch(() => {})
                ]);

                res.status(204).send();

            } catch (error) {
                console.error('删除上传失败:', error);
                res.status(500).json({ error: '内部服务器错误' });
            }
        });
    }

    parseMetadata(metadataHeader) {
        if (!metadataHeader) return {};

        const metadata = {};
        const pairs = metadataHeader.split(',');

        for (const pair of pairs) {
            const [key, encodedValue] = pair.trim().split(' ');
            if (key && encodedValue) {
                try {
                    metadata[key] = Buffer.from(encodedValue, 'base64').toString();
                } catch (e) {
                    console.warn(`无法解析元数据: ${pair}`);
                }
            }
        }

        return metadata;
    }

    async start(port = 3000) {
        // 确保上传目录存在
        await fs.mkdir(this.uploadDir, { recursive: true });

        this.app.listen(port, () => {
            console.log(`TUS服务器启动在端口 ${port}`);
        });
    }
}

    6.1 并行上传¶

    class ParallelTusUploader {
    constructor(file, options = {}) {
        this.file = file;
        this.endpoint = options.endpoint;
        this.chunkSize = options.chunkSize || 1024 * 1024;
        this.parallelUploads = options.parallelUploads || 3;
        this.onProgress = options.onProgress || (() => {});
    }

    async upload() {
        const totalChunks = Math.ceil(this.file.size / this.chunkSize);
        const chunks = [];

        // 创建所有上传会话
        for (let i = 0; i < totalChunks; i++) {
            const start = i * this.chunkSize;
            const end = Math.min(start + this.chunkSize, this.file.size);
            const chunk = this.file.slice(start, end);

            chunks.push({
                index: i,
                start,
                end,
                data: chunk,
                uploaded: false
            });
        }

        // 并行上传
        const uploadPromises = [];
        for (let i = 0; i < this.parallelUploads; i++) {
            uploadPromises.push(this.uploadWorker(chunks));
        }

        await Promise.all(uploadPromises);

        // 合并文件块
        await this.mergeChunks(chunks);
    }

    async uploadWorker(chunks) {
        while (true) {
            const chunk = chunks.find(c => !c.uploaded);
            if (!chunk) break;

            chunk.uploaded = true;
            await this.uploadChunk(chunk);

            const uploadedCount = chunks.filter(c => c.uploaded).length;
            this.onProgress(uploadedCount / chunks.length);
        }
    }

    async uploadChunk(chunk) {
        const uploader = new TusClient(new File([chunk.data], `chunk_${chunk.index}`), {
            endpoint: this.endpoint
        });

        await uploader.upload();
        chunk.uploadUrl = uploader.uploadUrl;
    }

    async mergeChunks(chunks) {
        // 发送合并请求到服务器
        const chunkUrls = chunks.map(c => c.uploadUrl);

        const response = await fetch(`${this.endpoint}/merge`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                chunks: chunkUrls,
                filename: this.file.name
            })
        });

        if (!response.ok) {
            throw new Error('合并文件块失败');
        }

        return response.headers.get('Location');
    }
}

    class ResumeManager {
    constructor() {
        this.storageKey = 'tus_uploads';
    }

    saveUpload(fileId, uploadInfo) {
        const uploads = this.getUploads();
        uploads[fileId] = {
            ...uploadInfo,
            lastModified: Date.now()
        };
        localStorage.setItem(this.storageKey, JSON.stringify(uploads));
    }

    getUpload(fileId) {
        const uploads = this.getUploads();
        return uploads[fileId];
    }

    getUploads() {
        const data = localStorage.getItem(this.storageKey);
        return data ? JSON.parse(data) : {};
    }

    removeUpload(fileId) {
        const uploads = this.getUploads();
        delete uploads[fileId];
        localStorage.setItem(this.storageKey, JSON.stringify(uploads));
    }

    cleanExpiredUploads(maxAge = 7 * 24 * 60 * 60 * 1000) { // 7天
        const uploads = this.getUploads();
        const now = Date.now();

        Object.keys(uploads).forEach(fileId => {
            if (now - uploads[fileId].lastModified > maxAge) {
                delete uploads[fileId];
            }
        });

        localStorage.setItem(this.storageKey, JSON.stringify(uploads));
    }

    async resumeUpload(fileId, file) {
        const uploadInfo = this.getUpload(fileId);
        if (!uploadInfo) {
            throw new Error('没有找到可恢复的上传');
        }

        const uploader = new TusClient(file, {
            endpoint: uploadInfo.endpoint
        });

        uploader.uploadUrl = uploadInfo.uploadUrl;
        await uploader.resume();

        this.removeUpload(fileId);
        return uploader.uploadUrl;
    }
}
