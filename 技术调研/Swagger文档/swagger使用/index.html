    一、Swagger 在 Web 开发中的重要性¶

    1.1 API 文档化¶

    • 自动生成文档：减少手动编写文档的工作量
    • 实时同步：代码变更后文档自动更新，避免文档与代码不一致
    • 标准化：遵循 OpenAPI 规范，便于团队协作

    1.2 前后端协作¶

    • 接口契约：前后端开发可以并行进行
    • Mock 数据：前端可基于 Swagger 生成 Mock 服务
    • 减少沟通成本：清晰的接口定义减少歧义

    1.3 API 测试¶

    • 在线调试：通过 Swagger UI 直接测试接口
    • 参数验证：自动校验请求参数格式
    • 响应预览：直观展示接口返回结果

    二、主流技术方案对比¶

    2.1 OpenAPI 规范版本¶

    版本 特点 推荐场景
    OpenAPI 2.0 (Swagger) 广泛支持，工具生态成熟 传统项目兼容
    OpenAPI 3.0 更强表达能力，支持 oneOf/anyOf 新项目首选
    OpenAPI 3.1 完全兼容 JSON Schema 复杂数据模型

    FastAPI（推荐，内置 OpenAPI 支持）¶
    from fastapi import FastAPI
from pydantic import BaseModel

app = FastAPI(
    title="Dataset API",
    description="数据集管理服务",
    version="1.0.0"
)

class Dataset(BaseModel):
    id: int
    name: str
    description: str | None = None

@app.get("/datasets", response_model=list[Dataset], tags=["dataset"])
async def get_datasets(page: int = 1, size: int = 10):
    """
    获取数据集列表

    - **page**: 页码，默认 1
    - **size**: 每页数量，默认 10
    """
    return []

@app.post("/datasets", response_model=Dataset, tags=["dataset"])
async def create_dataset(dataset: Dataset):
    """创建新数据集"""
    return dataset

# 自动生成文档访问地址:
# Swagger UI: http://localhost:8000/docs
# ReDoc: http://localhost:8000/redoc
# OpenAPI JSON: http://localhost:8000/openapi.json
    Flask + Flask-RESTX¶
    from flask import Flask
from flask_restx import Api, Resource, fields

app = Flask(__name__)
api = Api(app,
    title="Dataset API",
    version="1.0.0",
    description="数据集管理服务"
)

ns = api.namespace('datasets', description='数据集操作')

dataset_model = api.model('Dataset', {
    'id': fields.Integer(required=True, description='数据集ID'),
    'name': fields.String(required=True, description='数据集名称'),
    'description': fields.String(description='描述')
})

@ns.route('/')
class DatasetList(Resource):
    @ns.doc('list_datasets')
    @ns.marshal_list_with(dataset_model)
    def get(self):
        """获取所有数据集"""
        return []

    @ns.doc('create_dataset')
    @ns.expect(dataset_model)
    @ns.marshal_with(dataset_model, code=201)
    def post(self):
        """创建数据集"""
        return api.payload, 201

# Swagger UI: http://localhost:5000/

    Rust¶

    Actix-web + utoipa¶

    Cargo.toml 依赖：

    [dependencies]
actix-web = "4"
serde = { version = "1", features = ["derive"] }
serde_json = "1"
utoipa = { version = "4", features = ["actix_extras"] }
utoipa-swagger-ui = { version = "6", features = ["actix-web"] }

    代码示例：

    use actix_web::{web, App, HttpServer, HttpResponse};
use serde::{Deserialize, Serialize};
use utoipa::{OpenApi, ToSchema};
use utoipa_swagger_ui::SwaggerUi;

#[derive(Serialize, Deserialize, ToSchema)]
struct Dataset {
    /// 数据集ID
    id: i64,
    /// 数据集名称
    name: String,
    /// 数据集描述
    description: Option<String>,
}

#[derive(Deserialize, ToSchema)]
struct QueryParams {
    /// 页码
    page: Option<i32>,
    /// 每页数量
    size: Option<i32>,
}

/// 获取数据集列表
#[utoipa::path(
    get,
    path = "/api/datasets",
    params(
        ("page" = Option<i32>, Query, description = "页码"),
        ("size" = Option<i32>, Query, description = "每页数量")
    ),
    responses(
        (status = 200, description = "成功", body = Vec<Dataset>),
        (status = 500, description = "服务器错误")
    ),
    tag = "dataset"
)]
async fn get_datasets(query: web::Query<QueryParams>) -> HttpResponse {
    HttpResponse::Ok().json(Vec::<Dataset>::new())
}

/// 创建数据集
#[utoipa::path(
    post,
    path = "/api/datasets",
    request_body = Dataset,
    responses(
        (status = 201, description = "创建成功", body = Dataset),
        (status = 400, description = "参数错误")
    ),
    tag = "dataset"
)]
async fn create_dataset(dataset: web::Json<Dataset>) -> HttpResponse {
    HttpResponse::Created().json(dataset.into_inner())
}

#[derive(OpenApi)]
#[openapi(
    paths(get_datasets, create_dataset),
    components(schemas(Dataset, QueryParams)),
    tags(
        (name = "dataset", description = "数据集管理接口")
    ),
    info(
        title = "Dataset API",
        version = "1.0.0",
        description = "数据集管理服务"
    )
)]
struct ApiDoc;

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| {
        App::new()
            .service(
                SwaggerUi::new("/swagger-ui/{_:.*}")
                    .url("/api-docs/openapi.json", ApiDoc::openapi())
            )
            .route("/api/datasets", web::get().to(get_datasets))
            .route("/api/datasets", web::post().to(create_dataset))
    })
    .bind("127.0.0.1:8080")?
    .run()
    .await
}
    Axum + utoipa¶

    Cargo.toml 依赖：

    [dependencies]
axum = "0.7"
tokio = { version = "1", features = ["full"] }
serde = { version = "1", features = ["derive"] }
utoipa = { version = "4", features = ["axum_extras"] }
utoipa-swagger-ui = { version = "6", features = ["axum"] }

    代码示例：

    use axum::{routing::get, Router, Json};
use serde::{Deserialize, Serialize};
use utoipa::{OpenApi, ToSchema};
use utoipa_swagger_ui::SwaggerUi;

#[derive(Serialize, Deserialize, ToSchema)]
struct Dataset {
    id: i64,
    name: String,
}

#[utoipa::path(
    get,
    path = "/api/datasets",
    responses(
        (status = 200, description = "成功", body = Vec<Dataset>)
    ),
    tag = "dataset"
)]
async fn list_datasets() -> Json<Vec<Dataset>> {
    Json(vec![])
}

#[derive(OpenApi)]
#[openapi(paths(list_datasets), components(schemas(Dataset)))]
struct ApiDoc;

#[tokio::main]
async fn main() {
    let app = Router::new()
        .merge(SwaggerUi::new("/swagger-ui")
            .url("/api-docs/openapi.json", ApiDoc::openapi()))
        .route("/api/datasets", get(list_datasets));

    let listener = tokio::net::TcpListener::bind("0.0.0.0:8080").await.unwrap();
    axum::serve(listener, app).await.unwrap();
}

    三、Swagger 工具链详解¶

    3.1 Swagger Editor¶

    用途：在线编辑和预览 OpenAPI 文档

    # Docker 部署
docker run -d -p 8081:8080 swaggerapi/swagger-editor

# 访问地址
http://localhost:8081

    主要功能：

    • YAML/JSON 格式编辑
    • 实时语法校验
    • 即时预览渲染
    • 导出多种格式

    3.2 Swagger UI¶

    https://swagger.io/

    用途：渲染和交互式展示 API 文档

    # Docker 部署
docker run --rm -it -p 8082:8080 \
  -e SWAGGER_JSON=/app/swagger.json \
  -v /path/to/swagger.json:/app/swagger.json \
  swaggerapi/swagger-ui

# 或指定远程 URL (注意跨域问题)
docker run -d -p 8082:8080 \
  -e SWAGGER_JSON_URL=http://your-api/swagger.json \
  swaggerapi/swagger-ui

# ENV
SWAGGER_JSON=/swagger.json
SWAGGER_JSON_URL=http://127.0.0.1:8888/v1/swagger/json

    Go 项目集成示例：

    import (
    "github.com/gin-gonic/gin"
    swaggerFiles "github.com/swaggo/files"
    ginSwagger "github.com/swaggo/gin-swagger"
    _ "your-project/docs" // 导入生成的 docs
)

func main() {
    r := gin.Default()
    // Swagger 文档路由
    r.GET("/swagger/*any", ginSwagger.WrapHandler(swaggerFiles.Handler))
    r.Run(":8080")
}

    3.3 Swagger Codegen / OpenAPI Generator¶

    用途：根据 OpenAPI 规范生成代码

    # 安装 OpenAPI Generator (推荐)
npm install @openapitools/openapi-generator-cli -g

# 生成 Go 客户端
openapi-generator-cli generate \
  -i swagger.yaml \
  -g go \
  -o ./generated/go-client

# 生成 TypeScript 客户端
openapi-generator-cli generate \
  -i swagger.yaml \
  -g typescript-axios \
  -o ./generated/ts-client

# 生成服务端框架
openapi-generator-cli generate \
  -i swagger.yaml \
  -g go-gin-server \
  -o ./generated/go-server

    支持的生成器：

    类型 语言/框架
    客户端 go, java, TypeScript-axios, python, dart
    服务端 go-gin-server, spring, nodejs-express
    文档 HTML, markdown, asciidoc

    四、完整工作流¶

    4.1 代码优先 (Code First)¶

    代码注释 → swag init → swagger.json → Swagger UI

    # Go 项目示例
# 1. 添加注释
# 2. 生成文档
swag init -g main.go -o ./docs

# 3. 启动服务，访问 /swagger/index.html

    4.2 设计优先 (Design First)¶

    Swagger Editor → swagger.yaml → OpenAPI Generator → 代码框架

    # 1. 在 Swagger Editor 中设计 API
# 2. 导出 swagger.yaml
# 3. 生成代码
openapi-generator-cli generate -i swagger.yaml -g go-gin-server -o ./server

services:

    image: swaggerapi/swagger-editor
    ports:

    restart: unless-stopped

  swagger-ui:
    image: swaggerapi/swagger-ui
    ports:
      - "8082:8080"
    environment:
      - API_URL=/swagger.json
    volumes:
      - ./docs/swagger.json:/usr/share/nginx/html/swagger.json
    restart: unless-stopped
    depends_on:
      - api-server

  api-server:
    build: .
    ports:
      - "8080:8080"
    restart: unless-stopped
