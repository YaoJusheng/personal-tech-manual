    1.启动¶

    sqlite3 mydb.db
sqlite3 -readonly mydb.db          # 只读模式打开
sqlite3 -header -csv mydb.db       # CSV输出模式

    连接选项 (3.38+)¶

    • URI 文件名支持：启用 uri 功能后可使用参数化连接
    sqlite3 "file:mydb.db?mode=ro&immutable=1"  # 只读+不可变模式
sqlite3 "file:memdb1?mode=memory&cache=shared"  # 命名共享内存数据库

    2.核心元命令¶

    1. 基础操作¶

    • .help：显示所有可用的元命令及用法2。
    • .exit / .quit：退出 SQLite 命令行1。
    • .open [文件]：打开或创建数据库文件（默认创建临时数据库）4。
    • .database：列出当前附加的数据库（包括主库和临时库）2。

    2. 数据库结构修改 (3.35+)¶

    • .tables [模式]：列出所有表名（支持通配符，如 .tables user*）1。
    • .schema [表名]：显示表的创建语句（若指定表名，则仅显示该表结构）1。
    • .indexes [表名]：列出所有索引（可选指定表名）2。
    • ALTER TABLE DROP COLUMN：删除列（3.35.0+ 支持）
    ALTER TABLE users DROP COLUMN age;
ALTER TABLE users DROP COLUMN IF EXISTS email;  -- 安全删除
    • ALTER TABLE RENAME COLUMN：重命名列
    ALTER TABLE users RENAME COLUMN name TO username;

    3. 数据导入 / 导出¶

    • .import 文件 表名：从 CSV/TSV 文件导入数据（需先通过 .mode csv 设置格式）
    .mode csv
.import data.csv users
    • .export 文件：将查询结果导出到文件（需先执行 .output 重定向）
    .output result.csv
SELECT * FROM users;

    4. 输出格式控制¶

    • .mode 模式：设置输出模式（如 csv、column、html、line 等）。
    .mode column  -- 列对齐模式
.mode csv     -- CSV 格式

    • .headers on|off：显示或隐藏列头。

    • .width 宽度...：设置列宽（0 表示自适应）。

    5. 高级管理¶

    • .backup 文件：在线备份数据库（需 SQLite 3.8.5+）。
    .backup backup.db

    • .vacuum：清理数据库文件，回收未使用空间1。

    • .analyze：收集统计信息以优化查询1。

    3.事务处理¶

    1. 数据定义语言（DDL）¶

    • 创建表
    CREATE TABLE users (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  age INTEGER
);
    • 添加字段（需 SQLite 3.31.0+）：
    ALTER TABLE users ADD COLUMN email TEXT NOT NULL;
    • 删除表
    DROP TABLE users;

    2. 数据操作语言（DML）¶

    • 插入数据
    INSERT INTO users (name, age) VALUES ('Alice', 30);
    • 查询数据
    SELECT * FROM users WHERE age > 25;
    • 更新数据
    UPDATE users SET age = 31 WHERE name = 'Alice';
    • 删除数据
    DELETE FROM users WHERE age < 18;

    3. 事务处理¶

    • RETURNING 子句 (3.35+)：DML操作后返回修改的数据
    -- 插入并返回自动生成的ID
INSERT INTO users (name, age) VALUES ('Alice', 30)
RETURNING id, name;

-- 更新并返回修改前/后的值
UPDATE users SET age = age + 1 WHERE id = 1
RETURNING id, name, age AS new_age;

-- 删除并返回被删除的数据
DELETE FROM users WHERE age < 18
RETURNING id, name;
    • 开启事务
    BEGIN;  -- 或 BEGIN TRANSACTION;
    • 提交事务
    COMMIT;
    • 回滚事务
    ROLLBACK;
    • 保存点（Savepoint）
    SAVEPOINT sp1;  -- 设置保存点
ROLLBACK TO sp1;  -- 回滚到保存点
RELEASE sp1;     -- 释放保存点

    4.性能优化¶

    1. 索引管理¶

    • 创建索引
    CREATE INDEX idx_users_age ON users(age);
    • 删除索引
    DROP INDEX idx_users_age;

    2. 查询优化¶

    • 解释执行计划
    EXPLAIN QUERY PLAN SELECT * FROM users WHERE age > 25;
    • 查询计划可视化（3.42+ CLI）
    .expert  -- 获取索引建议
SELECT * FROM users WHERE email = 'alice@example.com';
    • 统计分析
    .scanstats on  -- 开启扫描统计
SELECT * FROM users;  -- 执行查询后显示性能数据
PRAGMA optimize;      -- 自动优化查询计划（推荐在schema变更后执行）

    3. 内置数学函数 (3.35+)¶

    无需加载扩展，直接可用的数学函数：

    SELECT
  abs(-42),           -- 绝对值
  ceil(4.2),          -- 向上取整
  floor(4.8),         -- 向下取整
  round(3.14159, 2),  -- 四舍五入
  power(2, 10),       -- 幂运算
  sqrt(16),           -- 平方根
  log10(100),         -- 常用对数
  ln(2.718),          -- 自然对数
  mod(17, 5),         -- 取模
  pi();               -- π 值

    三角函数：sin(), cos(), tan(), asin(), acos(), atan(), atan2()

    3. 事务性能优化¶

    • 批量插入
    BEGIN;
INSERT INTO users (name, age) VALUES ('Bob', 28);
INSERT INTO users (name, age) VALUES ('Charlie', 32);
COMMIT;

    4. 增强的 PRAGMA 调优¶

    • 内存映射 I/O (3.7.17+)：对大数据库可提升性能
    PRAGMA mmap_size = 268435456;  -- 256MB 内存映射
PRAGMA mmap_size = 0;          -- 禁用内存映射
    • 优化器控制 (3.9.0+)
    PRAGMA optimize;              -- 一次优化（推荐批处理脚本结束时）
PRAGMA analysis_limit = 400;  -- 限制ANALYZE采样行数
PRAGMA optimize = 0x10000;    -- 深度分析
    • 安全相关设置
    PRAGMA trusted_schema = OFF;  -- 禁用不可信扩展
PRAGMA cell_size_check = ON;  -- 启用单元格大小校验（调试）

    5. 数据库完整性检查¶

    PRAGMA integrity_check;          -- 快速检查
PRAGMA integrity_check(4);       -- 限制输出4行
PRAGMA foreign_key_check;        -- 外键约束检查
PRAGMA quick_check;              -- 快速完整性检查
    • 加密数据库（需 SQLite 加密扩展）：
    PRAGMA key = 'my_password';  -- 设置加密密钥
    • 加载扩展
    .load /path/to/extension.so

    5.实用技巧¶

    1. 批量执行 SQL 脚本¶

    • 从文件读取
    sqlite3 mydb.db < script.sql
    • 在命令行内执行
    echo "SELECT * FROM users;" | sqlite3 mydb.db
# e.g.
echo "ALTER TABLE user_log ADD COLUMN age INTEGER DEFAULT 0;" | sqlite3 example_edge.db

    2. 处理大型数据集¶

    • 分页查询
    SELECT * FROM users LIMIT 10 OFFSET 20;  -- 第 21-30 条记录

    3. 错误处理¶

    • 自动终止错误命令
    .bail on  -- 发生错误时停止后续命令

    4. 超时设置¶

    • 等待锁超时
    PRAGMA busy_timeout = 5000;  -- 5 秒后放弃等待锁

    6. 日志与持久化策略¶

    • WAL 模式（推荐大部分线上写入场景）
    PRAGMA journal_mode = WAL;          -- 启用写前日志
PRAGMA synchronous = NORMAL;        -- 平衡性能与安全（或 FULL 获取最强一致性）
PRAGMA wal_autocheckpoint = 1000;   -- 每 1000 页触发自动 checkpoint
    • WAL 支持多读单写并发，文件不会被 truncate，可定期执行 PRAGMA wal_checkpoint(TRUNCATE); 释放日志。

    • 对嵌入式设备，可结合 PRAGMA temp_store = MEMORY; 减少磁盘磨损。

    • 增量备份与在线恢复（3.27+）

    .backup --append backup_202402.db  -- 追加备份，减少全量复制时间
PRAGMA backup_step = 1000;         -- 大型 DB 分批拷贝，降低锁持有

    7. 结构约束与数据类型增强¶

    • STRICT 表（3.37+）：强类型校验，防止 TEXT/INTEGER 混用。
    CREATE TABLE users_strict (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  age INTEGER CHECK(age BETWEEN 0 AND 150)
) STRICT;
    • 生成列（3.31+）：避免重复存储，保证派生字段一致性。
    CREATE TABLE metrics (
  value INTEGER,
  doubled INTEGER GENERATED ALWAYS AS (value * 2) STORED
);
    • 唯一冲突处理：使用 ON CONFLICT/UPSERT 保持原子性。
    INSERT INTO users(id, name, age)
VALUES(1, 'Alice', 30)
ON CONFLICT(id) DO UPDATE SET age = excluded.age;

    8. JSON、全文检索与虚拟表¶

    • JSON1 扩展（内置于官方二进制）：
    SELECT json_extract(profile, '$.city') AS city
FROM users
WHERE json_type(profile, '$.tags[0]') = 'text';
    • json_each/json_tree 虚拟表：展开数组/对象以便 JOIN。
    SELECT u.name, tag.value
FROM users AS u,
     json_each(u.profile, '$.tags') AS tag;
    • JSON 聚合与生成函数 (3.38+)
    -- 将多行聚合为JSON数组
SELECT json_group_array(name) FROM users;

-- 将多行聚合为JSON对象
SELECT json_group_object(name, age) FROM users;

-- 构建JSON对象
INSERT INTO users (profile)
VALUES (json_object('city', 'Beijing', 'tags', json_array('a', 'b')));
    • JSONB 二进制格式 (3.45+)：更高性能的JSON存储
    SELECT jsonb('{"a": 1, "b": 2}');  -- 压缩为二进制格式
SELECT json(jsonb_data);            -- 转回文本格式

-- JSONB函数
SELECT jsonb_extract(profile, '$.city') FROM users;
SELECT jsonb_array(name, age) FROM users;
    • FTS5 全文索引：
    CREATE VIRTUAL TABLE docs USING fts5(title, body, tokenize = "unicode61 remove_diacritics 2");
SELECT title FROM docs WHERE docs MATCH 'sqlite NEAR/5 wal';
    • 其他虚拟表：rtree（空间索引）、csv（直接查询 CSV），可通过 .load 引入。

    9. 查询增强特性¶

    • 窗口函数（3.25+）
    SELECT name,
       age,
       AVG(age) OVER (PARTITION BY city) AS avg_city_age
FROM users;

    • 递归 CTE 与分层查询

    • MATERIALIZED / NOT MATERIALIZED 提示 (3.35+)：控制CTE物化行为

      -- 强制物化（适用于CTE被多次引用时）
WITH RECURSIVE ancestors(id, name) AS MATERIALIZED (
  SELECT id, name FROM users WHERE id = 1
  UNION ALL
  SELECT parent.id, parent.name FROM users AS parent
  JOIN ancestors ON parent.child_id = ancestors.id
)
SELECT * FROM ancestors;

-- 禁止物化（每次引用都重新执行）
WITH temp_stats AS NOT MATERIALIZED (
  SELECT count(*) as cnt FROM large_table
)
SELECT * FROM temp_stats WHERE cnt > 0;

    • 递归查询

      WITH RECURSIVE ancestors(id) AS (
  SELECT id FROM users WHERE id = 1
  UNION ALL
  SELECT parent.id FROM users AS parent
  JOIN ancestors ON parent.child_id = ancestors.id
)
SELECT * FROM ancestors;

    • 行值比较与批量 upsert：

    INSERT INTO inventory(sku, stock)
VALUES ('A-01', 10), ('A-02', 5)
ON CONFLICT(sku) DO UPDATE SET stock = stock + excluded.stock;
