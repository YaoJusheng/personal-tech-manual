    1.核心表结构设计¶

    1.1 数据集表 (datasets)¶

    mysql¶

    CREATE TABLE datasets (
    id VARCHAR(64) PRIMARY KEY COMMENT '数据集ID',
    name VARCHAR(255) NOT NULL COMMENT '数据集名称',
    description TEXT COMMENT '数据集描述',
    modality ENUM('image', 'text', 'audio', 'video', '3d', 'multimodal') NOT NULL COMMENT '数据类型',
    version VARCHAR(32) NOT NULL DEFAULT '1.0.0' COMMENT '版本号',
    status ENUM('draft', 'active', 'archived', 'deleted') DEFAULT 'draft' COMMENT '状态',

    -- 存储配置
    storage_policy ENUM('hot', 'warm', 'cold', 'auto') DEFAULT 'auto' COMMENT '存储策略',
    retention_days INT DEFAULT 365 COMMENT '保留天数',

    -- 权限控制
    visibility ENUM('private', 'team', 'public') DEFAULT 'private' COMMENT '可见性',
    created_by VARCHAR(64) NOT NULL COMMENT '创建者',
    owner_id VARCHAR(64) NOT NULL COMMENT '所有者',

    -- 统计信息
    file_count INT DEFAULT 0 COMMENT '文件数量',
    total_size BIGINT DEFAULT 0 COMMENT '总大小(字节)',

    -- 扩展元数据
    tags JSON COMMENT '标签数组',
    metadata JSON COMMENT '扩展元数据',

    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP NULL,

    -- 索引
    INDEX idx_name (name),
    INDEX idx_modality (modality),
    INDEX idx_created_by (created_by),
    INDEX idx_created_at (created_at),
    INDEX idx_status (status),
    UNIQUE INDEX uk_name_version (name, version)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='数据集表';

    1.2 文件表 (files)¶

    mysql¶

    CREATE TABLE files (
    id VARCHAR(64) PRIMARY KEY COMMENT '文件ID',
    dataset_id VARCHAR(64) NOT NULL COMMENT '数据集ID',
    name VARCHAR(512) NOT NULL COMMENT '文件名',
    original_name VARCHAR(512) COMMENT '原始文件名',
    path VARCHAR(1024) NOT NULL COMMENT '存储路径',

    -- 文件基本信息
    size BIGINT NOT NULL COMMENT '文件大小(字节)',
    file_hash VARCHAR(128) NOT NULL COMMENT '文件哈希(SHA256)',
    mime_type VARCHAR(128) COMMENT 'MIME类型',
    extension VARCHAR(32) COMMENT '文件扩展名',

    -- 存储信息
    storage_tier ENUM('hot', 'warm', 'cold') DEFAULT 'warm' COMMENT '存储层级',
    storage_provider ENUM('minio', 's3', 'oss', 'cos') DEFAULT 'minio' COMMENT '存储提供商',
    storage_bucket VARCHAR(255) NOT NULL COMMENT '存储桶',
    storage_key VARCHAR(1024) NOT NULL COMMENT '存储键',

    -- 多模态特定字段
    modality ENUM('image', 'text', 'audio', 'video', '3d') NOT NULL COMMENT '数据类型',
    duration INT COMMENT '时长(秒)-音视频专用',
    width INT COMMENT '宽度-图像视频专用',
    height INT COMMENT '高度-图像视频专用',
    frame_rate DECIMAL(5,2) COMMENT '帧率-视频专用',
    sample_rate INT COMMENT '采样率-音频专用',
    channels INT COMMENT '声道数-音频专用',

    -- 状态信息
    status ENUM('uploading', 'processing', 'ready', 'error', 'deleted') DEFAULT 'uploading',
    upload_progress INT DEFAULT 0 COMMENT '上传进度(0-100)',

    -- 访问统计
    access_count INT DEFAULT 0 COMMENT '访问次数',
    last_accessed TIMESTAMP NULL COMMENT '最后访问时间',

    -- 扩展元数据
    metadata JSON COMMENT '文件元数据',
    annotations JSON COMMENT '标注数据',
    properties JSON COMMENT '文件属性',

    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP NULL,

    -- 索引
    INDEX idx_dataset_id (dataset_id),
    INDEX idx_file_hash (file_hash),
    INDEX idx_storage_tier (storage_tier),
    INDEX idx_modality (modality),
    INDEX idx_status (status),
    INDEX idx_created_at (created_at),
    INDEX idx_last_accessed (last_accessed),
    INDEX idx_size (size),
    FOREIGN KEY (dataset_id) REFERENCES datasets(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='文件表';

    1.3 文件分片表 (file_chunks)¶

    mysql¶

    CREATE TABLE file_chunks (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    file_id VARCHAR(64) NOT NULL COMMENT '文件ID',
    upload_session_id VARCHAR(64) NOT NULL COMMENT '上传会话ID',
    chunk_index INT NOT NULL COMMENT '分片索引',
    chunk_size INT NOT NULL COMMENT '分片大小',
    chunk_hash VARCHAR(128) NOT NULL COMMENT '分片哈希',

    -- 存储信息
    storage_path VARCHAR(1024) NOT NULL COMMENT '存储路径',
    storage_provider ENUM('minio', 's3', 'oss', 'cos') DEFAULT 'minio',
    storage_bucket VARCHAR(255) NOT NULL,
    storage_key VARCHAR(1024) NOT NULL,

    -- 状态信息
    status ENUM('uploading', 'completed', 'error') DEFAULT 'uploading',
    retry_count INT DEFAULT 0 COMMENT '重试次数',

    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    -- 索引
    INDEX idx_file_id (file_id),
    INDEX idx_upload_session_id (upload_session_id),
    INDEX idx_chunk_index (chunk_index),
    INDEX idx_status (status),
    UNIQUE INDEX uk_session_chunk (upload_session_id, chunk_index),
    FOREIGN KEY (file_id) REFERENCES files(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='文件分片表';

    return nil

    UPDATE files
    SET access_count = access_count + 1, last_accessed = $1, updated_at = $1
    WHERE id IN (?) AND deleted_at IS NULL`

    var files []*model.File

    sqlQuery := `
    SELECT f.*
    FROM files f
    WHERE (
        to_tsvector('english', f.name) @@ plainto_tsquery('english', $1) OR
        to_tsvector('english', COALESCE(f.metadata::text, '')) @@ plainto_tsquery('english', $1)
    )
    AND f.deleted_at IS NULL
    AND f.status = 'ready'`

    args := []interface{}{query}
    argIndex := 2

    if modality != nil {
        sqlQuery += fmt.Sprintf(" AND f.modality = $%d", argIndex)
        args = append(args, *modality)
        argIndex++
    }

    sqlQuery += fmt.Sprintf(" ORDER BY ts_rank(to_tsvector('english', f.name), plainto_tsquery('english', $1)) DESC LIMIT $%d OFFSET $%d", argIndex, argIndex+1)
    args = append(args, limit, offset)

    err := r.db.SelectContext(ctx, &files, sqlQuery, args...)
    if err != nil {
        return nil, fmt.Errorf("search files failed: %v", err)
    }

    return files, nil
}

// 分层策略
type TierPolicy struct {
    ID          int           `json:"id"`
    Name        string        `json:"name"`
    Description string        `json:"description"`

    SourceTier  StorageTier   `json:"source_tier"`
    TargetTier  StorageTier   `json:"target_tier"`

    // 迁移条件
    Conditions  []Condition   `json:"conditions"`
    Enabled     bool          `json:"enabled"`
    Priority    int           `json:"priority"`
}

type Condition struct {
    Type  ConditionType   `json:"type"`
    Field string          `json:"field"`
    Op    ConditionOp     `json:"op"`
    Value interface{}     `json:"value"`
}

type ConditionType string
const (
    ConditionAge        ConditionType = "age"
    ConditionAccess     ConditionType = "access"
    ConditionSize       ConditionType = "size"
    ConditionModality   ConditionType = "modality"
)

type ConditionOp string
const (
    OpGT  ConditionOp = "gt"
    OpLT  ConditionOp = "lt"
    OpEQ  ConditionOp = "eq"
    OpNEQ ConditionOp = "neq"
)

// 策略评估器
type PolicyEvaluator struct {
    policies []*TierPolicy
    db       *gorm.DB
}

func (e *PolicyEvaluator) EvaluateFile(file *File) (StorageTier, []*TierPolicy) {
    var applicablePolicies []*TierPolicy
    currentTier := file.StorageTier

    for _, policy := range e.policies {
        if !policy.Enabled || policy.SourceTier != currentTier {
            continue
        }

        if e.evaluateConditions(policy, file) {
            applicablePolicies = append(applicablePolicies, policy)
        }
    }

    // 选择优先级最高的策略
    if len(applicablePolicies) > 0 {
        selectedPolicy := e.selectPolicy(applicablePolicies)
        return selectedPolicy.TargetTier, applicablePolicies
    }

    return currentTier, nil
}

func (e *PolicyEvaluator) evaluateConditions(policy *TierPolicy, file *File) bool {
    for _, condition := range policy.Conditions {
        if !e.evaluateCondition(condition, file) {
            return false
        }
    }
    return true
}

func (e *PolicyEvaluator) evaluateCondition(condition Condition, file *File) bool {
    switch condition.Type {
    case ConditionAge:
        fileAge := time.Since(file.CreatedAt).Hours() / 24 // 天数
        return e.compareValues(fileAge, condition.Op, condition.Value)

    case ConditionAccess:
        accessAge := time.Since(file.LastAccessed).Hours() / 24
        return e.compareValues(accessAge, condition.Op, condition.Value)

    case ConditionSize:
        return e.compareValues(file.Size, condition.Op, condition.Value)

    case ConditionModality:
        return file.Modality == condition.Value

    default:
        return false
    }
}

func (e *PolicyEvaluator) compareValues(actual interface{}, op ConditionOp, expected interface{}) bool {
    // 实现比较逻辑
    switch op {
    case OpGT:
        // 大于比较
        return true
    case OpLT:
        // 小于比较
        return true
    case OpEQ:
        // 等于比较
        return actual == expected
    case OpNEQ:
        // 不等于比较
        return actual != expected
    default:
        return false
    }
}

        return s.directDownload(ctx, file)

    case StorageTierCold:

    reader, err := storage.GetObject(ctx, file.StorageBucket, file.StorageKey)
    if err != nil {
        return nil, fmt.Errorf("get object failed: %v", err)
    }

    // 更新访问统计
    s.updateAccessStats(ctx, file)

    return &DownloadResponse{
        File:       file,
        Reader:     reader,
        Direct:     true,
        ExpiresAt:  time.Now().Add(1 * time.Hour),
    }, nil
}

// 恢复并下载（冷存储）
func (s *FileService) restoreAndDownload(ctx context.Context, file *File) (*DownloadResponse, error) {
    // 1. 检查是否已有恢复任务
    restoreTask, err := s.getActiveRestoreTask(ctx, file.ID)
    if err != nil {
        return nil, err
    }

    if restoreTask == nil {
        // 2. 创建新的恢复任务
        restoreTask, err = s.createRestoreTask(ctx, file)
        if err != nil {
            return nil, err
        }

        // 3. 触发恢复操作
        if err := s.triggerRestore(ctx, file, restoreTask); err != nil {
            return nil, err
        }
    }

    // 4. 检查恢复状态
    switch restoreTask.Status {
    case "completed":
        // 恢复完成，提供下载
        return s.downloadRestoredFile(ctx, file, restoreTask)

    case "restoring":
        // 恢复中，返回恢复信息
        return &DownloadResponse{
            File:       file,
            Direct:     false,
            Restoring:  true,
            EstimatedReady: restoreTask.EstimatedCompletion,
            Progress:   restoreTask.Progress,
        }, nil

    case "failed":
        return nil, fmt.Errorf("restore failed: %s", restoreTask.ErrorMessage)

    default:
        return nil, fmt.Errorf("unknown restore status: %s", restoreTask.Status)
    }
}

// 触发冷数据恢复
func (s *FileService) triggerRestore(ctx context.Context, file *File, task *RestoreTask) error {
    coldStorage := s.storageManager.GetTierStorage(StorageTierCold)
    warmStorage := s.storageManager.GetTierStorage(StorageTierWarm)

    // 恢复目标路径
    restorePath := fmt.Sprintf("restore/%s/%s", task.ID, file.Name)

    // 异步执行恢复
    go s.executeRestore(ctx, file, task, coldStorage, warmStorage, restorePath)

    return nil
}

func (s *FileService) executeRestore(ctx context.Context, file *File, task *RestoreTask,
    coldStorage, warmStorage ObjectStorage, restorePath string) {

    // 更新任务状态为恢复中
    s.updateRestoreTaskStatus(ctx, task.ID, "restoring", 0)

    // 从冷存储复制到温存储
    err := coldStorage.CopyObject(ctx, file.StorageBucket, file.StorageKey,
        warmStorage, warmStorage.GetBucket(StorageTierWarm), restorePath)

    if err != nil {
        s.updateRestoreTaskStatus(ctx, task.ID, "failed", 0, err.Error())
        return
    }

    // 更新任务状态为完成
    s.updateRestoreTaskStatus(ctx, task.ID, "completed", 100)

    // 更新文件临时层级（在恢复有效期内）
    s.updateFileTemporaryTier(ctx, file.ID, StorageTierWarm, restorePath,
        time.Now().Add(24*time.Hour)) // 恢复文件保留24小时
}

    // internal/service/storage/migration/auto_tiering.go
package migration

import (
    "context"
    "fmt"
    "time"
)

type AutoTieringService struct {
    storageManager *StorageManager
    policyEvaluator *PolicyEvaluator
    db             *gorm.DB
    config         *AutoTieringConfig
}

type AutoTieringConfig struct {
    Enabled         bool          `json:"enabled"`
    ScanInterval    time.Duration `json:"scan_interval"`
    BatchSize       int           `json:"batch_size"`
    MaxConcurrent   int           `json:"max_concurrent"`
}

func (s *AutoTieringService) Start() {
    if !s.config.Enabled {
        return
    }

    ticker := time.NewTicker(s.config.ScanInterval)
    defer ticker.Stop()

    for {
        select {
        case <-ticker.C:
            s.executeTieringCycle(context.Background())
        }
    }
}

func (s *AutoTieringService) executeTieringCycle(ctx context.Context) {
    // 1. 扫描需要评估的文件
    files, err := s.scanFilesForTiering(ctx)
    if err != nil {
        s.logger.Errorf("Scan files for tiering failed: %v", err)
        return
    }

    // 2. 评估每个文件的存储策略
    var migrationTasks []*MigrationTask
    for _, file := range files {
        targetTier, policies := s.policyEvaluator.EvaluateFile(file)
        if targetTier != file.StorageTier {
            migrationTasks = append(migrationTasks, &MigrationTask{
                FileID:      file.ID,
                SourceTier:  file.StorageTier,
                TargetTier:  targetTier,
                Policies:    policies,
                Priority:    s.calculateMigrationPriority(file, policies),
            })
        }
    }

    // 3. 执行迁移任务
    s.executeMigrationTasks(ctx, migrationTasks)
}

func (s *AutoTieringService) scanFilesForTiering(ctx context.Context) ([]*File, error) {
    var files []*File

    // 扫描最近未检查的文件
    err := s.db.Where("storage_tier IN (?)", []StorageTier{StorageTierHot, StorageTierWarm}).
        Where("last_tier_check < ? OR last_tier_check IS NULL", time.Now().Add(-24*time.Hour)).
        Limit(s.config.BatchSize).
        Find(&files).Error

    return files, err
}

func (s *AutoTieringService) executeMigrationTasks(ctx context.Context, tasks []*MigrationTask) {
    // 按优先级排序
    tasks = s.sortTasksByPriority(tasks)

    // 使用工作池执行迁移
    semaphore := make(chan struct{}, s.config.MaxConcurrent)

    for _, task := range tasks {
        semaphore <- struct{}{}

        go func(t *MigrationTask) {
            defer func() { <-semaphore }()

            if err := s.executeSingleMigration(ctx, t); err != nil {
                s.logger.Errorf("Migration failed for file %s: %v", t.FileID, err)
                s.recordMigrationFailure(ctx, t, err)
            } else {
                s.recordMigrationSuccess(ctx, t)
            }
        }(task)
    }
}

func (s *AutoTieringService) executeSingleMigration(ctx context.Context, task *MigrationTask) error {
    // 1. 获取文件信息
    file, err := s.getFile(ctx, task.FileID)
    if err != nil {
        return err
    }

    // 2. 锁定文件（防止并发修改）
    if err := s.lockFile(ctx, file.ID); err != nil {
        return err
    }
    defer s.unlockFile(ctx, file.ID)

    // 3. 执行数据迁移
    sourceStorage := s.storageManager.GetTierStorage(task.SourceTier)
    targetStorage := s.storageManager.GetTierStorage(task.TargetTier)

    // 目标存储路径
    targetKey := s.generateTargetStorageKey(file, task.TargetTier)

    // 复制文件
    err = sourceStorage.CopyObject(ctx, file.StorageBucket, file.StorageKey,
        targetStorage, targetStorage.GetBucket(task.TargetTier), targetKey)
    if err != nil {
        return fmt.Errorf("copy object failed: %v", err)
    }

    // 4. 验证目标文件
    if err := s.verifyMigration(file, targetStorage, targetKey); err != nil {
        // 验证失败，删除目标文件
        targetStorage.DeleteObject(ctx, targetStorage.GetBucket(task.TargetTier), targetKey)
        return fmt.Errorf("migration verification failed: %v", err)
    }

    // 5. 更新文件元数据
    if err := s.updateFileStorageInfo(ctx, file, task.TargetTier, targetKey); err != nil {
        return fmt.Errorf("update file storage info failed: %v", err)
    }

    // 6. 删除源文件（可选，根据策略决定）
    if s.shouldDeleteSource(file, task) {
        if err := sourceStorage.DeleteObject(ctx, file.StorageBucket, file.StorageKey); err != nil {
            s.logger.Warnf("Delete source file failed: %v", err)
        }
    }

    return nil
}
