    邮件相关¶

    目前 Go 语言在 Web 项目中添加邮件服务的主流实现主要有三种思路：

    1. 使用标准库 net/smtp 直接拼装 SMTP 会话（零依赖，但代码量大、易踩坑）。
    2. 使用社区封装库（jordan-wright/email、go-gomail、go-mail 等），一行代码完成 HTML/附件/抄送/TLS。
    3. 直接调用第三方邮件 SaaS（SendGrid、Mailgun、Amazon SES 等），通过官方 SDK 发送，省去 SMTP 运维。

    一、主流邮件库对比¶

    Go 生态中邮件相关库较多，核心差异体现在功能覆盖、API 设计和维护活跃度上。

    下面表格汇总了几个主流的Go邮件处理库及其特点，帮助你快速了解和选择：

    库名称 API友好度 功能丰富度 异步支持 安全性 推荐场景
    gomail ★★★★★ ★★★★☆ 需自定义 TLS/SSL 通用场景，HTML邮件、附件
    go-simple-mail ★★★★☆ ★★★★☆ 内置 TLS/SSL 需异步处理或较多附件的场景
    net/smtp ★★★☆☆ ★★☆☆☆ 无 TLS 简单文本邮件发送
    email ★★★★☆ ★★★☆☆ 需自定义 TLS/SSL 需要轻量级、易用API的场景
    go-message ★★★☆☆ ★★★★★ 需自定义 DKIM, TLS/SSL 需要处理复杂邮件格式或需要DKIM签名的场景
    go-mail ★★★★☆ ★★★★☆ 内置 TLS/SSL 高级需求（需签名、批量发送、安全合规）

    💡 说明与选型建议：

    • go-mail 和 go-simple-mail 是功能全面且开发者友好的选择，能满足大多数Web项目的邮件需求，例如交易邮件、通知等。
    • 标准库 net/smtp 更适合发送简单的文本邮件或者学习SMTP协议原理。
    • email 库提供了比标准库更友好的接口。
    • 若需要处理低层邮件格式、解析邮件或提供DKIM签名等高级功能，go-mail 和go-message 是个强大的选择。

    二、核心库使用示例¶

    1. go-mail（高级需求）¶

    go-mail 专注于现代邮件功能，支持 DKIM 签名（防邮件被标记为垃圾邮件）、SMTP 连接池（提升批量发送效率），适合对安全性和性能有高要求的项目。

    1.1.安装¶

    go get github.com/wneessen/go-mail

    1.2.使用 go-mail 发送邮件¶

    package main

import (
    "context"
    "log"
    "time"

    "github.com/wneessen/go-mail"
)

func main() {
    // 创建一个新的邮件消息
    msg := mail.NewMsg()
    if err := msg.From("发件人名称 <your_email@example.com>"); err != nil {
        log.Fatalf("Failed to set From: %s", err)
    }
    if err := msg.To("收件人名称 <recipient@example.com>"); err != nil {
        log.Fatalf("Failed to set To: %s", err)
    }
    msg.Subject("这是一封测试邮件")

    // 设置邮件正文：纯文本和HTML版本
    msg.SetBodyString(mail.TypeTextPlain, "这是一封纯文本测试邮件。")
    msg.SetBodyString(mail.TypeTextHTML, "<h1>这是一封HTML测试邮件</h1><p>Hello!</p>")

    // 添加附件（可选）
    // if err := msg.Attach("/path/to/your/file.pdf"); err != nil {
    //     log.Fatalf("Failed to attach file: %s", err)
    // }

    // 创建带有超时控制的上下文
    ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()

    // 使用SMTP客户端发送邮件
    client, err := mail.NewClient("smtp.example.com",
        mail.WithPort(587),                       // SMTP端口
        mail.WithSMTPAuth(mail.AuthLogin),        // 认证方式
        mail.WithUsername("your_username"),       // SMTP用户名
        mail.WithPassword("your_password"),       // SMTP密码或授权码
        mail.WithTLSPolicy(mail.TLSMandatory),    // TLS策略
    )
    if err != nil {
        log.Fatalf("Failed to create mail client: %s", err)
    }

    // 实际发送邮件
    if err := client.DialAndSendWithContext(ctx, msg); err != nil {
        log.Fatalf("Failed to send email: %s", err)
    }

    log.Println("Email sent successfully!")
}

    1.3.示例1：发送带 DKIM 签名的邮件¶

    package main

import (
    "log"
    "os"

    "github.com/wneessen/go-mail"
)

// 发送带DKIM签名的邮件（适合生产环境，提升邮件送达率）
func SendSignedMail(to, subject, htmlContent string) error {
    // 创建邮件消息
    m := mail.NewMsg()
    if err := m.From("Web服务 <your-email@example.com>"); err != nil {
        return err
    }
    if err := m.To(to); err != nil {
        return err
    }
    m.Subject(subject)
    m.SetBodyString(mail.TypeTextHTML, htmlContent)

    // 添加DKIM签名（需提前准备私钥文件，提升邮件可信度）
    // 生成DKIM密钥对：https://dkimcore.org/tools/
    dkimPrivKey, err := os.ReadFile("./dkim_private.pem") // 私钥文件路径
    if err != nil {
        return err
    }
    if err := m.SignDKIM(dkimPrivKey, "example.com", "mail"); err != nil { // 域名、选择器
        log.Printf("DKIM签名失败（非致命错误）: %v", err)
    }

    // 创建SMTP客户端（支持连接池）
    c, err := mail.NewClient("smtp.example.com",
        mail.WithPort(587),
        mail.WithSMTPAuth(mail.SMTPAuthPlain),
        mail.WithUsername("your-email@example.com"),
        mail.WithPassword(os.Getenv("SMTP_PASS")),
        mail.WithTLSConfig(&tls.Config{ServerName: "smtp.example.com"}), // 启用TLS
    )
    if err != nil {
        return err
    }
    defer c.Close()

    // 发送邮件
    return c.DialAndSend(m)
}

func main() {
    if err := SendSignedMail(
        "user@example.com",
        "安全验证通知",
        `<p>你的验证码是：<strong>654321</strong></p>`,
    ); err != nil {
        log.Fatalf("发送失败: %v", err)
    }
    log.Println("带DKIM签名的邮件发送成功")
}

    1.4.示例2：批量邮件¶

    package main

import (
    "log"
    "os"
    "sync"
    "time"

    "github.com/wneessen/go-mail"
)

// 批量发送任务
type BatchMailTask struct {
    To      string // 收件人
    Subject string // 主题
    Body    string // HTML内容
}

// 批量发送邮件
func BatchSendMails(tasks []BatchMailTask, maxConcurrency int) error {
    // 1. 创建SMTP客户端（可复用，类似连接池）
    c, err := mail.NewClient("smtp.example.com",
        mail.WithPort(587),
        mail.WithSMTPAuth(mail.SMTPAuthPlain),
        mail.WithUsername(os.Getenv("SMTP_USER")),
        mail.WithPassword(os.Getenv("SMTP_PASS")),
        mail.WithTLSConfig(&tls.Config{ServerName: "smtp.example.com"}),
        mail.WithConnectionTimeout(10*time.Second),
    )
    if err != nil {
        return err
    }
    defer c.Close() // 最后关闭连接

    // 2. 控制并发数量（避免压垮邮件服务器）
    sem := make(chan struct{}, maxConcurrency) // 信号量控制并发
    var wg sync.WaitGroup
    errCh := make(chan error, len(tasks))      // 收集错误

    // 3. 并发发送每封邮件
    for _, task := range tasks {
        wg.Add(1)
        sem <- struct{}{} // 占用信号量

        go func(t BatchMailTask) {
            defer wg.Done()
            defer func() { <-sem }() // 释放信号量

            // 创建邮件消息
            m := mail.NewMsg()
            if err := m.From("通知系统 <sender@example.com>"); err != nil {
                errCh <- err
                return
            }
            if err := m.To(t.To); err != nil {
                errCh <- err
                return
            }
            m.Subject(t.Subject)
            m.SetBodyString(mail.TypeTextHTML, t.Body)

            // 发送（重试3次，处理临时错误）
         retryLoop:
            for i := 0; i < 3; i++ {
                if err := c.DialAndSend(m); err != nil {
                    log.Printf("发送给 %s 失败（第%d次）: %v", t.To, i+1, err)
                    time.Sleep(time.Duration(i+1) * time.Second) // 指数退避重试
                    continue retryLoop
                }
                log.Printf("发送给 %s 成功", t.To)
                return
            }
            errCh <- err // 3次失败后记录错误
        }(task)
    }

    // 4. 等待所有任务完成并关闭错误通道
    go func() {
        wg.Wait()
        close(errCh)
    }()

    // 5. 收集错误
    var finalErr error
    for err := range errCh {
        if err != nil {
            log.Printf("最终失败: %v", err)
            finalErr = err // 简化处理，实际可收集所有错误
        }
    }
    return finalErr
}

func main() {
    // 示例：100封批量邮件任务
    tasks := make([]BatchMailTask, 100)
    for i := 0; i < 100; i++ {
        tasks[i] = BatchMailTask{
            To:      "user" + string(rune(i)) + "@example.com",
            Subject: "批量发送测试",
            Body:    `<p>这是第` + string(rune(i)) + `封批量邮件</p>`,
        }
    }

    // 并发发送（最多10个同时连接）
    if err := BatchSendMails(tasks, 10); err != nil {
        log.Fatalf("批量发送失败: %v", err)
    }
    log.Println("所有邮件发送完成")
}

    2.go-message¶

    • 构建符合 MIME 标准的邮件消息（支持文本、HTML、附件）；
    • 处理多部分邮件（如同时包含文本和 HTML 版本）；
    • 支持编码转换（解决中文等非 ASCII 字符显示问题）；
    • 解析收到的邮件消息（提取主题、正文、附件等）。

    2.1.安装¶

    go get github.com/emersion/go-message

    2.2.示例 1：构建简单文本邮件并发送¶

    package main

import (
    "net/smtp"
    "os"

    "github.com/emersion/go-message"
    "github.com/emersion/go-message/textproto"
)

func main() {
    // 1. 构建邮件消息
    // 创建消息头
    header := textproto.Header{}
    header.Set("From", "sender@example.com")
    header.Set("To", "recipient@example.com")
    header.Set("Subject", "测试文本邮件") // 若含中文，需确保编码正确

    // 创建消息体（纯文本）
    body := "这是一封通过go-message构建的纯文本邮件"

    // 用go-message封装消息（自动处理MIME格式）
    msg := message.NewMessage(header, []byte(body))

    // 2. 通过SMTP发送（使用标准库net/smtp）
    smtpServer := "smtp.example.com:587" // 如smtp.qq.com:587
    auth := smtp.PlainAuth(
        "",
        os.Getenv("SMTP_USER"), // 发件人邮箱
        os.Getenv("SMTP_PASS"), // 邮箱密码/授权码
        "smtp.example.com",
    )

    // 发送邮件（msg.WriteTo()获取完整邮件内容）
    var buf bytes.Buffer
    if err := msg.WriteTo(&buf); err != nil {
        panic(err)
    }
    if err := smtp.SendMail(
        smtpServer,
        auth,
        "sender@example.com", // 发件人
        []string{"recipient@example.com"}, // 收件人
        buf.Bytes(),
    ); err != nil {
        panic(err)
    }
}

    2.3.示例 2：构建 HTML 邮件（含文本备用版本）¶

    package main

import (
    "bytes"
    "net/smtp"
    "os"

    "github.com/emersion/go-message"
    "github.com/emersion/go-message/multipart"
    "github.com/emersion/go-message/textproto"
)

func main() {
    // 1. 创建邮件头
    header := textproto.Header{}
    header.Set("From", "sender@example.com")
    header.Set("To", "recipient@example.com")
    header.Set("Subject", "测试HTML邮件")

    // 2. 创建多部分消息（multipart/alternative）
    // 用于同时包含HTML和纯文本版本
    mw, err := message.CreateWriter(nil, header, "multipart/alternative")
    if err != nil {
        panic(err)
    }
    defer mw.Close()

    // 3. 添加纯文本部分（备用）
    textPartHeader := textproto.Header{}
    textPartHeader.Set("Content-Type", "text/plain; charset=utf-8")
    textPart, err := mw.CreatePart(textPartHeader)
    if err != nil {
        panic(err)
    }
    textPart.Write([]byte("这是纯文本备用内容，当客户端不支持HTML时显示"))

    // 4. 添加HTML部分（主要内容）
    htmlPartHeader := textproto.Header{}
    htmlPartHeader.Set("Content-Type", "text/html; charset=utf-8")
    htmlPart, err := mw.CreatePart(htmlPartHeader)
    if err != nil {
        panic(err)
    }
    htmlContent := `
        <html>
            <body>
                <h1>HTML邮件测试</h1>
                <p>这是一封<b>HTML格式</b>的邮件</p>
                <p><a href="https://example.com">点击访问</a></p>
            </body>
        </html>
    `
    htmlPart.Write([]byte(htmlContent))

    // 5. 完成消息构建
    mw.Close()

    // 6. 获取完整邮件内容并发送
    var buf bytes.Buffer
    if err := mw.WriteTo(&buf); err != nil {
        panic(err)
    }

    // 发送逻辑（同示例1）
    smtpServer := "smtp.example.com:587"
    auth := smtp.PlainAuth("", os.Getenv("SMTP_USER"), os.Getenv("SMTP_PASS"), "smtp.example.com")
    if err := smtp.SendMail(smtpServer, auth, "sender@example.com", []string{"recipient@example.com"}, buf.Bytes()); err != nil {
        panic(err)
    }
}

    package main

import (
    "bytes"
    "io"
    "net/smtp"
    "os"

    "github.com/emersion/go-message"
    "github.com/emersion/go-message/multipart"
    "github.com/emersion/go-message/textproto"
)

func main() {
    // 1. 创建邮件头
    header := textproto.Header{}
    header.Set("From", "sender@example.com")
    header.Set("To", "recipient@example.com")
    header.Set("Subject", "带附件的邮件")

    // 2. 创建multipart/mixed消息（用于包含正文和附件）
    mw, err := message.CreateWriter(nil, header, "multipart/mixed")
    if err != nil {
        panic(err)
    }
    defer mw.Close()

    // 3. 添加邮件正文（HTML+文本，复用示例2的逻辑）
    bodyHeader := textproto.Header{}
    bodyHeader.Set("Content-Type", "multipart/alternative")
    bodyPart, err := mw.CreatePart(bodyHeader)
    if err != nil {
        panic(err)
    }

    // 3.1 正文的纯文本部分
    textPartHeader := textproto.Header{}
    textPartHeader.Set("Content-Type", "text/plain; charset=utf-8")
    textPart, err := multipart.NewWriter(bodyPart).CreatePart(textPartHeader)
    if err != nil {
        panic(err)
    }
    textPart.Write([]byte("邮件包含附件，请查收"))

    // 3.2 正文的HTML部分
    htmlPartHeader := textproto.Header{}
    htmlPartHeader.Set("Content-Type", "text/html; charset=utf-8")
    htmlPart, err := multipart.NewWriter(bodyPart).CreatePart(htmlPartHeader)
    if err != nil {
        panic(err)
    }
    htmlPart.Write([]byte("<p>邮件包含附件，请查收</p>"))

    // 4. 添加附件（本地文件）
    attachmentPath := "document.pdf" // 附件路径
    attachmentFile, err := os.Open(attachmentPath)
    if err != nil {
        panic(err)
    }
    defer attachmentFile.Close()

    // 4.1 设置附件头（指定文件名和MIME类型）
    attachmentHeader := textproto.Header{}
    attachmentHeader.Set("Content-Type", "application/pdf") // 根据文件类型设置
    attachmentHeader.Set("Content-Disposition", `attachment; filename="document.pdf"`) // 附件名称

    // 4.2 创建附件部分并写入内容
    attachmentPart, err := mw.CreatePart(attachmentHeader)
    if err != nil {
        panic(err)
    }
    io.Copy(attachmentPart, attachmentFile) // 将文件内容写入附件部分

    // 5. 完成消息构建并发送
    mw.Close()
    var buf bytes.Buffer
    mw.WriteTo(&buf)

    // 发送逻辑
    smtpServer := "smtp.example.com:587"
    auth := smtp.PlainAuth("", os.Getenv("SMTP_USER"), os.Getenv("SMTP_PASS"), "smtp.example.com")
    smtp.SendMail(smtpServer, auth, "sender@example.com", []string{"recipient@example.com"}, buf.Bytes())
}

    2.5.示例4：批量邮件¶

    package main

import (
    "bytes"
    "log"
    "net/smtp"
    "os"
    "sync"
    "time"

    "github.com/emersion/go-message"
    "github.com/emersion/go-message/textproto"
)

// 构建单封邮件的MIME内容
func buildMessage(to, subject, body string) ([]byte, error) {
    header := textproto.Header{}
    header.Set("From", "sender@example.com")
    header.Set("To", to)
    header.Set("Subject", subject)
    header.Set("MIME-Version", "1.0")
    header.Set("Content-Type", "text/html; charset=utf-8")

    msg := message.NewMessage(header, []byte(body))
    var buf bytes.Buffer
    if err := msg.WriteTo(&buf); err != nil {
        return nil, err
    }
    return buf.Bytes(), nil
}

// 批量发送（每批复用一个SMTP连接）
func BatchSendMails(tasks []BatchMailTask, batchSize int) error {
    smtpServer := "smtp.example.com:587"
    auth := smtp.PlainAuth("",
        os.Getenv("SMTP_USER"),
        os.Getenv("SMTP_PASS"),
        "smtp.example.com",
    )

    var wg sync.WaitGroup
    errCh := make(chan error, len(tasks)/batchSize+1)

    // 按批次处理（每批用一个连接）
    for i := 0; i < len(tasks); i += batchSize {
        end := i + batchSize
        if end > len(tasks) {
            end = len(tasks)
        }
        batch := tasks[i:end]

        wg.Add(1)
        go func(batch []BatchMailTask) {
            defer wg.Done()

            // 建立连接（每批复用一个连接）
            conn, err := smtp.Dial(smtpServer)
            if err != nil {
                errCh <- err
                return
            }
            defer conn.Close()

            // 认证
            if err := conn.Auth(auth); err != nil {
                errCh <- err
                return
            }
            if err := conn.Mail("sender@example.com"); err != nil {
                errCh <- err
                return
            }

            // 批量发送当前批次
            for _, task := range batch {
                // 重置收件人（单连接发送多封邮件需重新设置RCPT）
                if err := conn.Rcpt(task.To); err != nil {
                    log.Printf("添加收件人 %s 失败: %v", task.To, err)
                    continue
                }

                // 构建并发送邮件内容
                msg, err := buildMessage(task.To, task.Subject, task.Body)
                if err != nil {
                    log.Printf("构建邮件失败: %v", err)
                    continue
                }

                w, err := conn.Data()
                if err != nil {
                    log.Printf("获取数据写入器失败: %v", err)
                    continue
                }
                if _, err := w.Write(msg); err != nil {
                    log.Printf("发送给 %s 失败: %v", task.To, err)
                } else {
                    log.Printf("发送给 %s 成功", task.To)
                }
                w.Close()
                time.Sleep(100 * time.Millisecond) // 控制发送速度
            }
        }(batch)
    }

    // 等待所有批次完成
    go func() {
        wg.Wait()
        close(errCh)
    }()

    // 收集错误
    var finalErr error
    for err := range errCh {
        if err != nil {
            log.Printf("批次错误: %v", err)
            finalErr = err
        }
    }
    return finalErr
}

func main() {
    // 示例：20封邮件，每批5封
    tasks := make([]BatchMailTask, 20)
    for i := 0; i < 20; i++ {
        tasks[i] = BatchMailTask{
            To:      "user" + string(rune(i)) + "@example.com",
            Subject: "go-message批量测试",
            Body:    `<p>这是第` + string(rune(i)) + `封邮件</p>`,
        }
    }

    if err := BatchSendMails(tasks, 5); err != nil {
        log.Fatalf("批量发送失败: %v", err)
    }
    log.Println("所有邮件发送完成")
}

    3. email（轻量快速集成）¶

    email 库 API 极简，适合快速实现基础邮件功能（如验证码、简单通知），无需复杂配置。

    3.1.安装¶

    3.2.示例1：快速发送 HTML 邮件¶

    package main

    "net/smtp"
    "os"

    "github.com/jordan-wright/email"
)

    e := email.NewEmail()

    auth := smtp.PlainAuth("",
        os.Getenv("SMTP_USER"),

        log.Fatalf("发送失败: %v", err)
    }
    log.Println("邮件发送成功")
}

    3.3.示例2：批量邮件¶

    package main

import (
    "log"
    "net/smtp"
    "os"
    "sync"
    "time"

    "github.com/jordan-wright/email"
)

// 批量发送（轻量实现）
func BatchSendMails(tasks []BatchMailTask, maxConcurrency int) error {
    smtpServer := "smtp.example.com:587"
    auth := smtp.PlainAuth("",
        os.Getenv("SMTP_USER"),
        os.Getenv("SMTP_PASS"),
        "smtp.example.com",
    )

    var wg sync.WaitGroup
    sem := make(chan struct{}, maxConcurrency) // 控制并发
    errCh := make(chan error, len(tasks))

    for _, task := range tasks {
        wg.Add(1)
        sem <- struct{}{}

        go func(t BatchMailTask) {
            defer wg.Done()
            defer func() { <-sem }()

            // 创建邮件
            e := email.NewEmail()
            e.From = "sender@example.com"
            e.To = []string{t.To}
            e.Subject = t.Subject
            e.HTML = []byte(t.Body)

            // 发送（重试2次）
            for i := 0; i < 2; i++ {
                if err := e.Send(smtpServer, auth); err != nil {
                    log.Printf("发送给 %s 失败（第%d次）: %v", t.To, i+1, err)
                    time.Sleep(time.Second)
                    continue
                }
                log.Printf("发送给 %s 成功", t.To)
                return
            }
            errCh <- err // 记录最终失败
        }(task)
    }

    // 等待并收集错误
    go func() {
        wg.Wait()
        close(errCh)
    }()

    var finalErr error
    for err := range errCh {
        if err != nil {
            finalErr = err
        }
    }
    return finalErr
}

func main() {
    // 示例：50封邮件，最大并发8
    tasks := make([]BatchMailTask, 50)
    for i := 0; i < 50; i++ {
        tasks[i] = BatchMailTask{
            To:      "user" + string(rune(i)) + "@example.com",
            Subject: "email库批量测试",
            Body:    `<p>这是第` + string(rune(i)) + `封邮件</p>`,
        }
    }

    if err := BatchSendMails(tasks, 8); err != nil {
        log.Fatalf("批量发送失败: %v", err)
    }
    log.Println("所有邮件发送完成")
}
