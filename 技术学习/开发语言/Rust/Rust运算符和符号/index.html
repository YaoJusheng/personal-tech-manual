    一、基础语法标点符号¶

    这些是构建代码结构的基础符号，用于分隔语句、定义类型和访问成员。

    符号 名称 主要用途与示例
    ; 分号 结束一条语句。let x = 5;
    : 冒号 用于类型标注、字段定义、返回值类型。let x: i32;, fn add(a: i32) -> i32 { ... }
    :: 双冒号 路径分隔符，用于访问模块、枚举变体、关联函数等。std::io::stdin()
    . 点号 访问结构体或对象的字段、方法。point.x，"hello".len()
    , 逗号 分隔列表中的元素，如函数参数、数组元素、结构体字段等
    -> 箭头 指定函数的返回类型。fn add(x: i32) -> i32 { ... }
    => 胖箭头 在 match 表达式中连接模式与要执行的代码。match value { Some(x) => x, _ => 0 }
    @ At 符号 在模式匹配中，将值同时绑定到新变量。id @ 1..=5 => println!("{}", id)
    _ 下划线 通配符模式，用于匹配并忽略任何值；也表示未使用变量（不产生警告）
    .. 两个点 右开区间范围，在模式中表示忽略剩余部分。1..5, 结构体更新 ..default
    ..= 三个点 闭区间范围，包含上下边界。1..=5
    ... 三个点（已废弃） 旧语法，Rust 2018 edition 开始已移除，请使用 ..= 或 ..

    分号 ;¶

    分号用于结束语句，是 Rust 中语句结束的标志。

    fn main() {
    // 基本语句
    let x = 5;          // 变量绑定语句
    let y = 10;         // 语句结束
    println!("{}", x + y);  // 函数调用语句

    // 循环中的分号
    let mut i = 0;
    while i < 3 {
        println!("i = {}", i);
        i += 1;  // 分号分隔循环体语句

    溢出处理方法¶

    Rust 提供了多种处理整数溢出的方法。

    fn main() {
    let max_u8 = u8::MAX;  // 255

    // 默认行为：Debug 模式 panic，Release 模式截断
    // let _ = max_u8 + 1;  // panic in debug, wraps in release

    // wrapping_* 方法：环绕到另一端
    println!("wrapping_add(1): {} -> {}", max_u8, max_u8.wrapping_add(1));  // 255 -> 0
    println!("wrapping_sub(1): {} -> {}", u8::MIN, u8::MIN.wrapping_sub(1)); // 0 -> 255
    println!("wrapping_mul(2): {} -> {}", 100u8, 100u8.wrapping_mul(2));     // 200

    // checked_* 方法：溢出时返回 None
    match max_u8.checked_add(1) {
        Some(n) => println!("Result: {}", n),
        None => println!("Overflow occurred!"),
    }

    // saturating_* 方法：限制在边界值
    println!("saturating_add(1): {} -> {}", max_u8, max_u8.saturating_add(1));  // 255 -> 255
    println!("saturating_sub(1): {} -> {}", u8::MIN, u8::MIN.saturating_sub(1)); // 0 -> 0

    // overflowing_* 方法：返回(结果, 是否溢出)
    let (result, overflowed) = 255u8.overflowing_add(1);
    println!("overflowing_add(1): result={}, overflowed={}", result, overflowed);  // 0, true

    // 位移溢出处理
    println!("shl(8) on u8: {}", 1u8.wrapping_shl(8));  // 0 (位移 8 位等于溢出)
}

    println!("a | b = {:04b} ({} | {})", a | b, a, b);  // 1110 (14)

    // 按位异或：对应位不同时为 1
    println!("a ^ b = {:04b} ({} ^ {})", a ^ b, a, b);  // 0110 (6)

    // 按位取反
    println!("!a = {:08b} (溢出后的 u8)", !a);  // 11110101 (245)

    // 左移
    println!("a << 1 = {:04b} ({} << 1)", a << 1, a);  // 0100 (20)
    println!("a << 2 = {:04b} ({} << 2)", a << 2, a);  // 1000 (40)

    // 右移（无符号数高位补 0）
    println!("a >> 1 = {:04b} ({} >> 1)", a >> 1, a);  // 0101 (5)
    println!("a >> 2 = {:04b} ({} >> 2)", a >> 2, a);  // 0010 (2)

    // 复合赋值
    let mut flags = 0b0001;  // bit 0 设置
    flags |= 0b0100;         // 设置 bit 2
    println!("flags = {:04b}", flags);  // 0101

    // 判断 bit 是否设置
    let bit_check = flags & 0b0100;
    println!("bit 2 is set: {}", bit_check != 0);  // true

    // 掩码操作
    let value: u8 = 0b1111_0000;  // 240
    let mask: u8 = 0b0000_1111;   // 15
    println!("value & mask = {:08b}", value & mask);  // 0000_0000
}

    引用与借用¶

    符号 名称 说明
    & 不可变引用 共享借用，可同时存在多个，不可修改

    // 不可变引用
    let s1 = String::from("hello");
    let r1 = &s1;           // 创建不可变引用
    let r2 = &s1;           // 可以有多个不可变引用

            y: self.y + other.y,
        }
    }

    // 泛型方法
    fn return_self<T: From<Point>>(self) -> T {
        T::from(self)
    }
}

    use std::collections::{HashMap, BTreeMap};

fn main() {
    // 创建 HashMap
    let mut map: HashMap<&str, i32> = HashMap::new();

    // 插入
    map.insert("Apple", 3);
    map.insert("Banana", 5);
    map.insert("Orange", 2);

    // with_capacity 预分配
    let mut map2 = HashMap::with_capacity(100);
    map2.insert("key", 1);

    // from 数组创建
    let map3: HashMap<&str, i32> = HashMap::from([
        ("a", 1),
        ("b", 2),
    ]);

    // 获取值
    match map.get("Apple") {
        Some(v) => println!("Apple: {}", v),
        None => println!("Not found"),
    }

    // get_with_default
    let count = map.get("Grape").copied().unwrap_or(0);
    println!("Grape count: {}", count);

    // or_insert
    let entry = map.entry("Grape").or_insert(0);
    *entry += 1;
    println!("Grape after or_insert: {}", map["Grape"]);

    // or_insert_with
    let entry = map.entry("Mango").or_insert_with(|| {
        println!("Creating Mango entry");
        10
    });
    println!("Mango: {}", entry);

    // 删除
    map.remove("Orange");
    println!("After removing Orange: {:?}", map.keys().collect::<Vec<_>>());

    // 包含检查
    println!("Has Apple: {}", map.contains_key("Apple"));

    // 迭代
    for (key, value) in &map {
        println!("{}: {}", key, value);
    }

    // 键值迭代
    let keys: Vec<_> = map.keys().collect();
    let values: Vec<_> = map.values().collect();
    println!("keys: {:?}, values: {:?}", keys, values);
}

    十六、Send 与 Sync Traits¶

    Trait 说明
    Send 类型可以在线程间转移所有权
    Sync 类型可以安全地从多个线程引用（&T 是 Sync）

    Send 示例¶

    // Rc 不是 Send（引用计数不是线程安全的）

    Sync 示例¶

    fn main() {
    // i32 是 Sync（&i32 可以安全地在多线程间共享）
    let data = 5;
    std::thread::spawn(|| {
        println!("data = {}", data);
    }).join().unwrap();

    // RefCell 不是 Sync（运行时借用检查）
    // let ref_cell = std::cell::RefCell::new(5);
    // std::thread::spawn(|| {  // 错误！RefCell 不是 Sync
    //     println!("{:?}", ref_cell);
    // });

    // Mutex 是 Sync（提供了线程安全的借用）
    use std::sync::Mutex;
    let mutex = Mutex::new(5);

    println!("mutex value: {}", *mutex.lock().unwrap());
}
