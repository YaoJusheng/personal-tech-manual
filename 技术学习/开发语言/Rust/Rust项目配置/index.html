    Rust项目配置¶

    Cargo.toml作为Rust 项目中的文件配置。这是 Rust 项目的核心配置文件，承载着项目的元数据、依赖管理和构建配置。

    一、基本结构¶

    Cargo.toml 采用 TOML 格式，主要包含以下几个核心部分：

    [package]          # 项目元数据
name = "my-project"           # 项目名称（必需）
version = "0.1.0"             # 版本号（必需），遵循语义化版本
edition = "2021"              # Rust版本（2021/2018/2015）

[dependencies]     # 生产依赖
serde = "1.0"

[dev-dependencies] # 开发依赖
tokio = { version = "1.0", features = ["full"] }

[lib]              # 库配置
name = "my_lib"
path = "src/lib.rs"

[[bin]]            # 二进制目标
name = "my_app"
path = "src/main.rs"

    二、[package] 节详解¶

    基础标识字段（必填）¶

    • name: 项目名称，需符合 Rust 标识符规则
    • version: 版本号，遵循语义化版本规范（SemVer）
    • edition: Rust 版本（2018、2021），决定语言特性集合

    描述性字段（选填）¶

    [package]
authors = ["Alice <alice@example.com>"]
description = "高性能网络服务"
homepage = "https://example.com"
repository = "https://github.com/example/project"
license = "MIT OR Apache-2.0" # 许可证
readme = "README.md"          # README文件
documentation = "https://docs.rs/crate"
keywords = ["web", "api"]
categories = ["web-programming"]

    高级配置字段¶

    [package]
default-run = "main_app"  # cargo run 默认执行的二进制
publish = false           # 禁止发布到 crates.io
resolver = "2"            # 依赖解析器版本
build = "build.rs"        # 构建脚本路径

    三、依赖管理详解¶

    依赖类型分类¶

    1. [dependencies]: 生产依赖，运行时必需
    2. [dev-dependencies]: 开发依赖，仅开发阶段使用
    3. [build-dependencies]: 构建依赖，用于 build.rs
    4. 基础依赖
    # 基础依赖声明
# 运行时依赖
[dependencies]
serde = "1.0"                    # 指定版本，等价于 ^1.0
tokio = { version = "1.0", features = ["full"] }

rand = "0.8"                    # 自动使用最新补丁版本
actix-web = "4.0.0-beta.8"      # 预发布版本

# 精确版本控制
log = "=0.4.14"                  # 精确版本
regex = "~1.5"                   # 仅补丁版本更新
chrono = ">=0.4, <0.5"           # 版本范围

# 开发依赖（仅用于测试、示例等）
[dev-dependencies]
assert_fs = "1.0"               # 文件系统断言库
tempfile = "3.3"

# 构建依赖
[build-dependencies]
cc = "1.0"                      # C代码编译工具
vergen = { version = "7", features = ["git"] }

    高级依赖配置¶

    [dependencies]
# Git 依赖
regex = { git = "https://github.com/rust-lang/regex" }
regex = { git = "https://github.com/rust-lang/regex", branch = "master" }
regex = { git = "https://github.com/rust-lang/regex", tag = "1.5.0" }
regex = { git = "https://github.com/rust-lang/regex", rev = "e6189e9" }

# 本地路径依赖
my-helper = { path = "../my-helper" }

# 可选依赖
my-optional = { version = "0.1", optional = true }

# 重命名依赖
reqwest_renamed = { package = "reqwest", version = "0.11", features = ["json"] }

# 平台特定依赖
[target.'cfg(windows)'.dependencies]
winapi = "0.3"

[target.'cfg(unix)'.dependencies]
nix = "0.24"
libc = "0.2"

    特性（Features）机制¶

    [features]
default = ["tls", "json", "compress"]    # 默认启用的特性
tls = ["tokio/rustls", "hyper-tls"]
json = ["serde/json"]            # 简单特性
compress = ["flate2"]            # 启用压缩支持
full = ["json", "compress", "log", "metrics"]  # 全功能特性
metrics = ["prometheus"]

# 特性间的依赖关系
advanced = ["json", "compress", "crypto"]
crypto = ["ring"]

# 可选依赖自动转为特性
[dependencies]
tokio = { version = "1.0", features = ["rt-multi-thread", "macros"] }
serde = { version = "1.0", features = ["derive"], default-features = false, optional = true }
flate2 = { version = "1.0", optional = true }
ring = { version = "0.16", optional = true }

# 开发特性
[dev-dependencies]
test-utils = { path = "test-utils", optional = true }

[features]
dev = ["test-utils"]

    四、构建配置详解¶

    目标配置¶

    1.库目标配置¶

    # 库配置
[lib]
name = "my_lib"                 # 库名（默认与包名相同）
path = "src/lib.rs"             # 源文件路径（默认）
crate-type = ["lib", "cdylib"]  # 生成类型：
                                 # - rlib: Rust库
                                 # - dylib: 动态库
                                 # - cdylib: C兼容动态库
                                 # - staticlib: 静态库
                                 # - proc-macro: 过程宏库

    2.二进制目标配置¶

    # 二进制配置（可多个）
# 单个二进制文件（简单配置）
[[bin]]
name = "myapp"                    # 可执行文件名
path = "src/bin/myapp.rs"        # 源文件路径
test = true                       # 是否参与测试（默认true）
bench = true                      # 是否参与基准测试
doc = true                        # 是否生成文档

# 多个二进制文件
[[bin]]
name = "server"
path = "src/bin/main.rs"

[[bin]]
name = "cli"
path = "src/bin/cli/main.rs"

# 二进制文件测试配置
[[test]]
name = "integration_test"
path = "tests/integration.rs"

    3.测试和示例配置¶

    # 示例程序
[[example]]
name = "simple-example"
path = "examples/simple.rs"

[[example]]
name = "advanced-example"
path = "examples/advanced/main.rs"

# 集成测试
[[test]]
name = "my-integration-test"
path = "tests/my_test.rs"

    构建配置¶

    1.Profile 配置¶

    # 开发模式优化
[profile.dev]
opt-level = 1                  # 轻度优化(优化级别 0-3)
debug = true                   # 保留调试信息
split-debuginfo = "unpacked"     # 调试信息处理方式
debug-assertions = true          # 启用调试断言
overflow-checks = true           # 启用溢出检查
rpath = false                    # 运行时路径
lto = false                      # 链接时优化
codegen-units = 16               # 代码生成单元数
panic = "unwind"                 # panic处理方式
incremental = true               # 增量编译
strip = "none"                   # 剥离符号

# 发布模式优化
[profile.release]
opt-level = 3          # 最高优化级别
lto = "fat"            # 启用链接时优化(thin/fat)
codegen-units = 1      # 减少代码生成单元
panic = "abort"        # panic 时直接终止
split-debuginfo = "packed"
debug-assertions = false
overflow-checks = false
incremental = false
strip = "debuginfo"              # 剥离调试信息

[profile.bench]
inherits = "release"             # 继承release配置
debug = true                     # 覆盖特定配置

# 测试模式
[profile.test]
inherits = "dev"
opt-level = 2          # 测试时更高优化

    2.构建脚本¶

    [package]
build = "build.rs"              # 构建脚本路径

# 构建依赖
[build-dependencies]
cc = "1.0"
cmake = "0.1"

    五、高级配置¶

    工作区（Workspace）配置¶

    # 根目录 Cargo.toml
[workspace]
members = [
    "core",
    "cli",
    "tests"，
    "crates/*",          # 包含所有子crate
]
# 排除目录
exclude = ["temp/*","crates/old-crate"]
# 工作区依赖解析配置
resolver = "2"           # 依赖解析器版本（Cargo 1.51+）
# 工作区默认成员
default-members = ["crates/core"]

# 工作区包配置
[workspace.package]
version = "0.1.0"
authors = ["Workspace Maintainers <team@example.com>"]
license = "MIT OR Apache-2.0"
edition = "2021"

# 工作区依赖（所有成员共享）
[workspace.dependencies]
serde = { version = "1.0", features = ["derive"] }
tokio = { version = "1.0", features = ["full"] }

# 工作区构建依赖
[workspace.build-dependencies]
vergen = "7.0"

# 子项目继承依赖
[dependencies]
serde.workspace = true
tokio.workspace = true

    补丁和替换¶

    # 临时修复上游 bug
[patch.crates-io]
uuid = { git = "https://github.com/uuid-rs/uuid", branch = "fix-branch" }

# 本地开发版本
[patch.crates-io]
my_dep = { path = "../my_dep" }

    继承¶

    工作区继承¶

    [package]
name = "toolchain-parser"
# 继承工作区配置
version.workspace = true       # 从工作区继承版本号
edition.workspace = true       # 从工作区继承 Rust 版本（edition）
publish.workspace = true       # 从工作区继承发布配置
rust-version.workspace = true  # 从工作区继承最低支持的 Rust 版本

license.workspace = true
authors.workspace = true
# 包特定的元数据
description = "语法分析器模块"
readme.workspace = true

[dependencies]
# 包特定依赖
nom = "7.1"
# 内部依赖（使用路径）
toolchain-parser = { path = "../parser" }

# 外部依赖
regex = "1.9"

# 特性
[features]
default = []
json = ["serde/json"]

[lib]
name = "compiler"
path = "src/lib.rs"
crate-type = ["rlib", "cdylib"]

    依赖继承¶

    # 继承工作区依赖
[dependencies]
anyhow.workspace = true
thiserror.workspace = true
tracing.workspace = true
serde.workspace = true
tokio.workspace = true

# 继承并启用可选特性
clap = { workspace = true, optional = true }

# 继承并添加额外特性
serde_json = { workspace = true, features = ["preserve_order"] }

    七、Cargo 命令使用¶

    7.1 项目创建与管理¶

    # 创建新项目
cargo new my-project           # 二进制项目
cargo new --lib my-lib         # 库项目
cargo new --vcs git my-project # 使用Git初始化

# 在现有目录初始化
cargo init

# 添加依赖
cargo add serde                # 添加最新版本
cargo add serde@1.0            # 指定版本
cargo add serde --features derive  # 启用特性
cargo add --dev tokio-test     # 添加开发依赖
cargo add --build cc           # 添加构建依赖

# 移除依赖
cargo remove serde

    7.2 构建与运行¶

    1.cargo¶

    # 构建项目
cargo build                   # 开发构建
cargo build --release         # 发布构建
cargo build --profile bench   # 使用特定profile

# 运行项目
cargo run                     # 运行主二进制文件
cargo run --bin myapp         # 运行指定二进制文件
cargo run --release           # 发布模式运行

# 检查代码（快速编译检查）
cargo check
cargo check --release

    2.单文件运行¶

    1. rust-script 或 cargo-script

    2. 安装

      # 安装
cargo install rust-script # or cargo-script

# 直接运行 .rs 文件
rust-script hello.rs

    3. 支持 Shebang（文件首行）

      #!/usr/bin/env rust-script
fn main() {
    println!("Hello!");
}

// 带依赖
#!/usr/bin/env rust-script
//! ```cargo
//! [dependencies]
//! reqwest = { version = "0.11", features = ["blocking"] }
//! ```

fn main() {
    let resp = reqwest::blocking::get("https://api.github.com").unwrap();
    println!("{:?}", resp.status());
}

    4. rustc直接编译

    # 编译
rustc hello.rs

# 运行
./hello      # Linux/Mac
hello.exe    # Windows
    1. examples目录中的示例运行
    cargo run --example auth_aksk

    关键点：

    • examples 下的每个 .rs 文件默认是 二进制目标（不是库目标）

    • 编译时使用 --example 告诉 Cargo "这是一个 example，不是我主的 lib/test"

    # 运行测试

    7.4 依赖管理¶

    # 更新依赖
cargo update                 # 更新所有依赖
cargo update -p serde        # 更新特定依赖

# 过时依赖检查

    # 发布准备

    八、最佳实践建议¶

    1. 精确版本控制: 避免使用 *，推荐使用 ^1.2.3 格式
    2. 最小化特性: 只启用必要的依赖特性，减小体积
    3. 分类管理: 严格区分生产、开发、构建依赖
    4. 工作区管理: 多 crate 项目使用工作区统一管理
    5. 定期更新: 使用 cargo update 定期更新依赖
    6. 文档注释: 对复杂配置添加注释说明用途
