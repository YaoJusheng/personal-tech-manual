    事件循环机制¶

    // 事件循环示例
console.log('1. 同步任务开始');

setTimeout(() => {
    console.log('2. 定时器任务');
}, 0);

Promise.resolve().then(() => {
    console.log('3. 微任务 - Promise');
});

console.log('4. 同步任务结束');

// 执行顺序：1 -> 4 -> 3 -> 2
// 同步 -> 微任务 -> 宏任务

    事件循环阶段：

    1. Timers：执行 setTimeout 和 setInterval 回调
    2. Pending I/O：执行 I/O 回调
    3. Idle/Prepare：内部使用
    4. Poll：获取新的 I/O 事件
    5. Check：setImmediate 回调执行
    6. Close Callbacks：关闭的回调函数

    ES Modules (ESM) 系统¶

    // math.mjs - ESM导出
export const PI = 3.14159;

export function add(a, b) {
    return a + b;
}

export function multiply(a, b) {
    return a * b;
}

// 默认导出
export default function calculate(operation, a, b) {
    switch (operation) {
        case 'add': return a + b;
        case 'multiply': return a * b;
        default: throw new Error('Unknown operation');
    }
}

// main.mjs - ESM导入
import calculate, { add, multiply } from './math.mjs';
import * as mathModule from './math.mjs';

console.log(add(2, 3)); // 5
console.log(mathModule.PI); // 3.14159

    文件系统操作 (fs & path)¶

    import fs from 'node:fs/promises';
import path from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));

// 异步文件操作
async function fileOperations() {
    try {
        // 创建目录
        const logsDir = path.join(__dirname, 'logs');
        await fs.mkdir(logsDir, { recursive: true });

        // 写入日志文件
        const logFile = path.join(logsDir, 'app.log');
        const timestamp = new Date().toISOString();
        const logEntry = `[${timestamp}] INFO: Application started\n`;

        await fs.appendFile(logFile, logEntry);

        // 读取文件内容
        const content = await fs.readFile(logFile, 'utf-8');
        console.log('日志内容:', content);

        // 文件状态检查
        const stats = await fs.stat(logFile);
        console.log('文件大小:', stats.size, 'bytes');
        console.log('创建时间:', stats.birthtime);
        console.log('修改时间:', stats.mtime);

        // 列出目录内容
        const files = await fs.readdir(logsDir);
        console.log('目录文件:', files);

    } catch (error) {
        console.error('文件操作失败:', error.message);
    }
}

// 文件监听
async function watchFiles() {
    const watchDir = path.join(__dirname, 'watch');
    await fs.mkdir(watchDir, { recursive: true });

    // 监听目录变化
    const watcher = fs.watch(watchDir, (eventType, filename) => {
        console.log(`文件变化: ${eventType}, 文件名: ${filename}`);
    });

    // 创建测试文件
    await fs.writeFile(path.join(watchDir, 'test.txt'), 'Hello Watch');

    // 5秒后清理
    setTimeout(() => {
        watcher.close();
        console.log('文件监听已停止');
    }, 5000);
}

    HTTP 服务器与客户端¶

    import http from 'node:http';
import { createReadStream } from 'node:fs';
import { pipeline } from 'node:stream/promises';
import fs from 'node:fs/promises';

// HTTP 服务器
const server = http.createServer(async (req, res) => {
    // CORS 头设置
    res.setHeader('Access-Control-Allow-Origin', '*');
    res.setHeader('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE');
    res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization');

    // 路由处理
    if (req.url === '/health') {
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ status: 'ok', timestamp: new Date().toISOString() }));
        return;
    }

    if (req.url === '/video' && req.method === 'GET') {
        // 流式文件传输
        const videoPath = './big-video.mp4';
        try {
            await fs.access(videoPath);
            res.writeHead(200, {
                'Content-Type': 'video/mp4',
                'Content-Length': (await fs.stat(videoPath)).size
            });
            await pipeline(createReadStream(videoPath), res);
        } catch (error) {
            res.writeHead(404, { 'Content-Type': 'text/plain' });
            res.end('Video file not found');
        }
        return;
    }

    if (req.url === '/api/users' && req.method === 'GET') {
        // API 响应
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({
            users: [
                { id: 1, name: 'Alice', email: 'alice@example.com' },
                { id: 2, name: 'Bob', email: 'bob@example.com' }
            ]
        }));
        return;
    }

    if (req.url === '/api/users' && req.method === 'POST') {
        // 处理 POST 请求
        let body = '';
        req.on('data', chunk => {
            body += chunk.toString();
        });

        req.on('end', async () => {
            try {
                const userData = JSON.parse(body);
                console.log('创建用户:', userData);

                res.writeHead(201, { 'Content-Type': 'application/json' });
                res.end(JSON.stringify({
                    message: 'User created successfully',
                    user: { ...userData, id: Date.now() }
                }));
            } catch (error) {
                res.writeHead(400, { 'Content-Type': 'application/json' });
                res.end(JSON.stringify({ error: 'Invalid JSON' }));
            }
        });
        return;
    }

    // 默认响应
    res.writeHead(404, { 'Content-Type': 'text/plain' });
    res.end('Not Found');
});

const PORT = process.env.PORT || 3000;
server.listen(PORT, () => {
    console.log(`HTTP服务器运行在端口 ${PORT}`);
});

// HTTP 客户端
async function httpClientExamples() {
    // 使用内置 http 模块
    const userData = JSON.stringify({
        name: 'Charlie',
        email: 'charlie@example.com'
    });

    const options = {
        hostname: 'localhost',
        port: 3000,
        path: '/api/users',
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'Content-Length': Buffer.byteLength(userData)
        }
    };

    const req = http.request(options, (res) => {
        console.log(`状态码: ${res.statusCode}`);

        let data = '';
        res.on('data', (chunk) => {
            data += chunk;
        });

        res.on('end', () => {
            console.log('响应:', data);
        });
    });

    req.on('error', (error) => {
        console.error('请求错误:', error.message);
    });

    req.write(userData);
    req.end();
}

    流处理与管道¶

    import { createReadStream, createWriteStream } from 'node:fs';
import { pipeline, Transform } from 'node:stream/promises';
import { createGzip, createGunzip } from 'node:zlib';
import { TransformStream } from 'node:stream/web';

// 自定义转换流
class UpperCaseTransform extends Transform {
    constructor(options) {
        super(options);
    }

    _transform(chunk, encoding, callback) {
        const upperCaseChunk = chunk.toString().toUpperCase();
        this.push(upperCaseChunk);
        callback();
    }
}

// 文件压缩示例
async function fileCompression() {
    try {
        const sourceFile = './large-file.txt';
        const compressedFile = './large-file.txt.gz';
        const decompressedFile = './decompressed-file.txt';

        // 压缩文件
        await pipeline(
            createReadStream(sourceFile),
            createGzip(),
            createWriteStream(compressedFile)
        );
        console.log('文件压缩完成');

        // 解压缩文件
        await pipeline(
            createReadStream(compressedFile),
            createGunzip(),
            createWriteStream(decompressedFile)
        );
        console.log('文件解压缩完成');

    } catch (error) {
        console.error('压缩操作失败:', error.message);
    }
}

// 流式数据处理
async function streamDataProcessing() {
    const transform = new TransformStream({
        transform(chunk, encoding, callback) {
            // 处理数据块
            const processed = chunk.toString().replace(/\n/g, '\r\n');
            callback(null, processed);
        }
    });

    await pipeline(
        process.stdin,
        transform,
        process.stdout
    );
}

    事件驱动编程¶

    import { EventEmitter } from 'node:events';
import { readFile } from 'node:fs/promises';

// 自定义事件类
class TaskManager extends EventEmitter {
    constructor() {
        super();
        this.tasks = new Map();
    }

    async addTask(taskId, taskData) {
        this.tasks.set(taskId, { ...taskData, status: 'pending', createdAt: new Date() });
        this.emit('task:added', { taskId, taskData });

        try {
            await this.processTask(taskId);
        } catch (error) {
            this.emit('task:error', { taskId, error: error.message });
        }
    }

    async processTask(taskId) {
        const task = this.tasks.get(taskId);
        if (!task) {
            throw new Error(`Task ${taskId} not found`);
        }

        this.emit('task:started', { taskId });

        try {
            // 模拟任务处理
            await new Promise(resolve => setTimeout(resolve, 1000));

            task.status = 'completed';
            task.completedAt = new Date();
            this.emit('task:completed', { taskId, task });

        } catch (error) {
            task.status = 'failed';
            task.error = error.message;
            this.emit('task:failed', { taskId, error: error.message });
            throw error;
        }
    }

    getTaskStatus(taskId) {
        return this.tasks.get(taskId);
    }

    listTasks() {
        return Array.from(this.tasks.entries()).map(([id, task]) => ({
            id,
            ...task
        }));
    }
}

// 使用事件管理器
async function eventDrivenExample() {
    const taskManager = new TaskManager();

    // 监听事件
    taskManager.on('task:added', ({ taskId }) => {
        console.log(`任务已添加: ${taskId}`);
    });

    taskManager.on('task:started', ({ taskId }) => {
        console.log(`任务开始执行: ${taskId}`);
    });

    taskManager.on('task:completed', ({ taskId, task }) => {
        console.log(`任务完成: ${taskId}`, task);
    });

    taskManager.on('task:error', ({ taskId, error }) => {
        console.error(`任务错误: ${taskId}`, error);
    });

    // 添加任务
    await taskManager.addTask('task-001', { name: '数据处理', priority: 'high' });
    await taskManager.addTask('task-002', { name: '文件备份', priority: 'medium' });

    // 列出所有任务
    console.log('所有任务:', taskManager.listTasks());

// 错误处理最佳实践
async function robustAsyncOperations() {
    const operations = [
        () => fs.readFile('file1.txt', 'utf-8'),
        () => fs.readFile('file2.txt', 'utf-8'),
        () => fs.readFile('file3.txt', 'utf-8')
    ];

    try {
        // 使用 Promise.allSettled 处理所有操作
        const results = await Promise.allSettled(
            operations.map(op => op())
        );

        const successful = results
            .filter(result => result.status === 'fulfilled')
            .map(result => result.value);

        const failed = results
            .filter(result => result.status === 'rejected')
            .map(result => result.reason);

        console.log('成功读取:', successful.length, '个文件');
        console.log('失败操作:', failed.length, '个');

        return { successful, failed };

    } catch (error) {
        console.error('批量操作失败:', error.message);
        throw error;
    }
}

    1. CPU 密集任务处理：
    import { Worker } from 'node:worker_threads';
import { availableParallelism } from 'node:os';

// Worker 池管理
class WorkerPool {
  constructor(workerScript, poolSize = availableParallelism()) {
    this.workers = [];
    this.queue = [];

    for (let i = 0; i < poolSize; i++) {
      this.workers.push(new Worker(workerScript));
    }
  }

  async execute(data) {
    return new Promise((resolve, reject) => {
      const worker = this.workers.find(w => !w.busy);

      if (worker) {
        worker.busy = true;
        worker.postMessage(data);
        worker.once('message', (result) => {
          worker.busy = false;
          resolve(result);
        });
        worker.once('error', (error) => {
          worker.busy = false;
          reject(error);
        });
      } else {
        this.queue.push({ data, resolve, reject });
      }
    });
  }

  terminate() {
    this.workers.forEach(worker => worker.terminate());
  }
}

// 重试机制
async function withRetry(operation, maxRetries = 3, delay = 1000) {
  let lastError;

  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await operation();
    } catch (error) {
      lastError = error;

      if (attempt < maxRetries) {
        console.log(`尝试 ${attempt} 失败，${delay}ms 后重试...`);
        await new Promise(resolve => setTimeout(resolve, delay));
        delay *= 2; // 指数退避
      }
    }
  }

  throw lastError;
}

// 熔断器模式
class CircuitBreaker {
  constructor(threshold = 5, timeout = 60000) {
    this.failureThreshold = threshold;
    this.timeout = timeout;
    this.failureCount = 0;
    this.lastFailureTime = null;
    this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
  }

  async execute(operation) {
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime > this.timeout) {
        this.state = 'HALF_OPEN';
      } else {
        throw new Error('Circuit breaker is OPEN');
      }
    }

    try {
      const result = await operation();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  onSuccess() {
    this.failureCount = 0;
    this.state = 'CLOSED';
  }

  onFailure() {
    this.failureCount++;
    this.lastFailureTime = Date.now();

    if (this.failureCount >= this.failureThreshold) {
      this.state = 'OPEN';
    }
  }
}
