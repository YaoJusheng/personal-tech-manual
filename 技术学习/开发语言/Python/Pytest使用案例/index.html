    2.4.9.模拟数据¶

    class Config:
    LICENSE_PATH = os.getenv("LICENSE_PATH") or "../app/.stub/lic.zip"
    AES_KEY = (os.getenv("AES_KEY") or "").encode() or os.urandom(32)

    def __init__(self, params: dict = None, data: dict = None, stamp: int = 0):
        self.params = params
        self.data = data
        self.stamp = stamp


def convert_to_py(data: dict):
    for k, v in data.items():
        if isinstance(v, datetime):
            data[k] = v.strftime("%Y-%m-%d %H:%M:%S")
        elif isinstance(v, Decimal):
            data[k] = str(v)
    return data


@pytest.fixture(scope="module", autouse=True)
def faker():

            data = {
                "name": faker.name(),
                "address": faker.address(),
                "email": faker.email(),
                "integer": faker.random_int(min=0, max=100),
                "boolean": faker.boolean()
            }
            return Config(params, data, stamp=time.time())

    faker.add_provider(ConfigProvider)
    yield faker

    # test_endpoints.py
import pytest

class TestAPIEndpoints:

    def test_read_root(self, test_client):
        response = test_client.get("/")
        assert response.status_code == 200
        assert response.json() == {"Hello": "World"}

    def test_create_item(self, test_client):
        response = test_client.post("/items/", json={"name": "item1", "price": 10.5})
        assert response.status_code == 200
        assert response.json() == {"name": "item1", "price": 10.5}

    def test_read_item(self, test_client):
        response = test_client.get("/items/1")
        assert response.status_code == 200
        assert response.json() == {"name": "item1", "price": 10.5}

    def test_update_item(self, test_client):
        response = test_client.put("/items/1", json={"name": "item1-updated", "price": 12.0})
        assert response.status_code == 200
        assert response.json() == {"name": "item1-updated", "price": 12.0}

    def test_delete_item(self, test_client):
        response = test_client.delete("/items/1")
        assert response.status_code == 200
        assert response.json() == {"ok": True}

    4.2.1.安装¶

    4.2.2.编写异步测试¶

    使用 pytest.mark.asyncio 标记异步测试函数：

    # conftest.py
import pytest
import httpx
from fastapi import FastAPI

app = FastAPI()

@app.get("/")
async def read_root():
    return {"Hello": "World"}

@pytest.fixture
async def async_client():
    async with httpx.AsyncClient(app=app, base_url="http://test") as client:
        yield client

# test_endpoints.py
import pytest

@pytest.mark.asyncio
async def test_read_root(test_client):
    response = await test_client.get("/")
    assert response.status_code == 200
    assert response.json() == {"Hello": "World"}

@pytest.mark.asyncio
async def test_create_item(test_client):
    response = await test_client.post("/items/", json={"name": "item1", "price": 10.5})
    assert response.status_code == 200
    assert response.json() == {"name": "item1", "price": 10.5"}

    4.2.3. 使用 httpx for Async Tests¶

    httpx 支持异步请求，可以替代 TestClient 进行更真实的异步测试：

    import pytest
import httpx
from app.main import app
from starlette.testclient import TestClient

@pytest.fixture(scope='module')
def test_app():
    client = TestClient(app)
    yield client

@pytest.mark.asyncio
async def test_read_root():
    async with httpx.AsyncClient(app=app, base_url="http://test") as client:
        response = await client.get("/")
    assert response.status_code == 200
    assert response.json() == {"Hello": "World"}

    4.2.4.最佳实践¶

    • 独立测试: 确保每个测试用例独立运行。
    • 使用异步库: 利用 httpx 等异步库进行更真实的测试。
    • 复用 Fixtures: 利用 pytest 的 fixtures 提供共享的测试资源。
    • 详细断言: 确保断言语句足够详细，以验证响应的正确性。
