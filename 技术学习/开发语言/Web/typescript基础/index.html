    TypeScript 语言特性与优势¶

    为何使用 TypeScript¶

    • 类型安全：在编译阶段捕获潜在错误，提升维护性。
    • 工程化能力：接口（interface）、枚举、泛型等让大型项目结构清晰。
    • 工具链支持：VS Code/JetBrains 拥有一流智能提示与重构能力。
    • 生态兼容：能够与现有 JavaScript 库和平共存，逐步迁移。
    • 现代 JavaScript 支持：原生支持 ES6+ 特性，自动转译为兼容版本。

    TypeScript 编译流程¶

    # 安装 TypeScript
npm install -g typescript

# 编译单个文件
tsc hello.ts

# 监视模式编译
tsc --watch

# 编译整个项目
tsc

# 生成 tsconfig.json
tsc --init

    基础类型系统¶

    // 基础类型
let isDone: boolean = false;
let decimal: number = 6;
let hex: number = 0xf00d;
let binary: number = 0b1010;
let octal: number = 0o744;
let color: string = "blue";
let list: number[] = [1, 2, 3];
let listGeneric: Array<number> = [1, 2, 3];

// 元组类型
let x: [string, number];
x = ["hello", 10]; // 正确
x = [10, "hello"]; // 错误

// 枚举类型
enum Color {
  Red = 1,
  Green = 2,
  Blue = 4
}
let c: Color = Color.Green;

// any 类型（谨慎使用）
let notSure: any = 4;
notSure = "maybe a string instead";
notSure = false;

// unknown 类型（更安全的 any）
let value: unknown = 4;
value = "maybe a string instead";

// void 类型
function warnUser(): void {
  console.log("This is a warning");
}

// null 和 undefined
let u: undefined = undefined;
let n: null = null;

// never 类型
function error(message: string): never {
  throw new Error(message);
}

// 类型断言
let someValue: any = "this is a string";
let strLength: number = (someValue as string).length;

// 联合类型
let unionType: string | number;
unionType = "hello";
unionType = 42;

// 类型守卫
function isString(value: unknown): value is string {
  return typeof value === "string";
}

interface User {
  readonly id: string; // 只读属性
  name: string;
  email: string;
  age?: number; // 可选属性
  greet(): string; // 方法签名
}

// 接口继承
interface Employee extends User {
  employeeId: string;
  department: string;
}

// 类实现接口
class RegularUser implements User {
  readonly id: string;
  name: string;
  email: string;

  constructor(id: string, name: string, email: string) {
    this.id = id;
    this.name = name;
    this.email = email;
  }

  greet(): string {
    return `Hello, I'm ${this.name}`;
  }
}

    联合类型与类型守卫详解¶

    // 联合类型
type ID = string | number;

function normalizeId(id: ID): string {
  return typeof id === 'string' ? id : id.toString();
}

// 类型守卫
function isUser(value: unknown): value is User {
  return typeof value === 'object' &&
         value !== null &&
         'name' in value &&
         'email' in value;
}

// 条件类型
type NonNullable<T> = T extends null | undefined ? never : T;

function processValue<T>(value: T): NonNullable<T> {
  if (value === null || value === undefined) {
    throw new Error('Value cannot be null or undefined');
  }
  return value;
}

    泛型详解与约束¶

    // 基础泛型
function identity<T>(arg: T): T {
  return arg;
}

let stringResult = identity<string>("myString");
let numberResult = identity<number>(100);

// 泛型约束
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

interface User {
  id: number;
  name: string;
  email: string;
}

const user: User = {
  id: 1,
  name: "John",
  email: "john@example.com"
};

const userName = getProperty(user, "name"); // 类型: string

// 泛型类
class GenericRepository<T> {
  private items: T[] = [];

  add(item: T): void {
    this.items.push(item);
  }

  get(index: number): T | undefined {
    return this.items[index];
  }

  getAll(): T[] {
    return [...this.items];
  }
}

const userRepository = new GenericRepository<User>();
userRepository.add(user);
const allUsers = userRepository.getAll();

// 泛型接口
interface Repository<T> {
  add(item: T): void;
  get(index: number): T | undefined;
  getAll(): T[];
}

class UserRepository implements Repository<User> {
  private users: User[] = [];

  add(item: User): void {
    this.users.push(item);
  }

  get(index: number): User | undefined {
    return this.users[index];
  }

  getAll(): User[] {
    return [...this.users];
  }
}

    高级类型工具详解¶

    Utility Types¶

    interface User {
  id: string;
  name: string;
  email: string;
  age: number;
  isActive: boolean;
}

// Partial - 所有属性变为可选
type PartialUser = Partial<User>;
// { id?: string; name?: string; email?: string; age?: number; isActive?: boolean }

// Required - 所有属性变为必需
type RequiredUser = Required<PartialUser>;

// Readonly - 所有属性变为只读
type ReadonlyUser = Readonly<User>;

// Pick - 选择特定属性
type UserSummary = Pick<User, 'id' | 'name' | 'email'>;

// Omit - 排除特定属性
type UserWithoutId = Omit<User, 'id'>;

// Record - 创建键值对类型
type UserMap = Record<string, User>;

// ReturnType - 获取函数返回类型
function createUser(name: string, email: string): User {
  return {
    id: Math.random().toString(),
    name,
    email,
    age: 0,
    isActive: true
  };
}

type CreateUserReturn = ReturnType<typeof createUser>;
// { id: string; name: string; email: string; age: number; isActive: boolean }

// Parameters - 获取函数参数类型
type CreateUserParams = Parameters<typeof createUser>;
// [name: string, email: string]

    条件类型详解¶

    // 基础条件类型
type IsString<T> = T extends string ? true : false;

type Test1 = IsString<string>; // true
type Test2 = IsString<number>; // false

// 分布式条件类型
type ToArray<T> = T extends any ? T[] : never;

type StringArrayOr = ToArray<string | number>;
// string[] | number[]

// 条件类型与泛型结合
type ApiResponse<T> = T extends string ? { data: T; status: 'success' } :
                     T extends number ? { data: T; status: 'success' } :
                     { data: T; status: 'error' };

type StringResponse = ApiResponse<string>;
// { data: string; status: 'success' }

type NumberResponse = ApiResponse<number>;
// { data: number; status: 'success' }

type ObjectResponse = ApiResponse<object>;
// { data: object; status: 'error' }

// 递归条件类型
type DeepReadonly<T> = T extends object
  ? { readonly [K in keyof T]: DeepReadonly<T[K]> }
  : T;

interface NestedObject {
  user: {
    name: string;
    address: {
      street: string;
      city: string;
    };
  };
}

type ReadonlyNested = DeepReadonly<NestedObject>;
// {
//   readonly user: {
//     readonly name: string;
//     readonly address: {
//       readonly street: string;
//       readonly city: string;
//     };
//   };
// }

    映射类型详解¶

    // 基础映射类型
type MappedType<T> = {
  [K in keyof T]: T[K];
};

// 属性重映射
type RenameKeys<T, R extends { [K in keyof T]: string }> = {
  [K in keyof T as R[K]]: T[K];
};

interface Original {
  id: number;
  name: string;
  email: string;
}

type Renamed = RenameKeys<Original, { id: 'userId'; name: 'fullName'; email: 'emailAddress' }>;
// {
//   userId: number;
//   fullName: string;
//   emailAddress: string;
// }

// 属性修饰符
type MakeOptional<T> = {
  [K in keyof T]?: T[K];
};

type MakeRequired<T> = {
  [K in keyof T]-?: T[K];
};

type MakeReadonly<T> = {
  readonly [K in keyof T]: T[K];
};

    模板字面量类型¶

    // 基础模板字面量类型
type EventName = `on${Capitalize<string>}`;
// "onClick" | "onHover" | "onFocus" | ...

// 联合模板字面量类型
type HttpMethod = 'GET' | 'POST' | 'PUT' | 'DELETE';
type ApiEndpoint = `/api/${HttpMethod}`;
// "/api/GET" | "/api/POST" | "/api/PUT" | "/api/DELETE"

// 复杂模板字面量类型
type UserAction =
  | `create${Capitalize<string>}`
  | `update${Capitalize<string>}`
  | `delete${Capitalize<string>}`;

type UserCreateAction = UserAction;
// "createUser" | "createProduct" | "createOrder" | ...

    类型级编程¶

    // 类型级计算
type Add<T extends number, U extends number> = [...Array<T>, ...Array<U>]['length'];

type Sum = Add<5, 3>; // 8

// 斐波那契数列类型
type Fibonacci<N extends number, Prev extends number = 0, Curr extends number = 1> =
  N extends 0 ? Prev :
  N extends 1 ? Curr :
  Fibonacci<N - 1, Curr, Add<Prev, Curr>>;

type Fib5 = Fibonacci<5>; // 5

      details: err.message

    error: 'Internal server error'
  });
};

// user.service.ts
import { Injectable, NotFoundException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { User } from './user.entity.js';

@Injectable()
export class UserService {
  constructor(
    @InjectRepository(User)
    private readonly userRepository: Repository<User>
  ) {}

  async createUser(userData: CreateUserDto): Promise<User> {
    const user = this.userRepository.create(userData);
    return await this.userRepository.save(user);
  }

  async findUserById(id: string): Promise<User> {
    const user = await this.userRepository.findOne({ where: { id } });

    if (!user) {
      throw new NotFoundException(`User with ID ${id} not found`);
    }

    return user;
  }

  async updateUser(id: string, updateData: UpdateUserDto): Promise<User> {
    const user = await this.findUserById(id);

    Object.assign(user, updateData);
    return await this.userRepository.save(user);
  }
}

    @Body() updateUserDto: UpdateUserDto
  ) {
    return await this.userService.updateUser(id, updateUserDto);
  }
