    1.简介¶

    Actix-web 是一个基于 Actix actor 系统的强大、实用且快速的 Rust Web 框架。

    特点¶

    • 高性能：在多个基准测试中名列前茅
    • 类型安全：利用 Rust 的类型系统确保安全性
    • 异步支持：基于 tokio 的异步运行时
    • 丰富的中间件：内置多种中间件支持
    • WebSocket 支持：原生 WebSocket 支持
    • HTTP/2 支持：完整的 HTTP/2 实现

    官方资源¶

    • 官网：https://actix.rs/
    • 文档：https://actix.rs/docs/
    • GitHub：https://github.com/actix/actix-web

    3.使用¶

    基础 Hello World¶

    use actix_web::{get, web, App, HttpResponse, HttpServer, Responder};

#[get("/")]
async fn hello() -> impl Responder {
    HttpResponse::Ok().body("Hello world!")
}

#[get("/echo/{name}")]
async fn echo(path: web::Path<String>) -> impl Responder {
    HttpResponse::Ok().body(format!("Hello {}!", path))
}

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| {
        App::new()
            .service(hello)
            .service(echo)
    })
    .bind(("127.0.0.1", 8080))?
    .run()
    .await
}

    JSON 请求和响应¶

    use actix_web::{post, web, App, HttpResponse, HttpServer, Result};
use serde::{Deserialize, Serialize};

#[derive(Deserialize, Serialize)]
struct User {
    id: u32,
    name: String,
    email: String,
}

#[post("/users")]
async fn create_user(user: web::Json<User>) -> Result<HttpResponse> {
    Ok(HttpResponse::Created().json(user.into_inner()))
}

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| {
        App::new()
            .service(create_user)
    })
    .bind(("127.0.0.1", 8080))?
    .run()
    .await
}

    4.进阶配置¶

    4.1 基础配置¶

    use actix_web::{middleware, web, App, HttpServer};
use std::env;

#[derive(Clone)]
struct AppConfig {
    database_url: String,
    redis_url: String,
    jwt_secret: String,
}

impl AppConfig {
    fn from_env() -> Self {
        dotenv::dotenv().ok();
        Self {
            database_url: env::var("DATABASE_URL").expect("DATABASE_URL must be set"),
            redis_url: env::var("REDIS_URL").unwrap_or("redis://127.0.0.1/".to_string()),
            jwt_secret: env::var("JWT_SECRET").expect("JWT_SECRET must be set"),
        }
    }
}

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    env_logger::init_from_env(env_logger::Env::new().default_filter_or("info"));

    let config = AppConfig::from_env();

    HttpServer::new(move || {
        App::new()
            .app_data(web::Data::new(config.clone()))
            .wrap(middleware::Logger::default())
            .wrap(middleware::Compress::default())
    })
    .bind(("0.0.0.0", 8080))?
    .workers(4)
    .run()
    .await
}

    4.2 中间件¶

    内置中间件¶

    use actix_web::{middleware, App, HttpServer};
use actix_cors::Cors;

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| {
        App::new()
            // 日志中间件
            .wrap(middleware::Logger::default())
            // 压缩中间件
            .wrap(middleware::Compress::default())
            // 规范化路径
            .wrap(middleware::NormalizePath::trim())
            // CORS
            .wrap(
                Cors::default()
                    .allowed_origin("http://localhost:3000")
                    .allowed_methods(vec!["GET", "POST", "PUT", "DELETE"])
                    .allowed_headers(vec!["Authorization", "Content-Type"])
                    .max_age(3600)
            )
    })
    .bind(("127.0.0.1", 8080))?
    .run()
    .await
}

    自定义中间件¶

    use actix_web::{
    dev::{forward_ready, Service, ServiceRequest, ServiceResponse, Transform},
    Error, HttpMessage,
};
use futures_util::future::LocalBoxFuture;
use std::future::{ready, Ready};

pub struct AuthMiddleware;

impl<S, B> Transform<S, ServiceRequest> for AuthMiddleware
where
    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error>,
    S::Future: 'static,
    B: 'static,
{
    type Response = ServiceResponse<B>;
    type Error = Error;
    type InitError = ();
    type Transform = AuthMiddlewareService<S>;
    type Future = Ready<Result<Self::Transform, Self::InitError>>;

    fn new_transform(&self, service: S) -> Self::Future {
        ready(Ok(AuthMiddlewareService { service }))
    }
}

pub struct AuthMiddlewareService<S> {
    service: S,
}

impl<S, B> Service<ServiceRequest> for AuthMiddlewareService<S>
where
    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error>,
    S::Future: 'static,
    B: 'static,
{
    type Response = ServiceResponse<B>;
    type Error = Error;
    type Future = LocalBoxFuture<'static, Result<Self::Response, Self::Error>>;

    forward_ready!(service);

    fn call(&self, req: ServiceRequest) -> Self::Future {
        let token = req
            .headers()
            .get("Authorization")
            .and_then(|v| v.to_str().ok())
            .and_then(|v| v.strip_prefix("Bearer "));

        if let Some(_token) = token {
            // 验证 token
            let fut = self.service.call(req);
            Box::pin(async move {
                let res = fut.await?;
                Ok(res)
            })
        } else {
            Box::pin(async {
                Err(actix_web::error::ErrorUnauthorized("Unauthorized"))
            })
        }
    }
}

    4.3 路由处理¶

    路由分组和作用域¶

    use actix_web::{web, App, HttpResponse, HttpServer};

// API v1 handlers
async fn v1_users() -> HttpResponse {
    HttpResponse::Ok().json(serde_json::json!({"version": "v1", "users": []}))
}

async fn v1_posts() -> HttpResponse {
    HttpResponse::Ok().json(serde_json::json!({"version": "v1", "posts": []}))
}

// API v2 handlers
async fn v2_users() -> HttpResponse {
    HttpResponse::Ok().json(serde_json::json!({"version": "v2", "users": []}))
}

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| {
        App::new()
            .service(
                web::scope("/api/v1")
                    .route("/users", web::get().to(v1_users))
                    .route("/posts", web::get().to(v1_posts))
            )
            .service(
                web::scope("/api/v2")
                    .route("/users", web::get().to(v2_users))
            )
    })
    .bind(("127.0.0.1", 8080))?
    .run()
    .await
}

    路由守卫¶

    use actix_web::{guard, web, App, HttpResponse, HttpServer};

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| {
        App::new()
            .service(
                web::resource("/api/data")
                    .route(
                        web::get()
                            .guard(guard::Header("content-type", "application/json"))
                            .to(|| async { HttpResponse::Ok().json("JSON only") })
                    )
                    .route(
                        web::post()
                            .guard(guard::fn_guard(|ctx| {
                                ctx.head().headers().contains_key("X-Api-Key")
                            }))
                            .to(|| async { HttpResponse::Created().finish() })
                    )
            )
    })
    .bind(("127.0.0.1", 8080))?
    .run()
    .await
}

    4.4 ORM 选型和实现¶

    SeaORM 集成¶

    SeaORM 是推荐的异步 ORM，功能完整且类型安全。

    use sea_orm::{Database, DatabaseConnection, EntityTrait, Set};
use actix_web::{web, App, HttpResponse, HttpServer};

    data: Option<T>,
    error: Option<String>,
}

    fn success(data: T) -> Self {
        Self {
            success: true,
            data: Some(data),
            error: None,
        }
    }

    fn error(message: String) -> Self {
        Self {
            success: false,
            data: None,
            error: Some(message),
        }
    }
}

    4.10 JWT 认证方案¶

    use actix_web::{web, App, HttpResponse, HttpServer, HttpRequest, Error};
use jsonwebtoken::{decode, encode, DecodingKey, EncodingKey, Header, Validation};
use serde::{Deserialize, Serialize};
use chrono::{Duration, Utc};

#[derive(Debug, Serialize, Deserialize)]
struct Claims {
    sub: String,
    exp: usize,
    iat: usize,
}

struct JwtConfig {
    secret: String,
}

fn create_jwt(user_id: &str, secret: &str) -> Result<String, jsonwebtoken::errors::Error> {
    let expiration = Utc::now()
        .checked_add_signed(Duration::hours(24))
        .expect("valid timestamp")
        .timestamp();

    let claims = Claims {
        sub: user_id.to_owned(),
        exp: expiration as usize,
        iat: Utc::now().timestamp() as usize,
    };

    encode(&Header::default(), &claims, &EncodingKey::from_secret(secret.as_ref()))
}

fn verify_jwt(token: &str, secret: &str) -> Result<Claims, jsonwebtoken::errors::Error> {
    decode::<Claims>(
        token,
        &DecodingKey::from_secret(secret.as_ref()),
        &Validation::default(),
    )
    .map(|data| data.claims)
}

#[derive(Deserialize)]
struct LoginRequest {
    username: String,
    password: String,
}

async fn login(
    credentials: web::Json<LoginRequest>,
    config: web::Data<JwtConfig>,
) -> HttpResponse {
    // 验证用户名和密码（这里简化处理）
    if credentials.username == "admin" && credentials.password == "password" {
        match create_jwt(&credentials.username, &config.secret) {
            Ok(token) => HttpResponse::Ok().json(serde_json::json!({
                "token": token
            })),
            Err(_) => HttpResponse::InternalServerError().finish(),
        }
    } else {
        HttpResponse::Unauthorized().finish()
    }
}

async fn protected_route(
    req: HttpRequest,
    config: web::Data<JwtConfig>,
) -> Result<HttpResponse, Error> {
    let token = req
        .headers()
        .get("Authorization")
        .and_then(|h| h.to_str().ok())
        .and_then(|h| h.strip_prefix("Bearer "))
        .ok_or_else(|| actix_web::error::ErrorUnauthorized("Missing token"))?;

    match verify_jwt(token, &config.secret) {
        Ok(claims) => Ok(HttpResponse::Ok().json(serde_json::json!({
            "message": "Access granted",
            "user": claims.sub
        }))),
        Err(_) => Err(actix_web::error::ErrorUnauthorized("Invalid token")),
    }
}

    use actix_multipart::Multipart;
use actix_web::{post, web, Error, HttpResponse};
use futures_util::TryStreamExt;
use std::io::Write;

#[post("/upload")]
async fn upload(mut payload: Multipart) -> Result<HttpResponse, Error> {
    while let Some(mut field) = payload.try_next().await? {
        let content_disposition = field.content_disposition();
        let filename = content_disposition
            .get_filename()
            .ok_or_else(|| actix_web::error::ErrorBadRequest("Missing filename"))?;

        let filepath = format!("./uploads/{}", filename);
        let mut f = web::block(|| std::fs::File::create(filepath))
            .await??;

        while let Some(chunk) = field.try_next().await? {
            f = web::block(move || f.write_all(&chunk).map(|_| f)).await??;
        }
    }

    Ok(HttpResponse::Ok().json(serde_json::json!({"status": "uploaded"})))
}

    4.21 健康检查¶

    use actix_web::{get, HttpResponse};
use serde_json::json;

#[get("/health")]
async fn health_check() -> HttpResponse {
    HttpResponse::Ok().json(json!({
        "status": "healthy",
        "timestamp": chrono::Utc::now().to_rfc3339()
    }))
}

#[get("/ready")]
async fn readiness_check(pool: web::Data<DbPool>) -> HttpResponse {
    // 检查数据库连接
    if pool.get().is_ok() {
        HttpResponse::Ok().json(json!({"status": "ready"}))
    } else {
        HttpResponse::ServiceUnavailable().json(json!({"status": "not ready"}))
    }
}

    4.22 优雅关闭¶

    use actix_web::{App, HttpServer};
use tokio::signal;

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    let server = HttpServer::new(|| {
        App::new()
            .route("/", web::get().to(|| async { "Hello!" }))
    })
    .bind(("127.0.0.1", 8080))?
    .run();

    let server_handle = server.handle();

    tokio::spawn(async move {
        signal::ctrl_c().await.expect("Failed to listen for Ctrl+C");
        println!("Shutting down gracefully...");
        server_handle.stop(true).await;
    });

    server.await
}

    4.23 环境配置管理¶

    use config::{Config, ConfigError, Environment, File};
use serde::Deserialize;

#[derive(Debug, Deserialize)]
struct AppSettings {
    server: ServerSettings,
    database: DatabaseSettings,
}

#[derive(Debug, Deserialize)]
struct ServerSettings {
    host: String,
    port: u16,
}

#[derive(Debug, Deserialize)]
struct DatabaseSettings {
    url: String,
    max_connections: u32,
}

impl AppSettings {
    fn new() -> Result<Self, ConfigError> {
        let env = std::env::var("RUN_ENV").unwrap_or_else(|_| "development".into());

        let s = Config::builder()
            .add_source(File::with_name("config/default"))
            .add_source(File::with_name(&format!("config/{}", env)).required(false))
            .add_source(Environment::with_prefix("APP"))
            .build()?;

        s.try_deserialize()
    }
}

FROM rust:1.75 as builder
