    1.з®Җд»Ӣ¶

    Rocket жҳҜдёҖдёӘжҲҗзҶҹгҖҒз”ҹдә§е°ұз»Әзҡ„ Rust Web жЎҶжһ¶пјҢд»Ҙе…¶жҳ“з”ЁжҖ§е’Ңзұ»еһӢе®үе…ЁиҖҢй—»еҗҚгҖӮ

    зү№зӮ№¶

    • зұ»еһӢе®үе…Ёпјҡзј–иҜ‘ж—¶дҝқиҜҒиҜ·жұӮеӨ„зҗҶзҡ„жӯЈзЎ®жҖ§
    • жҳ“дәҺдҪҝз”Ёпјҡзӣҙи§Ӯзҡ„е®Ҹе’Ң API и®ҫи®Ў
    • еҸҜжү©еұ•пјҡзҒөжҙ»зҡ„иҜ·жұӮе®ҲеҚ«зі»з»ҹ
    • ејӮжӯҘж”ҜжҢҒпјҡе®Ңж•ҙзҡ„ејӮжӯҘ I/O ж”ҜжҢҒ
    • жЁЎжқҝеј•ж“ҺпјҡеҶ…зҪ®еӨҡз§ҚжЁЎжқҝеј•ж“Һж”ҜжҢҒ
    • иЎЁеҚ•еӨ„зҗҶпјҡејәеӨ§зҡ„иЎЁеҚ•йӘҢиҜҒе’ҢеӨ„зҗҶ

    е®ҳж–№иө„жәҗ¶

    • е®ҳзҪ‘пјҡhttps://rocket.rs/
    • ж–ҮжЎЈпјҡhttps://rocket.rs/guide/
    • GitHubпјҡhttps://github.com/rwf2/Rocket

    еҹәзЎҖ Hello World¶

    #[macro_use] extern crate rocket;

#[get("/")]
fn index() -> &'static str {
    "Hello, world!"
}

#[get("/hello/<name>")]
fn hello(name: &str) -> String {
    format!("Hello, {}!", name)
}

#[launch]
fn rocket() -> _ {
    rocket::build()
        .mount("/", routes![index, hello])
}

struct AuthToken(String);

#[rocket::async_trait]
impl<'r> FromRequest<'r> for AuthToken {
    type Error = ();

    async fn from_request(req: &'r Request<'_>) -> request::Outcome<Self, Self::Error> {
        match req.headers().get_one("Authorization") {
            Some(token) if token.starts_with("Bearer ") => {
                Outcome::Success(AuthToken(token[7..].to_string()))
            }
            _ => Outcome::Error((Status::Unauthorized, ())),
        }
    }
}

#[get("/protected")]
fn protected(token: AuthToken) -> String {
    format!("Token: {}", token.0)
}

use rocket::serde::{Deserialize, Serialize};
use rocket::State;

#[derive(Debug, Serialize, Deserialize)]
struct Claims {
    sub: String,
    exp: usize,
}

struct JwtSecret(String);

#[post("/login", data = "<credentials>")]
fn login(credentials: Json<LoginRequest>, secret: &State<JwtSecret>) -> Json<serde_json::Value> {
    let claims = Claims {
        sub: credentials.username.clone(),
        exp: (chrono::Utc::now() + chrono::Duration::hours(24)).timestamp() as usize,
    };

    let token = encode(
        &Header::default(),
        &claims,
        &EncodingKey::from_secret(secret.0.as_ref())
    ).unwrap();

    Json(serde_json::json!({ "token": token }))
}

#[serde(crate = "rocket::serde")]
struct LoginRequest {
    username: String,
    password: String,
