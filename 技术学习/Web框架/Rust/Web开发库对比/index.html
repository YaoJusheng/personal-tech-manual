    特性 Tokio Hyper async-std

    在实际项目中，它们通常是组合出现的，而非“三选一”：

    1. Tokio + Hyper（主流组合）：这是构建高性能 HTTP 服务的黄金标准。Tokio 提供任务调度和异步 I/O 驱动，Hyper 处理 HTTP 协议层。许多知名框架（如 Axum）都基于此组合构建-4。
    2. Async-std + 兼容库：你可以使用 async-std 作为运行时，并搭配能在其上运行的 HTTP 库（例如 surf）。不过，由于 Hyper 与 Tokio 深度集成-3，要直接使用 Hyper 可能需要额外的兼容层（如 async-compat）-2。
    3. 单独使用：单独使用 Tokio 或 async-std 可以处理文件I/O、定时器、自定义协议等；单独使用 Hyper 则需要你为其提供一个运行时环境。

    核心理由 路由定义极简（get("/", handler)），中间件按需添加，无多余抽象，适合快速搭建单文件 API，支持 async/await 语法。 设计极简（Route::new().at("/", get(handler))），基于 async-std，API 简洁，适合轻量无依赖场景。

    文档质量 ★★★☆☆（核心功能覆盖，示例简洁） ★★★☆☆（文档基础，部分功能需查源码）

    Server::new(acceptor).serve(router).await;

impl ServerHook for SendBodyMiddleware {

    }

            .await
            .set_response_status_code(200)
            .await
            .set_response_header(SERVER, HYPERLANE)
            .await
            .set_response_header(CONNECTION, KEEP_ALIVE)
            .await
            .set_response_header(CONTENT_TYPE, TEXT_PLAIN)
            .await
            .set_response_header(ACCESS_CONTROL_ALLOW_ORIGIN, WILDCARD_ANY)
            .await
            .set_response_header("SocketAddr", &self.socket_addr)
            .await;
    }
}

impl ServerHook for UpgradeMiddleware {
    async fn new(_ctx: &Context) -> Self {
        Self
    }

    async fn handle(self, ctx: &Context) {
        if !ctx.get_request().await.is_ws() {
            return;
        }
        if let Some(key) = &ctx.try_get_request_header_back(SEC_WEBSOCKET_KEY).await {
            let accept_key: String = WebSocketFrame::generate_accept_key(key);
            ctx.set_response_status_code(101)
                .await
                .set_response_header(UPGRADE, WEBSOCKET)
                .await
                .set_response_header(CONNECTION, UPGRADE)
                .await
                .set_response_header(SEC_WEBSOCKET_ACCEPT, &accept_key)
                .await
                .set_response_body(&vec![])
                .await
                .send()
                .await
                .unwrap();
        }
    }
}

impl ServerHook for ResponseMiddleware {
    async fn new(_ctx: &Context) -> Self {
        Self
    }

            return;

impl ServerHook for RootRoute {
    async fn new(ctx: &Context) -> Self {
        let path: RequestPath = ctx.get_request_path().await;
        let response_body: String = format!("Hello hyperlane => {}", path);
        let cookie1: String = CookieBuilder::new("key1", "value1").http_only().build();
        let cookie2: String = CookieBuilder::new("key2", "value2").http_only().build();
        Self {
            response_body,
            cookie1,
            cookie2,
        }

        ctx.add_response_header(SET_COOKIE, &self.cookie1)
            .await
            .add_response_header(SET_COOKIE, &self.cookie2)
            .await
            .set_response_body(&self.response_body)
            .await;
    }
}

impl WebsocketRoute {
    async fn send_body_hook(&self, ctx: &Context) {
        let body: ResponseBody = ctx.get_response_body().await;
        if ctx.get_request().await.is_ws() {
            let frame_list: Vec<ResponseBody> = WebSocketFrame::create_frame_list(&body);
            ctx.send_body_list_with_data(&frame_list).await.unwrap();
        } else {
            ctx.send_body().await.unwrap();
        }
    }
}

impl ServerHook for WebsocketRoute {
    async fn new(_ctx: &Context) -> Self {
        Self
    }

    async fn handle(self, ctx: &Context) {
        while ctx.ws_from_stream(4096).await.is_ok() {
            let request_body: Vec<u8> = ctx.get_request_body().await;
            ctx.set_response_body(&request_body).await;
            self.send_body_hook(ctx).await;
        }
    }
}

            .await
            .set_response_header(SERVER, HYPERLANE)
            .await
            .set_response_header(CONTENT_TYPE, &self.content_type)
            .await
            .set_response_body(&self.response_body)
            .await
            .send()
            .await;
    }
}

#[tokio::main]
async fn main() {
    let config: ServerConfig = ServerConfig::new().await;
    config.host("0.0.0.0").await;
    config.port(60000).await;
    config.buffer(4096).await;
    config.disable_linger().await;
    config.disable_nodelay().await;

    server.request_middleware::<SendBodyMiddleware>().await;
    server.request_middleware::<UpgradeMiddleware>().await;
    server.response_middleware::<ResponseMiddleware>().await;
    server.panic_hook::<ServerPanicHook>().await;
    server.route::<RootRoute>("/").await;
    server.route::<WebsocketRoute>("/websocket").await;
    server.route::<SseRoute>("/sse").await;
    server.route::<DynamicRoute>("/dynamic/{routing}").await;
    server.route::<DynamicRoute>("/regex/{file:^.*$}").await;
    let server_lifecycle: ServerControlHook = server.run().await.unwrap_or_default();
    server_lifecycle.wait().await;
}

    3. 全局配置封装（src/config.rs 核心文件，契合你的项目结构）¶

    这是生产级最佳实践：将配置封装为全局单例，项目任意模块通过 CONFIG.database.url 读取，强类型、无冗余、统一管理。

    // src/config.rs
use anyhow::Result;
use config::{Config, ConfigError, Environment, File};
use once_cell::sync::Lazy;
use serde::Deserialize;
use std::path::Path;

// ========== 核心：定义和配置文件一一对应的强类型结构体 ==========
#[derive(Debug, Deserialize, Clone)]
pub struct AppConfig {
    pub server: ServerConfig,
    pub database: DatabaseConfig,
    pub redis: RedisConfig,
    pub log: LogConfig,
}

#[derive(Debug, Deserialize, Clone)]
pub struct ServerConfig {
    pub port: u16,
    pub host: String,
    pub workers: usize,
}

#[derive(Debug, Deserialize, Clone)]
pub struct DatabaseConfig {
    pub url: String,
    pub max_connections: u32,
}

#[derive(Debug, Deserialize, Clone)]
pub struct RedisConfig {
    pub url: String,
}

#[derive(Debug, Deserialize, Clone)]
pub struct LogConfig {
    pub level: String,
    pub format: String,
}

// ========== 全局单例：项目任意地方直接使用 CONFIG.xxx ==========
pub static CONFIG: Lazy<AppConfig> = Lazy::new(|| load_config().unwrap());

// ========== 核心：加载配置的核心逻辑 ==========
pub fn load_config() -> Result<AppConfig, ConfigError> {
    // 获取运行环境：development/production/test，默认development
    let env = std::env::var("APP_ENV").unwrap_or_else(|_| "development".into());
    let config_dir = Path::new("./config");

    // 加载配置的核心逻辑：优先级从低到高【默认配置 → 环境配置 → 环境变量】
    let config = Config::builder()
        // 1. 加载默认公共配置
        .add_source(File::from(config_dir.join("default.toml")).required(true))
        // 2. 加载对应环境的配置（开发/生产）
        .add_source(File::from(config_dir.join(format!("{}.toml", env))).required(true))
        // 3. 加载.env文件的环境变量，优先级更高
        .add_source(Environment::default().prefix("APP").separator("__"))
        // 4. 加载系统环境变量，优先级最高（部署时用）
        .add_source(Environment::default().separator("__"))
        .build()?;

    // 反序列化为强类型结构体
    config.try_deserialize::<AppConfig>()
}

    2.日志库工具¶

    Rust Web 项目的日志核心需求（生产级）

    ① 支持日志级别过滤（debug/info/warn/error），开发环境看 debug，生产环境只看 info+error；

    ② 支持结构化日志（JSON 格式），生产环境日志收集工具（ELK/Loki）必备；

    ③ 支持异步无阻塞输出，不影响 Web 服务的性能；

    ④ 支持上下文日志（比如请求 ID、用户 ID），便于问题排查；

    ⑤ 轻量、无冗余，日志输出性能高。

    日志级别规范（行业通用，必须遵守）

    trace < debug < info < warn < error
    • trace：极致细节日志，比如函数入参出参、循环步骤，仅本地调试用；
    • debug：调试日志，比如数据库 SQL 语句、接口请求参数，开发环境用；
    • info：业务核心日志，比如服务启动、接口调用成功、任务完成，生产环境必看；
    • warn：警告日志，比如参数校验失败、缓存击穿、非致命错误，需要关注但不影响服务运行；
    • error：错误日志，比如数据库连接失败、接口调用异常、业务逻辑报错，必须及时处理。

    主流日志库对比：

    对比维度 env_logger tracing-subscriber
    核心定位 简单、专注的日志实现器。为 log 门面库提供即时、易用的输出。 tracing 生态的核心组件。
    一个功能强大的结构化日志与分布式追踪框架的订阅器（Subscriber）。
    设计理念 传统、扁平的日志记录。 基于事件（Event） 和跨度（Span） 的上下文感知、结构化诊断。
    关键特性 1. 通过 RUST_LOG 环境变量过滤。
    2. 轻量、开箱即用。    		 1. 强大的结构化字段和上下文传递。
    2. 支持异步友好的跨度追踪。
    3. 丰富的输出格式（JSON, Pretty等）和可扩展性。
    性能表现 较高（单条日志耗时约3640纳秒，吞吐量约27万条/秒）。 相对较低（单条日志耗时约8797纳秒，吞吐量约11万条/秒）。
    学习曲线 平缓，API简单直观。 较陡峭，需理解事件、跨度、订阅器等概念。
    典型适用场景 1. 命令行工具。
    2. 微服务或业务逻辑不复杂的应用。
    3. 需要快速原型验证或对性能要求极高的场景。    		 1. 复杂的异步应用、Web服务（如你提到的Salvo, Actix-web, Poem）。
    2. 需要链路追踪的分布式系统。
    3. 生产环境需要对接ELK等日志分析平台。
    • 选择 env_logger 如果：你的项目是传统同步应用、命令行工具，或者你追求极致的日志性能与最简单的配置，且暂时不需要结构化和链路追踪功能。
    • 选择 tracing-subscriber 如果：你的项目是异步的Web服务或分布式系统，需要利用结构化日志进行高效排查，或者未来计划引入复杂的链路追踪（Tracing）。这正是大多数现代Rust Web框架（如Axum）推荐或默认使用它的原因。

    2.1.🌟 第一梯队：tracing + tracing-subscriber (首推，异步天花板，使用率 85%+，无替代方案)¶

    核心定位：Rust 异步 Web 生态的事实标准日志方案，也是生产级 Rust Web 项目的绝对首选，没有之一。

    • 门面层：tracing（新一代日志标准，异步优先，结构化日志 + 链路追踪）；
    • 实现层：tracing-subscriber（官方配套实现，功能全、性能高、适配完美）。

use tracing_subscriber::{fmt, layer::SubscriberExt, util::SubscriberInitExt, EnvFilter};

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    // ========== 🔥 核心：全局日志初始化（生产级配置，一行搞定） ==========
    tracing_subscriber::registry()
        // 日志级别过滤：RUST_LOG环境变量 或 配置文件中的日志级别
        .with(EnvFilter::new(&CONFIG.log.level))
        // 开发环境：控制台美观格式；生产环境：JSON格式（CONFIG.log.format = "json"）
        .with(if CONFIG.log.format == "json" {

            fmt::layer().pretty().boxed()
        })
        .init();

    // ========== 日志使用示例 ==========
    tracing::trace!("trace: 极致调试日志，开发环境用");
    tracing::debug!("debug: SQL语句、请求参数等调试信息");
    tracing::info!("info: 服务启动成功，地址：{}:{}", CONFIG.server.host, CONFIG.server.port);
    tracing::warn!("warn: 参数校验失败，用户ID：10086");
    tracing::error!("error: 数据库连接失败，url：{}", CONFIG.database.url);

    // ========== 结构化日志（生产级刚需，重点） ==========
    tracing::info!(
        %CONFIG.server.port,  // 简写：等同于 port = CONFIG.server.port
        db_url = %CONFIG.database.url,
        "结构化日志示例：服务配置加载完成"
    );

    Ok(())
}
    3. 结合中间件实现「请求日志」（契合你的 middleware 层，生产级必备）¶

    这是你之前项目结构中的src/middleware/logger.rs，所有请求都会打印请求方法、路径、状态码、耗时，是 Web 项目的核心日志能力：

    // src/middleware/logger.rs
use axum::{
    http::Request,
    middleware::Next,
    response::Response,
};
use std::time::Instant;
use tracing::info;

pub async fn logger_middleware<B>(mut req: Request<B>, next: Next<B>) -> Response {
    let start = Instant::now();
    let method = req.method().clone();
    let uri = req.uri().clone();

    // 执行后续请求处理
    let response = next.run(req).await;
    let status = response.status();
    let duration = start.elapsed();

    // 结构化请求日志
    info!(
        method = %method,
        uri = %uri,
        status = %status,
        duration = %duration.as_millis(),
        "请求处理完成"
    );

    response
}

    4.1.使用 lettre 发送邮件¶

    📌 场景 1：基础使用 —— Lettre 发送「纯文本中文邮件」（无乱码，最常用）¶

    ✅ 适配场景：验证码邮件、密码找回邮件等简单邮件；

    ✅ 核心：用 email-encoding 编码中文主题，彻底解决乱码。

    # Cargo.toml
[dependencies]
lettre = { version = "0.11", features = ["tokio1-native-tls"] }
tokio = { version = "1", features = ["full"] }

    use lettre::{Tokio1Executor, AsyncSmtpTransport, AsyncTransport, Message};
use lettre::transport::smtp::authentication::Credentials;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // 1. 构建邮件
    let email = Message::builder()
        .from("发件人 <from@example.com>".parse()?)
        .to("收件人 <to@example.com>".parse()?)
        .subject("来自 Lettre 的问候")
        .body("你好，这是一封使用 Lettre 发送的测试邮件。".to_string())?;

    // 2. 创建 SMTP 传输器（以 Gmail 为例）
    let creds = Credentials::new("你的邮箱@gmail.com".to_string(), "你的应用密码".to_string());
    let mailer: AsyncSmtpTransport<Tokio1Executor> =
        AsyncSmtpTransport::<Tokio1Executor>::relay("smtp.gmail.com")?
            .port(587)
            .credentials(creds)
            .build();

    // 3. 发送邮件
    mailer.send(email).await?;

    println!("邮件发送成功！");
    Ok(())
}

    📌 场景 3：生产级最佳实践 —— Axum Web 中「异步发送邮件 + 全局共享发送器」（Web 开发标配）¶

    ✅ 核心痛点：Web 请求中绝对不能每次发送邮件都创建新的 SMTP 连接，会导致性能极低、连接数超限；

    ✅ 解决方案：将 Lettre 的 Tokio1Transport 封装为 全局共享的单例，注入 Axum 的 State 中，所有请求复用同一个发送器（内置连接池）；

    ✅ 这是 Rust Web 开发中邮件发送的标准写法，必须掌握！

    use axum::{
    extract::State,
    http::StatusCode,
    routing::post,
    Json, Router,
};
use lettre::{
    message::{Message, SinglePart},
    transport::smtp::{authentication::Credentials, client::Tls},
    Tokio1Transport,
};
use email_encoding::email::header::encode_header;
use serde::Deserialize;
use std::sync::Arc;

// 定义全局共享的邮件发送器
type Mailer = Arc<Tokio1Transport>;

// 请求体：用户注册信息
#[derive(Deserialize)]
struct RegisterRequest {
    email: String,
    username: String,
}

// 核心：Axum接口中异步发送注册验证码邮件
async fn send_register_mail(
    State(mailer): State<Mailer>,
    Json(req): Json<RegisterRequest>,
) -> (StatusCode, &'static str) {
    // 构建验证码邮件
    let email = match Message::builder()
        .from("Rust Web 注册 <your-email@163.com>".parse())
        .to(req.email.parse().unwrap())
        .subject(encode_header("【Rust Web】注册验证码"))
        .body(SinglePart::plain(format!(
            "您好 {}, 您的注册验证码是：668899，有效期5分钟。",
            req.username
        ))) {
        Ok(e) => e,
        Err(_) => return (StatusCode::INTERNAL_SERVER_ERROR, "邮件构建失败"),
    };

    // 复用全局发送器，异步发送邮件
    match mailer.send(email).await {
        Ok(_) => (StatusCode::OK, "验证码邮件发送成功"),
        Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "邮件发送失败"),
    }
}

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // 1. 初始化全局邮件发送器（单例，只创建一次）
    let creds = Credentials::new("your-email@163.com".into(), "your-auth-code".into());
    let mailer = Arc::new(
        Tokio1Transport::relay("smtp.163.com")?
            .port(465)
            .tls(Tls::Required)
            .credentials(creds)
            .build(),
    );

    // 2. 构建Axum路由，注入全局State
    let app = Router::new()
        .route("/send-register-mail", post(send_register_mail))
        .with_state(mailer);

    // 3. 启动服务
    axum::Server::bind(&([127, 0, 0, 1], 8080).into())
        .serve(app.into_make_service())
        .await?;

    Ok(())
}

    4.3.终极生产级方案 —— 「RabbitMQ 消息队列 + Lettre」异步发送邮件（高并发必备）¶

    ✅ 核心痛点：即使是异步发送邮件，在高并发场景下，邮件发送的 IO 耗时依然会阻塞 Web 请求的响应；比如：用户注册后，需要发送邮件 + 创建用户 + 返回响应，邮件发送的耗时会导致响应变慢；

    ✅ 黄金解决方案：Web 请求只负责「将邮件任务发送到消息队列」，立即返回响应；后台消费线程负责「从队列中取出任务，调用 Lettre 发送邮件」；

    ✅ 这是 高并发 Rust Web 服务的标配，也是大厂的通用做法，无任何性能瓶颈；

    ✅ 核心依赖：lapin (RabbitMQ 异步客户端) + lettre，二者完美兼容 Tokio 异步生态。

    核心流程:

    1. 用户注册 → Axum 接口 → 将「收件人 + 邮件内容」发送到 RabbitMQ 队列 → 立即返回「注册成功」；
    2. 后台异步消费线程 → 监听队列 → 取出邮件任务 → 调用 Lettre 发送邮件；
    3. 邮件发送失败时，队列自动重试，保证邮件最终送达。

    4.4.总结¶

    1. lettre 是 Rust Web 邮件开发的事实标准，没有任何替代品，是唯一的核心选择；
    2. email-encoding 是必配的辅助库，解决所有中文乱码问题；
    3. 生产级高并发场景，必须结合消息队列异步发送邮件，这是性能的绝对保障；
    4. Rust 邮件生态的分工明确，无内卷，学习成本极低，掌握这两个库即可覆盖所有邮件需求。

    6.1.🌟 天花板首选：clap (95% 生产场景必用，完全对标 Cobra + )¶

    ✅ 核心定位 & 生态地位¶

    clap 是 Rust 生态中事实标准的 CLI 解析框架，没有之一，GitHub star 30k+，几乎所有知名的 Rust CLI 工具（cargo-watch/sqlx-cli/cross 等你之前用到的工具）底层都是基于 clap 开发的。

    • 核心对标：完美覆盖 Go Cobra 的全部核心功能，且在「强类型安全、零样板代码、错误提示友好、扩展性」上全面超越 Cobra；
    • 两大使用模式：
    • derive 派生宏模式（推荐，99% 场景）：基于 serde 风格的宏注解，一行注解实现所有解析逻辑，极简优雅，对标 Cobra 的简洁写法，学习成本极低；
    • builder 构建器模式：纯 Rust 代码手动构建命令，极致灵活，适合超复杂的 CLI 场景（如动态生成子命令）。
    ✅ 核心优势（对比 Go Cobra + Rust 其他工具）¶
    ✅ 超越 Cobra 的核心亮点¶
    1. 编译期强类型安全：所有参数 / 标志的类型、约束都在编译期校验完成，零运行时解析错误，Cobra 是动态解析，运行时才会发现参数类型错误；
    2. 零样板代码：derive 模式下，仅需给结构体加注解，即可实现命令 / 参数 / 标志的全部解析逻辑，Cobra 需要写大量的 AddCommand/Flag 代码；
    3. 极致友好的错误提示：参数错误时，clap 会给出精准的错误原因 + 修复建议，甚至自动补全可能的命令，Cobra 的错误提示相对简陋；
    4. 原生支持丰富的校验规则：必填参数、范围校验、枚举值校验、正则校验等，无需手动写校验逻辑；
    5. 更小的编译产物：Rust 编译的二进制文件比 Go 更小，clap 的核心依赖极小，无冗余。
    ✅ 对标 Cobra 的核心能力（全部实现）¶
    • ✅ 多级嵌套子命令（Cobra 核心）；
    • ✅ 参数 / 标志解析（短 / 长标志、默认值、必填 / 可选）；
    • ✅ 自动生成帮助文档 / 版本信息；
    • ✅ 自动生成 Shell 补全脚本（bash/zsh/fish/powershell）；
    • ✅ 子命令分组、钩子函数、全局标志；
    ✅ 缺点（几乎无，吹毛求疵）¶
    1. derive 模式下需要依赖宏，对「零宏洁癖」开发者不友好（可选择 builder 模式）；
    2. 功能过于丰富，文档内容较多，新手需要几分钟熟悉注解规则。

async fn expensive_query(key: String) -> String {
    // 模拟耗时操作（如数据库查询）
    tokio::time::sleep(Duration::from_millis(500)).await;
    format!("Result for {}", key)
}

#[tokio::main]
async fn main() {
    println!("第一次调用（会执行函数体）...");
    println!("结果: {}", expensive_query("hello".to_string()).await);
    println!("第二次调用（可能直接从缓存读取）...");
    println!("结果: {}", expensive_query("hello".to_string()).await);
}
    2.使用 redis-rs 进行分布式缓存¶

    当需要在多个服务实例间共享缓存时，Redis 是标准选择。

    // 获取缓存
    let value: Option<String> = conn.get("my_key").await?;
    println!("从Redis获取的值: {:?}", value);

    Ok(())
}

    use anyhow::Result;
use axum::{
    extract::State,
    routing::get,
    Router, Json,
};
use bb8_redis::{
    redis::{AsyncCommands, RedisError},
    RedisConnectionManager,
};
use serde::Serialize;
use std::sync::Arc;
use std::time::Duration;

// 定义全局共享的连接池类型，Arc包裹实现多线程共享
type RedisPool = bb8::Pool<RedisConnectionManager>;

#[derive(Serialize)]
struct RedisResponse {
    result: String,
    data: Option<String>,
}

// 初始化Redis连接池（生产级完整配置）
async fn init_redis_pool(redis_url: &str) -> Result<RedisPool> {
    // 创建Redis连接管理器，基于redis-rs，自动处理连接创建/销毁
    let manager = RedisConnectionManager::new(redis_url)?;

    // 构建连接池，配置生产级核心参数（必配！）
    let pool = bb8::Pool::builder()
        .max_size(20) // 最大连接数：根据Redis配置的maxclients调整，建议20-50
        .min_idle(5)  // 最小空闲连接数：保证有足够的空闲连接响应请求
        .idle_timeout(Some(Duration::from_secs(60))) // 空闲连接回收时间：60秒无使用则销毁
        .connection_timeout(Duration::from_secs(3))   // 获取连接超时时间：3秒获取不到则报错
        .build(manager)
        .await?;

    println!("✅ Redis连接池初始化成功，最大连接数: 20，最小空闲连接数:5");
    Ok(pool)
}

// 示例1：基础Redis命令（set/get）
async fn redis_set_get(State(pool): State<Arc<RedisPool>>) -> Json<RedisResponse> {
    // 从连接池获取连接：核心API，自动复用空闲连接
    let mut conn = pool.get().await.unwrap();

    // 执行Redis命令：完全复用redis-rs的AsyncCommands trait，无学习成本
    let _: () = conn.set("rust:bb8:test", "hello bb8-redis").await.unwrap();
    let val: String = conn.get("rust:bb8:test").await.unwrap();

    Json(RedisResponse {
        result: "success".into(),
        data: Some(val),
    })
}

// 示例2：Redis哈希/列表命令 + 连接健康检测
async fn redis_hash_list(State(pool): State<Arc<RedisPool>>) -> Json<RedisResponse> {
    let mut conn = pool.get().await.unwrap();

    // 哈希命令
    let _: () = conn.hset("rust:bb8:user:1", "name", "bb8-redis").await.unwrap();
    let _: () = conn.hset("rust:bb8:user:1", "age", "20").await.unwrap();
    let user: Vec<(String, String)> = conn.hgetall("rust:bb8:user:1").await.unwrap();

    // 列表命令
    let _: () = conn.lpush("rust:bb8:list", "item1").await.unwrap();
    let list: Vec<String> = conn.lrange("rust:bb8:list", 0, -1).await.unwrap();

    Json(RedisResponse {
        result: "success".into(),
        data: Some(format!("user: {:?}, list: {:?}", user, list)),
    })
}

// 示例3：连接失效自动重试（生产级刚需）
async fn redis_health_check(State(pool): State<Arc<RedisPool>>) -> Json<RedisResponse> {
    let mut conn = pool.get().await.unwrap();
    // 执行PING命令检测连接健康，bb8会自动销毁失效连接并重建
    let pong: String = conn.ping().await.unwrap();

    Json(RedisResponse {
        result: "success".into(),
        data: Some(format!("Redis健康检测: {}", pong)),
    })
}

#[tokio::main]
async fn main() -> Result<()> {
    // Redis地址：支持 redis://user:pass@host:port/db 格式，TLS则用 rediss://
    let redis_url = "redis://127.0.0.1:6379/0";

    println!("✅ 服务启动成功: http://{}", addr);
    axum::Server::bind(&addr)
        .serve(app.into_make_service())
        .await?;

    Ok(())
}

// ========== 🌟 核心：初始化rust-i18n，必须放在main函数前 ==========
// 扫描locales目录下的语言文件，支持yaml格式，编译期检查
rust_i18n::i18n!("locales");

// ========== 🌟 Axum中间件：自动解析请求头的Accept-Language，设置全局语言 ==========
async fn i18n_middleware(mut req: Request, next: Next) -> Response {
    // 从请求头获取语言，默认值：zh-CN
    let lang_header = req.headers()
        .get(ACCEPT_LANGUAGE)
        .and_then(|v| v.to_str().ok())
        .unwrap_or("zh-CN");

    // 解析语言标识（兼容 zh-CN/zh-cn/en-US 等格式）
    let lang: LanguageIdentifier = lang_header.parse().unwrap_or_else(|_| "zh-CN".parse().unwrap());
    let lang_code = lang.to_string();

    // 设置全局语言，rust-i18n会自动匹配最接近的语言包（如zh-CN匹配zh-CN.yml）
    set_locale(&lang_code);
    println!("✅ 当前请求语言: {}, 全局语言: {}", lang_code, locale!());

    // 继续处理请求
    next.run(req).await
}

    use axum::{
    extract::Request,
    http::header::ACCEPT_LANGUAGE,
    middleware::Next,
    response::Response,
    routing::get,
    Router, Json,
};
use fluent::{FluentArgs, FluentBundle, FluentResource};
use fluent_langneg::{negotiate_languages, NegotiationStrategy};
use unic_langid::{langid, LanguageIdentifier};
use std::collections::HashMap;
use std::sync::Arc;

// 预定义支持的语言
const SUPPORTED_LANGS: &[LanguageIdentifier] = &[langid!("zh-CN"), langid!("en-US")];
// 默认语言
const DEFAULT_LANG: LanguageIdentifier = langid!("zh-CN");

// 加载FTL资源文件，全局复用
fn load_fluent_resource(lang: &LanguageIdentifier) -> Arc<FluentResource> {
    let content = match lang.to_string().as_str() {
        "zh-CN" => include_str!("../locales/zh-CN.ftl"),
        "en-US" => include_str!("../locales/en-US.ftl"),
        _ => include_str!("../locales/zh-CN.ftl"),
    };
    Arc::new(FluentResource::try_new(content.to_string()).unwrap())
}

// 翻译函数：核心封装
fn translate(lang: &LanguageIdentifier, key: &str, args: Option<FluentArgs>) -> String {
    let res = load_fluent_resource(lang);
    let mut bundle = FluentBundle::new(vec![lang.clone()]);
    bundle.add_resource(res).unwrap();
    let args = args.unwrap_or_default();
    bundle.format_pattern(&bundle.get_message(key).unwrap().value().unwrap(), Some(&args), &mut vec![]).unwrap().to_string()
}

// Axum国际化中间件：解析请求头语言
async fn fluent_middleware(req: Request, next: Next) -> Response {
    let lang_header = req.headers().get(ACCEPT_LANGUAGE).and_then(|v| v.to_str().ok()).unwrap_or("zh-CN");
    let user_langs: Vec<LanguageIdentifier> = lang_header.split(',').filter_map(|s| s.parse().ok()).collect();
    // 语言协商：匹配最适合的语言
    let selected_lang = negotiate_languages(
        &user_langs,
        SUPPORTED_LANGS,
        Some(&DEFAULT_LANG),
        NegotiationStrategy::Filtering,
    ).first().unwrap();
    println!("✅ 当前请求语言: {}", selected_lang);
    next.run(req).await
}

// 示例接口
async fn hello() -> Json<HashMap<String, String>> {
    let mut res = HashMap::new();
    let lang = &langid!("en-US"); // 可替换为解析后的语言
    // 基础翻译
    res.insert("hello".into(), translate(lang, "hello", None));
    // 带插值
    let mut args = FluentArgs::new();
    args.set("name", "Rust Developer");
    res.insert("greet".into(), translate(lang, "greet", Some(args)));
    // 复数规则
    let mut args = FluentArgs::new();
    args.set("count", 5);
    res.insert("article_count".into(), translate(lang, "article-count", Some(args)));
    Json(res)
}

    核心特性回顾¶

    i18n-embed 是 Rust 生态中专注于资源嵌入的国际化库，它的核心卖点是：将语言资源文件编译嵌入到二进制产物中，部署时无需携带任何语言文件，单机 / 容器 / 嵌入式环境部署友好，零依赖。

    核心亮点：极致轻量，零运行时依赖，编译期将语言文件打包进二进制，部署体积极小，适合「轻量化 Web 服务」「嵌入式 Rust Web」「无文件系统访问权限的环境」。

    补充：i18n-embed 本身是资源嵌入工具，通常搭配 rust-i18n / fluent-rs 使用，也可单独使用，支持所有主流语言文件格式。

    核心使用示例（Axum 集成，资源嵌入二进制）¶
    # Cargo.toml
[dependencies]
i18n-embed = { version = "0.13", features = ["yaml"] }
i18n-embed-axum = "0.13" # Axum官方适配库
unic-langid = "0.9"
axum = { version = "0.7", features = ["json"] }
tokio = { version = "1.0", features = ["full"] }
rust-embed = "6.0" # 资源嵌入核心库

    核心特点：通过#[derive(RustEmbed)]将locales目录嵌入二进制，运行时无需读取文件，直接从内存加载，部署零依赖。

    11.4.总结推荐¶

    ✅ 通用最佳实践（所有库都适用，核心！）

    1. 统一语言解析策略：必须从Accept-Language请求头解析语言，这是 HTTP 标准，前端无需传额外参数，无感切换多语言；兼容 zh-CN/zh-cn/en-US/en 等格式，自动匹配最接近的语言包。
    2. 设置合理的默认语言：建议默认语言为 zh-CN 或 en-US，避免解析失败时无兜底。
    3. 编译期检查翻译键：优先使用支持「编译期检查」的库（rust-i18n/i18n-embed），提前发现缺失的翻译键，避免线上出现 missing key: xxx。
    4. 资源文件管理规范：按「语言 + 模块」拆分文件，如 locales/zh-CN/user.yml、locales/en-US/article.yml，避免单文件过大，维护成本高。
    5. 缓存翻译结果：对于高频访问的翻译键，可缓存翻译结果，避免重复解析，提升性能（rust-i18n 已内置缓存）。
    6. 支持语言切换接口：提供 /api/set-lang?lang=en-US 接口，让用户手动切换语言，并存入 Cookie/Header，优先级高于Accept-Language。

    ✅ 部署最佳实践

    1. rust-i18n：开发时外置语言文件，方便热更新；生产时可通过features = ["embed"]将语言文件嵌入二进制，部署零依赖。
    2. fluent-rs：建议外置语言文件，支持热更新，适合频繁迭代的项目。
    3. i18n-embed：强制嵌入二进制，部署零依赖，适合容器化 / 嵌入式环境。

    ✅ 按场景精准选型（优先级排序）

    1. 90% 的 Rust Web 项目 → 无脑选 rust-i18n：极简易用、功能完备、Web 适配度拉满、生态活跃，学习成本极低，开发效率最高，完美覆盖所有常规国际化场景。
    2. 复杂国际化规则（多语言语法差异大） → 选 fluent-rs：火狐官方标准，工业级稳定，功能最全，适合面向全球用户的商业级产品。
    3. 轻量化部署 / 嵌入式 Web 服务 → 选 i18n-embed：资源嵌入二进制，部署零依赖，极致轻量，适合容器化 / 无文件系统访问权限的环境。

    12.跨平台编译/打包¶

    12.1.cargo-strip - 二进制产物瘦身工具（生产必装，体积减小 80%+，TOP1）¶

    ✅ 核心定位¶

    Rust 的cargo build --release编译出的二进制文件，默认包含调试符号、符号表、冗余信息，体积很大（比如一个简单的 Web 服务，编译后体积可能有 50MB+）；cargo-strip的核心功能是：剥离二进制文件中的所有调试符号和冗余信息，只保留运行时必要的代码。

    ✅ 核心价值¶

    ✅ 生产级核心刚需：二进制体积减小 80%~90%，比如 50MB 的二进制文件，瘦身後只有 5~10MB，大幅降低部署包体积、减少服务器磁盘占用、提升启动速度。

    ✅ 关键亮点：不影响程序运行，不损失任何性能，只是剥离调试信息，生产环境完全不需要这些信息。

    ✅ 核心使用示例¶
    # 全局安装（必装，无任何特征）
cargo install cargo-strip

# 🔥 生产环境编译+瘦身（核心命令，上线唯一打包方式，必记）
cargo build --release && cargo strip

    ✅ 补充说明

    • 瘦身後的二进制文件，无法使用gdb/lldb调试，生产环境无需调试；
    • 开发环境不要使用，否则无法调试程序。
    ✅ 适用场景¶

    ✅ 所有生产级 Rust Web 项目必装，上线的硬性标准，没有例外。

    12.2.cross - 交叉编译工具（生产必装，跨平台部署天花板）¶

    ✅ 核心定位¶

    Rust Web 项目的部署环境通常是 Linux x86_64（云服务器主流），而开发环境可能是 MacOS/Windows，直接在开发机上编译的二进制文件，无法在 Linux 服务器上运行；cross是 Rust 生态中 事实标准的交叉编译工具，基于 Docker 实现，能在任意系统上编译出「目标平台的二进制文件」。

    ✅ 核心价值¶
    • 无需在开发机上安装 Linux 环境，无需搭建虚拟机，一行命令编译出 Linux 服务器的可执行文件；
    • 支持所有主流平台：Linux x86_64、Linux aarch64（ARM 服务器）、Windows、MacOS 等；
    • 编译产物纯净，无系统依赖，直接上传到服务器即可运行，无需安装任何运行时。
    ✅ 优缺点分析¶

    ✅ 优点：零配置、开箱即用、支持所有平台、编译产物纯净、无系统依赖；

    ❌ 缺点：第一次使用需要拉取 Docker 镜像，略耗时，后续编译速度极快。

    ✅ 核心使用示例（生产级部署核心命令，必记）¶
    # 全局安装
cargo install cross

# 🔥 编译Linux x86_64平台的release版本（云服务器主流，核心命令）
cross build --target x86_64-unknown-linux-musl --release

# 编译完成后，产物路径：target/x86_64-unknown-linux-musl/release/your-web-project
# 直接上传到Linux服务器，赋予执行权限即可运行：chmod +x your-web-project && ./your-web-project

    ✅ 关键说明

    • musl是静态链接的 C 库，编译出的二进制文件无任何系统依赖，能在所有 Linux 发行版上运行（CentOS/Ubuntu/Debian 等）；
    • 这是生产部署的最佳实践，没有之一。
    ✅ 适用场景¶

    ✅ 所有生产级 Rust Web 项目必装，跨平台部署的唯一工具，没有例外。

    12.3.cargo-deb / cargo-rpm - 系统包打包工具（生产运维必装）¶

    ✅ 核心定位¶

    生产环境的运维通常需要将应用打包为 系统包（Debian/Ubuntu 的.deb 包，CentOS/RHEL 的.rpm 包），方便通过apt/yum安装、升级、卸载，还能配置开机自启、日志轮转、服务管理；cargo-deb和cargo-rpm是 Rust 生态中对应的打包工具，能将二进制文件打包为系统包。

    ✅ 核心价值¶
    • 一键打包，无需手动编写打包脚本；
    • 自动生成服务配置文件，支持开机自启、日志轮转、服务管理；
    • 运维友好，符合 Linux 系统的运维规范，大幅降低部署和维护成本。
    ✅ 核心使用示例¶
    # 安装deb打包工具（Debian/Ubuntu服务器）
cargo install cargo-deb

# 打包为.deb包，产物路径：target/debian/your-web-project_0.1.0_amd64.deb
cargo deb --release

# 安装rpm打包工具（CentOS/RHEL服务器）
cargo install cargo-rpm

# 打包为.rpm包
cargo rpm build --release
    ✅ 适用场景¶

    ✅ 企业级生产项目必装，个人项目可按需安装，适合需要规范化运维的场景。

    12.4.cargo-flamegraph - 性能分析工具（生产排障必装）¶

    ✅ 核心定位¶

    Rust Web 项目上线后，可能会遇到「接口响应慢、CPU 占用高、内存泄漏」等性能问题；cargo-flamegraph是 Rust 生态中 事实标准的性能分析工具，能生成「火焰图」，直观展示程序的 CPU 占用情况、函数调用耗时、性能瓶颈。

    ✅ 核心价值¶
    • 无需修改任何代码，一行命令生成火焰图，直观定位性能瓶颈；
    • 能精准找到耗时的函数、数据库查询、业务逻辑，是生产环境性能优化的核心工具。
    ✅ 核心使用示例¶
    # 全局安装
cargo install flamegraph

# 生成火焰图，运行程序并采集性能数据
cargo flamegraph --release --bin your-web-project

# 生成的火焰图文件：flamegraph.svg，用浏览器打开即可查看
    ✅ 适用场景¶

    ✅ 所有生产级 Rust Web 项目必装，性能优化和排障的核心工具，没有例外。
