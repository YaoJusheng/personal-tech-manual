    1.简介¶

    Poem 是一个功能完整且易于使用的 Rust Web 框架，提供了强大的类型安全和出色的性能。

    特点¶

    • 优雅的 API 设计：直观易用的接口设计
    • 类型安全：充分利用 Rust 类型系统
    • 中间件支持：丰富的内置中间件
    • OpenAPI 集成：原生支持 OpenAPI 3.0 规范
    • WebSocket 支持：完整的 WebSocket 实现
    • 高性能：基于 Tokio 异步运行时

    官方资源¶

    • 官网：https://github.com/poem-web/poem
    • 文档：https://docs.rs/poem/
    • GitHub：https://github.com/poem-web/poem

    3.使用¶

    基础 Hello World¶

    use poem::{get, handler, listener::TcpListener, web::Path, Route, Server};

#[handler]
fn hello() -> String {
    "Hello, World!".to_string()
}

#[handler]
fn greet(Path(name): Path<String>) -> String {
    format!("Hello, {}!", name)
}

#[tokio::main]
async fn main() -> Result<(), std::io::Error> {
    let app = Route::new()
        .at("/", get(hello))
        .at("/greet/:name", get(greet));

    Server::new(TcpListener::bind("127.0.0.1:8080"))
        .run(app)
        .await
}

    使用 OpenAPI¶

    use poem::{listener::TcpListener, Route, Server};
use poem_openapi::{param::Path, payload::Json, OpenApi, OpenApiService};

#[derive(Debug, serde::Serialize, serde::Deserialize, poem_openapi::Object)]
struct User {
    id: u64,
    name: String,
    email: String,
}

struct Api;

#[OpenApi]
impl Api {
    #[oai(path = "/users/:id", method = "get")]
    async fn get_user(&self, Path(id): Path<u64>) -> Json<User> {
        Json(User {
            id,
            name: "John Doe".to_string(),
            email: "john@example.com".to_string(),
        })
    }

    #[oai(path = "/users", method = "post")]
    async fn create_user(&self, Json(user): Json<User>) -> Json<User> {
        Json(user)
    }
}

#[tokio::main]
async fn main() -> Result<(), std::io::Error> {
    let api_service = OpenApiService::new(Api, "User API", "1.0")
        .server("http://localhost:8080");

    let ui = api_service.swagger_ui();

    let app = Route::new()
        .nest("/", api_service)
        .nest("/docs", ui);

    Server::new(TcpListener::bind("127.0.0.1:8080"))
        .run(app)
        .await
}

    listener::TcpListener,

    自定义中间件¶

    use poem::{Endpoint, IntoResponse, Middleware, Request, Response, Result};

pub struct AuthMiddleware {
    secret: String,
}

impl<E: Endpoint> Middleware<E> for AuthMiddleware {
    type Output = AuthMiddlewareImpl<E>;

    fn transform(&self, ep: E) -> Self::Output {
        AuthMiddlewareImpl {
            ep,
            secret: self.secret.clone(),
        }
    }
}

pub struct AuthMiddlewareImpl<E> {
    ep: E,
    secret: String,
}

impl<E: Endpoint> Endpoint for AuthMiddlewareImpl<E> {
    type Output = Response;

    async fn call(&self, req: Request) -> Result<Self::Output> {
        let token = req
            .headers()
            .get("Authorization")
            .and_then(|v| v.to_str().ok())
            .and_then(|v| v.strip_prefix("Bearer "));

        if let Some(_token) = token {
            // 验证token
            let res = self.ep.call(req).await;
            res.map(IntoResponse::into_response)
        } else {
            Err(poem::Error::from_status(poem::http::StatusCode::UNAUTHORIZED))
        }
    }
}

    4.4 ORM 集成（SeaORM）¶

    use poem::{handler, listener::TcpListener, web::{Data, Json, Path}, Route, Server};
use sea_orm::{Database, DatabaseConnection, EntityTrait, Set};

// 实体定义
use sea_orm::entity::prelude::*;

#[derive(Clone, Debug, PartialEq, DeriveEntityModel, serde::Serialize)]
#[sea_orm(table_name = "users")]
pub struct Model {
    #[sea_orm(primary_key)]
    pub id: i32,
    pub name: String,
    pub email: String,
}

#[derive(Copy, Clone, Debug, EnumIter, DeriveRelation)]
pub enum Relation {}

impl ActiveModelBehavior for ActiveModel {}

#[handler]
async fn get_users(Data(db): Data<&DatabaseConnection>) -> Json<Vec<Model>> {
    let users = Entity::find().all(db).await.unwrap();
    Json(users)
}

#[handler]
async fn get_user(
    Path(id): Path<i32>,
    Data(db): Data<&DatabaseConnection>,
) -> Json<Model> {
    let user = Entity::find_by_id(id).one(db).await.unwrap().unwrap();
    Json(user)
}

#[tokio::main]
async fn main() -> Result<(), std::io::Error> {
    let db = Database::connect("postgres://user:pass@localhost/db")
        .await
        .unwrap();

    let app = Route::new()
        .at("/users", get(get_users))
        .at("/users/:id", get(get_user))
        .data(db);

    Server::new(TcpListener::bind("127.0.0.1:8080"))
        .run(app)
        .await
}

    4.5 参数验证¶

    use poem_openapi::{payload::Json, Object, OpenApi};
use validator::Validate;

#[derive(Debug, serde::Deserialize, Object, Validate)]
struct CreateUserRequest {
    #[validate(length(min = 3, max = 50))]
    name: String,

    #[validate(email)]
    email: String,

    #[validate(length(min = 8))]
    password: String,
}

struct Api;

#[OpenApi]
impl Api {
    #[oai(path = "/users", method = "post")]
    async fn create_user(&self, Json(req): Json<CreateUserRequest>) -> Json<String> {
        match req.validate() {
            Ok(_) => Json("User created".to_string()),
            Err(e) => Json(format!("Validation failed: {:?}", e)),
        }
    }
}

    4.6 JWT 认证¶

    use jsonwebtoken::{decode, encode, DecodingKey, EncodingKey, Header, Validation};
use poem::{handler, web::Data, IntoResponse};
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
struct Claims {
    sub: String,
    exp: usize,
}

fn create_token(user_id: &str, secret: &str) -> String {
    let claims = Claims {
        sub: user_id.to_string(),
        exp: (chrono::Utc::now() + chrono::Duration::hours(24)).timestamp() as usize,
    };
    encode(&Header::default(), &claims, &EncodingKey::from_secret(secret.as_ref())).unwrap()
}

#[handler]
fn login(Data(secret): Data<&String>) -> impl IntoResponse {
    let token = create_token("user123", secret);
    poem::Response::builder()
        .header("Authorization", format!("Bearer {}", token))
        .body(token)
}

    4.7 Redis 缓存¶

    use poem::{handler, web::Data};
use redis::{Client, Commands};

#[handler]
async fn get_cache(
    Data(redis): Data<&Client>,
    poem::web::Path(key): poem::web::Path<String>,
) -> String {
    let mut conn = redis.get_connection().unwrap();
    conn.get(&key).unwrap_or_else(|_| "Not found".to_string())
}

#[handler]
async fn set_cache(
    Data(redis): Data<&Client>,
    poem::web::Path(key): poem::web::Path<String>,
    poem::web::Query(value): poem::web::Query<String>,
) -> &'static str {
    let mut conn = redis.get_connection().unwrap();
    let _: () = conn.set(&key, value).unwrap();
    "Cached"
}

    4.9 CLI 集成¶

    use clap::{Parser, Subcommand};
use poem::{listener::TcpListener, Route, Server};

#[derive(Parser)]
#[command(name = "poem-api")]
struct Cli {
    #[command(subcommand)]
    command: Commands,
}

#[derive(Subcommand)]
enum Commands {
    Serve {
        #[arg(short, long, default_value_t = 8080)]
        port: u16,
    },
    Migrate,
}

#[tokio::main]
async fn main() -> Result<(), std::io::Error> {
    let cli = Cli::parse();

    match cli.command {
        Commands::Serve { port } => {
            println!("Starting server on port {}", port);
            let app = Route::new();
            Server::new(TcpListener::bind(format!("127.0.0.1:{}", port)))
                .run(app)
                .await
        }
        Commands::Migrate => {
            println!("Running migrations...");
            Ok(())
        }
    }
}

    4.10 文件上传与下载¶

    use poem::{
    handler,
    web::{Data, Multipart},
    IntoResponse, Response,
};
use tokio::{fs::File, io::AsyncWriteExt};

#[handler]
async fn upload(mut multipart: Multipart) -> poem::Result<Response> {
    while let Ok(Some(field)) = multipart.next_field().await {
        let name = field.file_name().map(ToString::to_string);
        if let Some(filename) = name {
            let data = field.bytes().await?;
            let path = format!("./uploads/{}", filename);
            let mut file = File::create(&path).await?;
            file.write_all(&data).await?;
        }
    }
    Ok("File uploaded".into_response())
}

#[handler]
async fn download(poem::web::Path(filename): poem::web::Path<String>) -> poem::Result<Response> {
    let path = format!("./uploads/{}", filename);
    Ok(poem::web::StaticFileResponse::new(path).into_response())
}

    4.12 会话管理¶

    use poem::{
    handler,
    listener::TcpListener,
    middleware::CookieJarManager,
    session::{CookieConfig, CookieSession, Session},
    web::cookie::CookieKey,
    Route, Server,
};

#[handler]
async fn login(session: &Session) -> String {
    session.set("user_id", "123");
    session.set("username", "admin");
    "Logged in".to_string()
}

#[handler]
async fn profile(session: &Session) -> String {
    if let Some(user_id) = session.get::<String>("user_id") {
        format!("User: {}", user_id)
    } else {
        "Not logged in".to_string()
    }
}

#[tokio::main]
async fn main() -> Result<(), std::io::Error> {
    let app = Route::new()
        .at("/login", poem::post(login))
        .at("/profile", poem::get(profile))
        .with(CookieJarManager::new())
        .with(CookieSession::new(CookieConfig::default().secure(false)));

    Server::new(TcpListener::bind("127.0.0.1:8080"))
        .run(app)
        .await
}

    4.17 优雅关闭¶

    use tokio::signal;

#[tokio::main]
async fn main() -> Result<(), std::io::Error> {
    let app = Route::new();
    let listener = TcpListener::bind("127.0.0.1:8080");

    let server = Server::new(listener).run(app);

    tokio::select! {
        _ = server => {},
        _ = signal::ctrl_c() => {
            println!("Shutting down gracefully...");
        }
    }

    Ok(())
}
