    v.SetConfigName(env)

}

    "os"
    "time"

    "go.uber.org/zap"
    "go.uber.org/zap/zapcore"

// InitLogger еҲқе§ӢеҢ–ж—Ҙеҝ—з»„д»¶
func InitLogger(cfg config.LogConfig) (*zap.Logger, error) {
    // 1. й…ҚзҪ®ж—Ҙеҝ—зә§еҲ«
    level := zapcore.DebugLevel
    if err := level.UnmarshalText([]byte(cfg.Level)); err != nil {
        return nil, fmt.Errorf("ж— ж•Ҳзҡ„ж—Ҙеҝ—зә§еҲ«: %w", err)
    }

    // 2. й…ҚзҪ®зј–з ҒеҷЁпјҲж–Үжң¬/JSONпјү
    encoderCfg := zap.NewProductionEncoderConfig()
    encoderCfg.EncodeTime = zapcore.ISO8601TimeEncoder // ж—¶й—ҙж јејҸ
    encoderCfg.EncodeLevel = zapcore.CapitalLevelEncoder // зә§еҲ«еӨ§еҶҷ

    var encoder zapcore.Encoder
    if cfg.Format == "text" {
        encoder = zapcore.NewConsoleEncoder(encoderCfg) // ж–Үжң¬ж јејҸпјҲејҖеҸ‘еҸӢеҘҪпјү
    } else {
        encoder = zapcore.NewJSONEncoder(encoderCfg)    // JSON ж јејҸпјҲз”ҹдә§еҸӢеҘҪпјү
    }

    // 3. й…ҚзҪ®иҫ“еҮәзӣ®ж ҮпјҲж–Үд»¶/ж ҮеҮҶиҫ“еҮәпјү
    var writer zapcore.WriteSyncer
    if cfg.Output == "file" {
        // дҪҝз”Ё lumberjack е®һзҺ°ж—Ҙеҝ—иҪ®иҪ¬
        writer = zapcore.AddSync(&lumberjack.Logger{
            Filename:   cfg.FilePath,    // ж—Ҙеҝ—ж–Үд»¶и·Ҝеҫ„
            MaxSize:    cfg.MaxSize,      // жңҖеӨ§еӨ§е°ҸпјҲMBпјү
            MaxAge:     cfg.MaxAge,       // дҝқз•ҷеӨ©ж•°
            MaxBackups: cfg.MaxBackups,   // жңҖеӨ§еӨҮд»Ҫж•°
            Compress:   true,             // еҺӢзј©ж—§ж—Ҙеҝ—
        })
    } else {
        writer = zapcore.AddSync(os.Stdout) // ж ҮеҮҶиҫ“еҮә
    }

    // 4. еҲӣе»әж ёеҝғз»„д»¶
    core := zapcore.NewCore(encoder, writer, level)

    // 5. жһ„е»ә LoggerпјҲж·»еҠ и°ғз”ЁиҖ…дҝЎжҒҜпјү
    logger := zap.New(core, zap.AddCaller(), zap.AddCallerSkip(1))
    return logger, nil
}
