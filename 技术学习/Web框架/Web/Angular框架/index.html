    概述¶

    Angular是由Google开发的企业级前端框架，用于构建单页应用(SPA)和响应式Web应用。它基于TypeScript，提供了完整的开发工具链和丰富的生态系统。

    核心特性¶

    1. 组件化架构¶

    Angular采用组件化设计，将UI拆分为独立的、可复用的组件。

    // app.component.ts
import { Component } from '@angular/core';

@Component({
  selector: 'app-root',
  template: `
    <div class="container">
      <h1>{{ title }}</h1>
      <app-user-list [users]="users"></app-user-list>
    </div>
  `,
  styleUrls: ['./app.component.scss']
})
export class AppComponent {
  title = 'Angular应用';
  users = [
    { id: 1, name: '张三', email: 'zhangsan@example.com' },
    { id: 2, name: '李四', email: 'lisi@example.com' }
  ];
}

    2. 双向数据绑定¶

    Angular提供强大的数据绑定机制，支持插值表达式、属性绑定、事件绑定等。

    // user-form.component.ts
import { Component } from '@angular/core';

@Component({
  selector: 'app-user-form',
  template: `
    <form (ngSubmit)="onSubmit()">
      <input
        type="text"
        [(ngModel)]="user.name"
        name="username"
        placeholder="用户名"
        required
      />
      <input
        type="email"
        [(ngModel)]="user.email"
        name="email"
        placeholder="邮箱"
        required
      />
      <button type="submit">提交</button>
    </form>
    <p *ngIf="submitted">用户 {{ user.name }} 已提交</p>
  `
})
export class UserFormComponent {
  user = { name: '', email: '' };
  submitted = false;

  onSubmit() {
    this.submitted = true;
    console.log('表单提交:', this.user);
  }
}

    3. 依赖注入(DI)¶

    Angular内置强大的依赖注入系统，用于管理服务和其他依赖。

    // user.service.ts
import { Injectable } from '@angular/core';
import { Observable, of } from 'rxjs';
import { delay } from 'rxjs/operators';

export interface User {
  id: number;
  name: string;
  email: string;
}

@Injectable({
  providedIn: 'root'
})
export class UserService {
  private users: User[] = [
    { id: 1, name: '张三', email: 'zhangsan@example.com' },
    { id: 2, name: '李四', email: 'lisi@example.com' }
  ];

  getUsers(): Observable<User[]> {
    return of(this.users).pipe(delay(500));
  }

  getUserById(id: number): Observable<User | undefined> {
    const user = this.users.find(u => u.id === id);
    return of(user).pipe(delay(300));
  }

  addUser(user: User): Observable<User> {
    this.users.push(user);
    return of(user).pipe(delay(200));
  }
}

    2. 指令系统¶

    结构指令¶

    // *ngIf 条件渲染
<div *ngIf="isLoggedIn; else loginTemplate">
  欢迎回来, {{ userName }}!
</div>

<ng-template #loginTemplate>
  <p>请登录以继续</p>
  <button (click)="login()">登录</button>
</ng-template>

// *ngFor 循环渲染
<ul>
  <li *ngFor="let user of users; index as i; trackBy: trackByUserId">
    {{ i + 1 }}. {{ user.name }} ({{ user.email }})
  </li>
</ul>

// *ngSwitch 多条件渲染
<div [ngSwitch]="userRole">
  <p *ngSwitchCase="'admin'">管理员权限</p>
  <p *ngSwitchCase="'editor'">编辑权限</p>
  <p *ngSwitchCase="'user'">用户权限</p>
  <p *ngSwitchDefault>未知角色</p>
</div>

    属性指令¶

    // ngClass 类名控制
<div [ngClass]="{
  'active': isActive,
  'highlight': isHighlighted,
  'disabled': isDisabled
}">内容</div>

// ngStyle 样式控制
<div [ngStyle]="{
  'color': textColor,
  'background-color': bgColor,
  'font-size.px': fontSize
}">样式内容</div>

// 自定义指令
import { Directive, ElementRef, Input } from '@angular/core';

@Directive({
  selector: '[appHighlight]'
})
export class HighlightDirective {
  @Input() appHighlight = 'yellow';

  constructor(private el: ElementRef) {}

  ngOnInit() {
    this.el.nativeElement.style.backgroundColor = this.appHighlight;
  }
}

    3. 组件通信¶

    输入属性(@Input)¶

import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-user-card',
  template: `
    <div class="user-card">
      <h3>{{ user.name }}</h3>
      <p>邮箱: {{ user.email }}</p>

export class UserCardComponent {

    输出属性(@Output)¶

    // 子组件
import { Component, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-user-list',
  template: `
    <ul>
      <li *ngFor="let user of users">
        {{ user.name }}
        <button (click)="selectUser(user)">选择</button>
      </li>
    </ul>
  `
})
export class UserListComponent {
  @Output() userSelected = new EventEmitter<any>();

  users = [
    { id: 1, name: '张三' },
    { id: 2, name: '李四' }
  ];

  selectUser(user: any) {
    this.userSelected.emit(user);
  }
}

// 父组件接收
<app-user-list
  [users]="users"
  (userSelected)="onUserSelected($event)">
</app-user-list>

    服务通信¶

    // 共享服务
import { Injectable } from '@angular/core';
import { BehaviorSubject } from 'rxjs';

@Injectable()
export class UserStateService {
  private currentUserSubject = new BehaviorSubject<any>(null);
  currentUser$ = this.currentUserSubject.asObservable();

  setCurrentUser(user: any) {
    this.currentUserSubject.next(user);
  }

  getCurrentUser() {
    return this.currentUserSubject.value;
  }
}

// 组件A
export class ComponentA {
  constructor(private userState: UserStateService) {}

  login(user: any) {
    this.userState.setCurrentUser(user);
  }
}

    // custom-validators.ts
import { AbstractControl, ValidationErrors } from '@angular/forms';

export class CustomValidators {
  // 密码强度验证
  static passwordStrength(control: AbstractControl): ValidationErrors | null {
    const value = control.value;
    if (!value) return null;

    const hasNumber = /\d/.test(value);
    const hasLetter = /[a-zA-Z]/.test(value);
    const hasSpecial = /[!@#$%^&*(),.?":{}|<>]/.test(value);

    const isValid = hasNumber && hasLetter && hasSpecial;
    return isValid ? null : { passwordStrength: true };
  }

  // 确认密码验证
  static passwordMatch(group: AbstractControl): ValidationErrors | null {
    const password = group.get('password')?.value;
    const confirmPassword = group.get('confirmPassword')?.value;

    if (password && confirmPassword && password !== confirmPassword) {
      return { passwordMismatch: true };
    }
    return null;
  }

  // 唯一用户名验证
  static uniqueUsername(control: AbstractControl): Promise<ValidationErrors | null> {
    return new Promise((resolve) => {
      // 模拟异步验证
      setTimeout(() => {
        const isTaken = ['admin', 'user', 'test'].includes(control.value);
        resolve(isTaken ? { uniqueUsername: true } : null);
      }, 1000);
    });
  }
}

// 使用自定义验证器

  username: ['', [Validators.required], [CustomValidators.uniqueUsername]],
  password: ['', [Validators.required, CustomValidators.passwordStrength]],

    // http.service.ts
import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders, HttpParams } from '@angular/common/http';
import { Observable, throwError } from 'rxjs';
import { catchError, retry } from 'rxjs/operators';

export interface ApiResponse<T> {
  data: T;
  message: string;
  status: number;
}

@Injectable({
  providedIn: 'root'
})
export class HttpService {
  private apiUrl = 'https://api.example.com';

  constructor(private http: HttpClient) {}

  // GET请求
  getUsers(params?: any): Observable<ApiResponse<any[]>> {
    const httpParams = new HttpParams({ fromObject: params || {} });

    return this.http.get<ApiResponse<any[]>>(`${this.apiUrl}/users`, {
      params: httpParams,
      headers: new HttpHeaders({
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.getToken()}`
      })
    }).pipe(
      retry(3), // 失败重试3次
      catchError(this.handleError)
    );
  }

  // POST请求
  createUser(user: any): Observable<ApiResponse<any>> {
    return this.http.post<ApiResponse<any>>(`${this.apiUrl}/users`, user, {
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      })
    }).pipe(
      catchError(this.handleError)
    );
  }

  // PUT请求
  updateUser(id: number, user: any): Observable<ApiResponse<any>> {
    return this.http.put<ApiResponse<any>>(`${this.apiUrl}/users/${id}`, user).pipe(
      catchError(this.handleError)
    );
  }

  // DELETE请求
  deleteUser(id: number): Observable<ApiResponse<any>> {
    return this.http.delete<ApiResponse<any>>(`${this.apiUrl}/users/${id}`).pipe(
      catchError(this.handleError)
    );
  }

  // 错误处理
  private handleError(error: any): Observable<never> {
    console.error('HTTP错误:', error);

    let errorMessage = '发生未知错误';

    if (error.error instanceof ErrorEvent) {
      // 客户端错误
      errorMessage = `客户端错误: ${error.error.message}`;
    } else {
      // 服务端错误
      errorMessage = `服务端错误: ${error.status} - ${error.message}`;
    }

    return throwError(() => new Error(errorMessage));
  }

  private getToken(): string {
    return localStorage.getItem('authToken') || '';
  }
}

    2. 拦截器使用¶

    // auth.interceptor.ts
import { Injectable } from '@angular/core';
import { HttpInterceptor, HttpRequest, HttpHandler, HttpEvent } from '@angular/common/http';
import { Observable } from 'rxjs';

@Injectable()
export class AuthInterceptor implements HttpInterceptor {
  intercept(req: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
    const token = localStorage.getItem('authToken');

    if (token) {
      const authReq = req.clone({
        headers: req.headers.set('Authorization', `Bearer ${token}`)
      });
      return next.handle(authReq);
    }

    return next.handle(req);
  }
}

// loading.interceptor.ts
import { Injectable } from '@angular/core';
import { HttpInterceptor, HttpRequest, HttpHandler, HttpEvent } from '@angular/common/http';
import { Observable } from 'rxjs';
import { finalize } from 'rxjs/operators';
import { LoadingService } from './loading.service';

@Injectable()
export class LoadingInterceptor implements HttpInterceptor {
  private activeRequests = 0;

  constructor(private loadingService: LoadingService) {}

  intercept(req: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
    if (this.activeRequests === 0) {
      this.loadingService.show();
    }
    this.activeRequests++;

    return next.handle(req).pipe(
      finalize(() => {
        this.activeRequests--;
        if (this.activeRequests === 0) {
          this.loadingService.hide();
        }
      })
    );
  }
}

    // async-user.service.ts
import { Injectable } from '@angular/core';
import { BehaviorSubject, Observable, of } from 'rxjs';
import { delay, tap } from 'rxjs/operators';

export interface User {
  id: number;
  name: string;
  email: string;
  role: string;
}

@Injectable({
  providedIn: 'root'
})
export class AsyncUserService {
  private usersSubject = new BehaviorSubject<User[]>([]);
  users$ = this.usersSubject.asObservable();

  private loadingSubject = new BehaviorSubject<boolean>(false);
  loading$ = this.loadingSubject.asObservable();

  constructor() {
    this.loadUsers();
  }

  loadUsers(): void {
    this.loadingSubject.next(true);

    // 模拟API调用
    of([
      { id: 1, name: '张三', email: 'zhangsan@example.com', role: 'admin' },
      { id: 2, name: '李四', email: 'lisi@example.com', role: 'user' }
    ]).pipe(
      delay(1000),
      tap(users => {
        this.usersSubject.next(users);
        this.loadingSubject.next(false);
      })
    ).subscribe();
  }

  addUser(user: User): Observable<User> {
    return of(user).pipe(
      delay(500),
      tap(newUser => {
        const currentUsers = this.usersSubject.value;
        this.usersSubject.next([...currentUsers, newUser]);
      })
    );
  }

  updateUser(id: number, updates: Partial<User>): Observable<User> {
    return of({ id, ...updates } as User).pipe(
      delay(300),
      tap(updatedUser => {
        const currentUsers = this.usersSubject.value;
        const updatedUsers = currentUsers.map(user =>
          user.id === id ? { ...user, ...updates } : user
        );
        this.usersSubject.next(updatedUsers);
      })
    );
  }
}

// ✅ 正确示例：处理异步验证状态
@Component()
export class UserFormComponent implements OnInit {
  userForm: FormGroup;
  usernameStatus = '';

  constructor(private fb: FormBuilder, private httpService: HttpService) {
    this.userForm = this.fb.group({
      username: ['', {
        validators: [Validators.required, Validators.minLength(3)],
        asyncValidators: [this.uniqueUsernameValidator.bind(this)],
        updateOn: 'blur'
      }]
    });
  }

  ngOnInit() {
    const usernameControl = this.userForm.get('username');

    usernameControl?.statusChanges.subscribe(status => {
      this.usernameStatus = status;
    });

    usernameControl?.valueChanges.subscribe(value => {
      if (value && value.length >= 3) {
        // 清除之前的验证错误
        usernameControl.setErrors(null);
      }
    });
  }

  private uniqueUsernameValidator(control: AbstractControl): Promise<ValidationErrors | null> {
    if (!control.value) {
      return Promise.resolve(null);
    }

    return this.httpService.checkUsername(control.value).pipe(
      map(isTaken => isTaken ? { uniqueUsername: true } : null),
      catchError(() => of({ uniqueUsername: null })) // 处理网络错误
    ).toPromise();
  }
}

    3. RxJS调试技巧¶

    // 自定义操作符用于调试
import { Observable } from 'rxjs';
import { tap, map } from 'rxjs/operators';

export function debug<T>(name: string, logFn: (value: T) => void = console.log) {
  return (source: Observable<T>) =>
    source.pipe(
      tap({
        next: (value) => {
          logFn(value);
          console.log(`${name} - Next:`, value);
        },
        error: (error) => console.log(`${name} - Error:`, error),
        complete: () => console.log(`${name} - Complete`)
      })
    );
}

    // WebSocket实时数据服务
@Injectable({
  providedIn: 'root'
})
export class RealtimeDataService {
  private socket: WebSocket;
  private dataSubject = new Subject<any>();
  public data$ = this.dataSubject.asObservable();

  constructor() {
    this.connect();
  }

  private connect() {
    this.socket = new WebSocket('ws://localhost:8080/realtime');

    this.socket.onopen = () => {
      console.log('WebSocket连接已建立');
    };

    this.socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.dataSubject.next(data);
    };

    this.socket.onclose = () => {
      console.log('WebSocket连接已关闭，尝试重连...');
      setTimeout(() => this.connect(), 5000);
    };

    this.socket.onerror = (error) => {
      console.error('WebSocket错误:', error);
    };
  }

  sendMessage(message: any) {
    if (this.socket.readyState === WebSocket.OPEN) {
      this.socket.send(JSON.stringify(message));
    }
  }

  disconnect() {
    if (this.socket) {
      this.socket.close();
    }
  }
}
