    概述¶

    Electron是由GitHub开发的一个跨平台桌面应用开发框架，使用Web技术（HTML、CSS、JavaScript）构建原生桌面应用。它结合了Chromium渲染引擎和Node.js运行时，允许开发者使用前端技术栈开发跨平台桌面应用。

    核心特性¶

    1. 跨平台支持¶

    Electron支持Windows、macOS和Linux三大主流操作系统，一套代码可以打包到多个平台。

    // main.js - 主进程
const { app, BrowserWindow, Menu } = require('electron');
const path = require('path');

function createWindow() {
  // 创建浏览器窗口
  const mainWindow = new BrowserWindow({
    width: 1200,
    height: 800,
    webPreferences: {
      nodeIntegration: true, // 在渲染进程中启用Node.js
      contextIsolation: false // 禁用上下文隔离
    },
    icon: path.join(__dirname, 'assets/icon.png') // 应用图标
  });

  // 加载应用
  mainWindow.loadFile('index.html');

  // 打开开发者工具
  mainWindow.webContents.openDevTools();

  // 当窗口被关闭，这个事件会被触发
  mainWindow.on('closed', () => {
    mainWindow = null;
  });
}

// Electron会在初始化后并准备创建浏览器窗口的时候，调用这个函数
app.whenReady().then(createWindow);

// 当全部窗口关闭时退出
app.on('window-all-closed', () => {
  // 在macOS上，应用和菜单栏通常会保持活跃状态，直到用户使用Cmd + Q明确退出
  if (process.platform !== 'darwin') {
    app.quit();
  }
});

app.on('activate', () => {
  // 在macOS上，当单击dock图标并且没有其他窗口打开时，通常会重新创建窗口
  if (BrowserWindow.getAllWindows().length === 0) {
    createWindow();
  }
});

    2. 主进程和渲染进程¶

    Electron架构基于两个核心进程：主进程和渲染进程。

function createMainWindow() {
  const mainWindow = new BrowserWindow({
    width: 1000,

      nodeIntegration: false, // 渲染进程中禁用Node.js

    基础语法详解¶

    1. 窗口管理¶

    // main.js - 窗口管理详解
const { app, BrowserWindow, Menu, dialog, shell } = require('electron');
const path = require('path');

// 创建主窗口
function createMainWindow() {
  const mainWindow = new BrowserWindow({
    width: 1200,
    height: 800,
    minWidth: 800,
    minHeight: 600,
    show: false, // 先不显示，等加载完成后再显示
    center: true, // 窗口居中
    resizable: true, // 允许调整大小
    closable: true, // 允许关闭
    minimizable: true, // 允许最小化
    maximizable: true, // 允许最大化
    fullscreenable: true, // 允许全屏
    skipTaskbar: false, // 在任务栏显示
    kiosk: false, // 不使用 kiosk 模式
    webPreferences: {
      nodeIntegration: false,
      contextIsolation: true,
      preload: path.join(__dirname, 'preload.js'),
      webSecurity: true, // 启用web安全
      allowRunningInsecureContent: false, // 不允许运行不安全内容
      experimentalFeatures: false // 禁用实验性功能
    }
  });

  // 窗口事件处理
  mainWindow.once('ready-to-show', () => {
    mainWindow.show();
  });

  // 处理页面跳转
  mainWindow.webContents.on('will-navigate', (event, url) => {
    // 阻止外部链接
    if (url !== mainWindow.loadFile('index.html').toString()) {
      event.preventDefault();
      shell.openExternal(url);
    }
  });

  // 处理新窗口创建
  mainWindow.webContents.setWindowOpenHandler(({ url }) => {
    // 只允许特定URL在新窗口中打开
    if (url.startsWith('https://trusted-domain.com')) {
      return { action: 'allow' };
    }
    return { action: 'deny' };
  });

  return mainWindow;
}

// 创建子窗口
function createChildWindow(parentWindow, url) {
  const childWindow = new BrowserWindow({
    width: 600,
    height: 400,
    parent: parentWindow,
    modal: true, // 模态窗口
    show: false,
    resizable: false,
    minimizable: false,
    maximizable: false,
    closable: true
  });

  childWindow.loadURL(url);

  childWindow.once('ready-to-show', () => {
    childWindow.show();
  });

  return childWindow;
}

class IPCChannel {
  constructor(channelName) {
    this.channelName = channelName;
    this.handlers = new Map();
  }

  on(channel, handler) {
    if (!this.handlers.has(channel)) {
      this.handlers.set(channel, []);
    }
    this.handlers.get(channel).push(handler);
  }

  emit(channel, data) {
    const channelHandlers = this.handlers.get(channel);
    if (channelHandlers) {
      channelHandlers.forEach(handler => handler(data));

  }

      this.watchers.get(filePath).close();
    }

    const watcher = fs.watch(filePath, (eventType, filename) => {
      callback({
        path: filePath,
        event: eventType,
        filename: filename,
        timestamp: new Date().toISOString()
      });
    });

    this.watchers.set(filePath, watcher);
  }

  unwatchFile(filePath) {
    const watcher = this.watchers.get(filePath);
    if (watcher) {
      watcher.close();
      this.watchers.delete(filePath);
    }
  }

  unwatchAll() {
    this.watchers.forEach(watcher => watcher.close());
    this.watchers.clear();
  }
}

class HTTPClient {
  constructor(options = {}) {
    this.defaultOptions = {
      timeout: options.timeout || 30000,
      headers: options.headers || {},
      followRedirects: options.followRedirects !== false
    };
  }

  async request(url, options = {}) {
    return new Promise((resolve, reject) => {
      const urlObj = new URL(url);
      const client = urlObj.protocol === 'https:' ? https : http;

      const requestOptions = {
        hostname: urlObj.hostname,
        port: urlObj.port || (urlObj.protocol === 'https:' ? 443 : 80),
        path: urlObj.pathname + urlObj.search,
        method: options.method || 'GET',
        headers: { ...this.defaultOptions.headers, ...options.headers }
      };

      const req = client.request(requestOptions, (res) => {
        let data = '';

        res.on('data', (chunk) => {
          data += chunk;
        });

        res.on('end', () => {
          resolve({
            statusCode: res.statusCode,
            headers: res.headers,
            data: data,
            url: url
          });
        });
      });

      req.on('error', (error) => {
        reject(error);
      });

      req.setTimeout(this.defaultOptions.timeout, () => {
        req.destroy();
        reject(new Error('Request timeout'));
      });

      if (options.data) {
        req.write(options.data);
      }

      req.end();
    });
  }

  async get(url, options = {}) {
    return await this.request(url, { ...options, method: 'GET' });
  }

  async post(url, data, options = {}) {
    const postData = typeof data === 'string' ? data : JSON.stringify(data);
    return await this.request(url, {
      ...options,
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Content-Length': Buffer.byteLength(postData)
      },
      data: postData

      method: 'PUT',
      headers: {
        'Content-Type': 'application/json',
        'Content-Length': Buffer.byteLength(putData)
      },
      data: putData
    });
  }

  async delete(url, options = {}) {
    return await this.request(url, { ...options, method: 'DELETE' });
  }

    // main.js - WebSocket处理
const { BrowserWindow } = require('electron');
const WebSocket = require('ws');

class WebSocketManager {
  constructor() {
    this.connections = new Map();
    this.reconnectAttempts = new Map();
    this.maxReconnectAttempts = 5;
  }

  connect(url, options = {}) {
    return new Promise((resolve, reject) => {
      try {
        const ws = new WebSocket(url, options);

        ws.on('open', () => {
          console.log('WebSocket连接已建立:', url);
          this.connections.set(url, ws);
          this.reconnectAttempts.set(url, 0);

          // 通知渲染进程连接成功
          BrowserWindow.getAllWindows().forEach(window => {
            window.webContents.send('websocket:connected', { url });
          });

          resolve(ws);
        });

        ws.on('message', (data) => {
          // 转发消息到渲染进程
          BrowserWindow.getAllWindows().forEach(window => {
            window.webContents.send('websocket:message', {
              url,
              data: JSON.parse(data)
            });
          });
        });

        ws.on('close', (code, reason) => {
          console.log('WebSocket连接关闭:', url, code, reason);
          this.connections.delete(url);

          // 自动重连
          if (this.reconnectAttempts.get(url) < this.maxReconnectAttempts) {
            this.scheduleReconnect(url, options);
          }

          BrowserWindow.getAllWindows().forEach(window => {
            window.webContents.send('websocket:disconnected', {
              url,
              code,
              reason: reason.toString()
            });
          });
        });

        ws.on('error', (error) => {
          console.error('WebSocket错误:', url, error);

          BrowserWindow.getAllWindows().forEach(window => {
            window.webContents.send('websocket:error', {
              url,
              error: error.message
            });
          });
        });

      } catch (error) {
        reject(error);
      }
    });
  }

  send(url, data) {
    const ws = this.connections.get(url);
    if (ws && ws.readyState === WebSocket.OPEN) {
      ws.send(typeof data === 'string' ? data : JSON.stringify(data));
      return { success: true };
    }
    return { success: false, error: 'WebSocket未连接' };
  }

  disconnect(url) {
    const ws = this.connections.get(url);
    if (ws) {
      ws.close();
      this.connections.delete(url);
      return { success: true };
    }
    return { success: false, error: 'WebSocket连接不存在' };
  }

  scheduleReconnect(url, options) {
    const attempts = this.reconnectAttempts.get(url) || 0;
    const delay = Math.min(1000 * Math.pow(2, attempts), 30000); // 指数退避

    setTimeout(() => {
      if (this.reconnectAttempts.get(url) < this.maxReconnectAttempts) {
        this.reconnectAttempts.set(url, attempts + 1);
        console.log(`尝试重连 WebSocket (${attempts + 1}/${this.maxReconnectAttempts}):`, url);

        this.connect(url, options).catch(error => {
          console.error('WebSocket重连失败:', error);
        });
      }
    }, delay);
  }

  getConnectionStatus(url) {
    const ws = this.connections.get(url);
    if (!ws) return { connected: false };

    return {
      connected: ws.readyState === WebSocket.OPEN,
      readyState: ws.readyState,
      url: url,
      attempts: this.reconnectAttempts.get(url) || 0
    };
  }
}

  return new Promise((resolve, reject) => {
    const req = require('https').get(url, (res) => {
      // 检查响应状态
      if (res.statusCode >= 400) {
        reject(new Error(`HTTP错误: ${res.statusCode} - ${res.statusMessage}`));
        return;
      }

      let data = '';
      res.on('data', chunk => data += chunk);
      res.on('end', () => {
        // 检查内容类型
        const contentType = res.headers['content-type'];
        if (contentType && contentType.includes('application/json')) {
          try {
            resolve(JSON.parse(data));
          } catch (parseError) {
            reject(new Error('JSON解析失败: ' + parseError.message));
          }
        } else {
          resolve(data);
        }
      });
    });

    req.on('error', (error) => {
      if (error.code === 'ENOTFOUND') {
        reject(new Error('域名解析失败: ' + url));
      } else if (error.code === 'ECONNREFUSED') {
        reject(new Error('连接被拒绝: ' + url));
      } else if (error.code === 'ETIMEDOUT') {
        reject(new Error('请求超时: ' + url));
      } else {
        reject(new Error('网络错误: ' + error.message));
      }
    });

    req.setTimeout(30000, () => {
      req.destroy();
      reject(new Error('请求超时'));
    });
  });
}

        dataSize: result.data ? result.data.length : 0
      });

      return result;
    } catch (error) {
      const duration = Date.now() - startTime;

      console.error(`[HTTP] 请求失败: ${requestId}`, {
        url,
        duration: `${duration}ms`,
        error: error.message
      });

      throw error;
    }
  }

  async makeRequest(url, options) {
    // 实际的HTTP请求实现
    return new Promise((resolve, reject) => {
      const urlObj = new URL(url);
      const client = urlObj.protocol === 'https:' ? require('https') : require('http');

      const req = client.request({
        hostname: urlObj.hostname,
        port: urlObj.port || (urlObj.protocol === 'https:' ? 443 : 80),
        path: urlObj.pathname + urlObj.search,
        method: options.method || 'GET',
        headers: options.headers
      }, (res) => {
        let data = '';
        res.on('data', chunk => data += chunk);
        res.on('end', () => {
          resolve({
            statusCode: res.statusCode,
            headers: res.headers,
            data: data
          });
        });
      });

      req.on('error', reject);
      req.setTimeout(30000, () => {
        req.destroy();
        reject(new Error('Request timeout'));
      });

      req.end();
    });
  }

  getRequestHistory() {
    return Array.from(this.requests.entries()).map(([id, request]) => ({
      id,
      url: request.url,
      duration: Date.now() - request.startTime,
      status: request.completed ? 'completed' : 'pending'
    }));
  }

    const measurement = {
      timestamp: Date.now(),
      rss: usage.rss,
      heapTotal: usage.heapTotal,
      heapUsed: usage.heapUsed,
      external: usage.external
    };

    this.measurements.push(measurement);

    // 保持最近100次测量

      this.measurements.shift();
    }

    return measurement;
  }

  getMemoryTrend() {
    if (this.measurements.length < 10) {
      return 'insufficient-data';
    }

    const recent = this.measurements.slice(-10);
    const first = recent[0];
    const last = recent[recent.length - 1];

    const heapGrowth = last.heapUsed - first.heapUsed;
    const rssGrowth = last.rss - first.rss;

    if (heapGrowth > 10 * 1024 * 1024) { // 10MB增长
      return 'memory-leak-detected';
    }

    return 'normal';
  }

  generateReport() {
    const current = this.measurements[this.measurements.length - 1];
    const baseline = this.baselineMemory;

    return {
      baseline: {
        heapUsed: baseline.heapUsed,
        rss: baseline.rss
      },
      current: {
        heapUsed: current.heapUsed,
        rss: current.rss
      },
      growth: {
        heapUsed: current.heapUsed - baseline.heapUsed,
        rss: current.rss - baseline.rss
      },
      trend: this.getMemoryTrend(),
      measurements: this.measurements.length
    };
  }
}

    const errorInfo = {
      timestamp: Date.now(),
      channel,
      error: error.message,
      stack: error.stack,
      context: context
    };

    const count = this.errorCount.get(channel) || 0;
    this.errorCount.set(channel, count + 1);

    }

    // 移除敏感信息
    if (typeof data === 'object' && data !== null) {
      const sanitized = { ...data };
      if (sanitized.password) {
        sanitized.password = '***';
      }
      if (sanitized.token) {
        sanitized.token = '***';
      }
      return sanitized;
    }
    return data;
  }

  getMessageHistory(filter = {}) {
    return this.messageHistory.filter(message => {
      if (filter.channel && message.channel !== filter.channel) {
        return false;
      }
      if (filter.direction && message.direction !== filter.direction) {
        return false;
      }
      return true;
    });
  }

  }

        // 数据过滤
        const filteredData = dataFilter.filter(data);

        // 数据聚合
        const aggregatedData = dataAggregator.aggregate(source, filteredData);

        // 发送到监控窗口
        this.broadcastToWindows('realtime-data', {
          source,
          data: aggregatedData,
          timestamp: Date.now()
        });

        // 检查告警条件
        this.alertManager.checkAlerts(aggregatedData);

      } catch (error) {
        console.error('实时数据处理错误:', error);
      }
    });
  }

  setupAlertSystem() {
    this.alertManager.on('alert', (alert) => {
      // 发送告警通知
      this.sendAlertNotification(alert);

      // 显示告警窗口
      this.showAlertWindow(alert);
    });

    // 配置告警规则
    this.alertManager.addRule({
      name: 'high-cpu-usage',
      condition: (data) => data.cpuUsage > 80,
      severity: 'warning',
      message: 'CPU使用率过高'
    });

    this.alertManager.addRule({
      name: 'critical-error',
      condition: (data) => data.errorRate > 0.1,
      severity: 'critical',
      message: '错误率过高'
    });
  }

  broadcastToWindows(event, data) {
    for (const [windowId, window] of this.monitoringWindows) {
      if (!window.isDestroyed()) {
        window.webContents.send(event, data);
      }
    }
  }

  sendAlertNotification(alert) {
    const notification = new Notification({
      title: `告警: ${alert.severity}`,
      body: alert.message,
      icon: path.join(__dirname, 'assets', 'alert-icon.png')
    });

    notification.on('click', () => {
      const dashboardWindow = this.monitoringWindows.get('dashboard');
      if (dashboardWindow) {
        dashboardWindow.show();
        dashboardWindow.focus();
      }
    });

    notification.show();
  }

  showAlertWindow(alert) {
    const alertWindow = new BrowserWindow({
      width: 500,
      height: 300,
      title: '告警通知',
      modal: true,
      show: false,
      webPreferences: {
        nodeIntegration: false,
        contextIsolation: true
      }
    });

    alertWindow.loadFile('alert-window.html');

    alertWindow.once('ready-to-show', () => {
      alertWindow.show();

      // 3秒后自动关闭
      setTimeout(() => {
        alertWindow.close();
      }, 3000);
    });
  }
}

class WebSocketDataSource extends DataSource {
  constructor(url) {
    super(url);
    this.websocket = null;
  }

  connect() {
    try {
      this.websocket = new WebSocket(this.url);

      this.websocket.on('open', () => {
        console.log('WebSocket数据源已连接:', this.url);
        this.connected = true;
        this.emit('connected');
      });

      this.websocket.on('message', (data) => {
        try {
          const parsedData = JSON.parse(data);
          this.emit('data', parsedData);
        } catch (error) {
          console.error('WebSocket数据解析错误:', error);
        }
      });

      this.websocket.on('close', () => {
        console.log('WebSocket数据源连接关闭:', this.url);
        this.connected = false;
        this.emit('disconnected');

        // 自动重连
        setTimeout(() => {
          this.connect();
        }, 5000);
      });

      this.websocket.on('error', (error) => {
        console.error('WebSocket数据源错误:', error);
        this.emit('error', error);
      });

    } catch (error) {
      console.error('WebSocket连接失败:', error);
      this.emit('error', error);
    }
  }

  disconnect() {
    if (this.websocket) {
      this.websocket.close();
      this.websocket = null;
    }
  }
}

// HTTP数据源
class HTTPDataSource extends DataSource {
  constructor(url) {
    super(url);
    this.pollingInterval = null;
  }

  connect() {
    this.startPolling();
  }

  startPolling() {
    this.pollingInterval = setInterval(async () => {
      try {
        const response = await this.makeHTTPRequest(this.url);
        if (response.statusCode === 200) {
          const data = JSON.parse(response.data);
          this.emit('data', data);
        }
      } catch (error) {
        console.error('HTTP轮询错误:', error);
      }
    }, 5000); // 每5秒轮询一次
  }

  disconnect() {
    if (this.pollingInterval) {
      clearInterval(this.pollingInterval);
      this.pollingInterval = null;
    }
  }

  async makeHTTPRequest(url) {
    return new Promise((resolve, reject) => {
      const urlObj = new URL(url);
      const client = urlObj.protocol === 'https:' ? require('https') : require('http');

      const req = client.get(url, (res) => {
        let data = '';
        res.on('data', chunk => data += chunk);
        res.on('end', () => {
          resolve({
            statusCode: res.statusCode,
            data: data
          });
        });
      });

      req.on('error', reject);
      req.setTimeout(30000, () => {
        req.destroy();
        reject(new Error('Request timeout'));
      });

      req.end();
    });
  }
}

// 告警管理器
class AlertManager extends EventEmitter {
  constructor() {
    super();
    this.rules = new Map();
    this.alertHistory = [];
  }

  addRule(rule) {
    this.rules.set(rule.name, rule);
  }

  checkAlerts(data) {
    for (const [ruleName, rule] of this.rules) {
      try {
        if (rule.condition(data)) {
          const alert = {
            id: Date.now() + Math.random(),
            ruleName,
            severity: rule.severity,
            message: rule.message,
            data: data,
            timestamp: new Date().toISOString()
          };

          this.handleAlert(alert);
        }
      } catch (error) {
        console.error(`告警规则检查错误 (${ruleName}):`, error);
      }
    }
  }

  handleAlert(alert) {
    // 添加到历史记录
    this.alertHistory.push(alert);

    // 保持最近100条告警记录
    if (this.alertHistory.length > 100) {
      this.alertHistory.shift();
    }

    // 发出告警事件
    this.emit('alert', alert);

    console.warn(`告警触发: ${alert.ruleName} - ${alert.message}`, alert);
  }
}
