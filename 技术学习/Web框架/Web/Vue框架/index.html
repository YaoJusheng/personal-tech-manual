    概述¶

    Vue.js是一套用于构建用户界面的渐进式JavaScript框架。由尤雨溪创建，Vue.js专注于视图层，具有简单灵活的API和优秀的性能。

    1. 响应式数据绑定¶

    // Vue 3 响应式系统

  setup() {
    // 基础响应式数据

});

        );
      }

      // 排序

        let bValue = b[this.sortBy];

        if (typeof aValue === 'string') {
          aValue = aValue.toLowerCase();
          bValue = bValue.toLowerCase();
        }

        if (this.sortOrder === 'asc') {
          return aValue > bValue ? 1 : -1;
        } else {
          return aValue < bValue ? 1 : -1;
        }
      });
    },

        this.users.splice(index, 1);

    },

    // composables/useWebSocket.js
import { ref, onMounted, onUnmounted } from 'vue';

export function useWebSocket(url, options = {}) {
  const ws = ref(null);
  const isConnected = ref(false);
  const messages = ref([]);
  const error = ref(null);

  const connect = () => {
    try {
      ws.value = new WebSocket(url);

      ws.value.onopen = () => {
        isConnected.value = true;
        console.log('WebSocket connected');
      };

      ws.value.onmessage = (event) => {
        const data = JSON.parse(event.data);
        messages.value.push({
          ...data,
          timestamp: new Date().toISOString()
        });
      };

      ws.value.onclose = () => {
        isConnected.value = false;
        console.log('WebSocket disconnected');
      };

      ws.value.onerror = (event) => {
        error.value = event;
        console.error('WebSocket error:', event);
      };
    } catch (err) {
      error.value = err;
    }
  };

  const disconnect = () => {
    if (ws.value) {
      ws.value.close();
    }
  };

  const sendMessage = (message) => {
    if (ws.value && isConnected.value) {
      ws.value.send(JSON.stringify(message));
    }
  };

  onMounted(() => {
    connect();
  });

  onUnmounted(() => {
    disconnect();
  });

  return {
    ws: ws.value,
    isConnected,
    messages,
    error,
    connect,
    disconnect,
    sendMessage
  };
}

    // composables/useDataTable.js
import { ref, computed, watch } from 'vue';

export function useDataTable(initialData = []) {
  const data = ref([...initialData]);
  const filteredData = ref([]);
  const sortConfig = ref({
    column: null,
    direction: 'asc'
  });
  const pagination = ref({
    page: 1,
    pageSize: 50,
    total: 0
  });

  const columns = computed(() => {
    if (data.value.length === 0) return [];
    return Object.keys(data.value[0]).map(key => ({
      key,
      label: key.charAt(0).toUpperCase() + key.slice(1),
      sortable: true,
      filterable: true
    }));
  });

  const paginatedData = computed(() => {
    const start = (pagination.value.page - 1) * pagination.value.pageSize;
    const end = start + pagination.value.pageSize;
    return filteredData.value.slice(start, end);
  });

  const totalPages = computed(() => {
    return Math.ceil(filteredData.value.length / pagination.value.pageSize);
  });

  const sortData = (column) => {
    if (sortConfig.value.column === column) {
      sortConfig.value.direction = sortConfig.value.direction === 'asc' ? 'desc' : 'asc';
    } else {
      sortConfig.value.column = column;
      sortConfig.value.direction = 'asc';
    }

    filteredData.value.sort((a, b) => {
      const aVal = a[column];
      const bVal = b[column];

      if (typeof aVal === 'number' && typeof bVal === 'number') {
        return sortConfig.value.direction === 'asc' ? aVal - bVal : bVal - aVal;
      }

      const comparison = aVal > bVal ? 1 : -1;
      return sortConfig.value.direction === 'asc' ? comparison : -comparison;
    });
  };

  const filterData = (filters) => {
    filteredData.value = data.value.filter(item => {
      return Object.entries(filters).every(([key, value]) => {
        if (!value) return true;
        return item[key].toString().toLowerCase().includes(value.toLowerCase());
      });
    });
  };

  const setPage = (page) => {
    pagination.value.page = Math.max(1, Math.min(page, totalPages.value));
  };

  const setPageSize = (pageSize) => {
    pagination.value.pageSize = pageSize;
    pagination.value.page = 1;
  };

  watch(data, () => {
    filteredData.value = [...data.value];
    pagination.value.total = data.value.length;
  }, { immediate: true });

  return {
    data,
    filteredData,
    paginatedData,
    columns,
    pagination,
    sortConfig,
    totalPages,
    sortData,
    filterData,
    setPage,
    setPageSize
  };
}

// composables/useUser.js
import { ref, computed } from 'vue';
import { useUserStore } from '@/stores/user';

export function useUser() {
  const userStore = useUserStore();
  const isLoading = ref(false);

  const currentUser = computed(() => userStore.currentUser);
  const isAuthenticated = computed(() => userStore.isAuthenticated);

  const login = async (credentials) => {
    isLoading.value = true;
    try {
      await userStore.login(credentials);
    } finally {
      isLoading.value = false;
    }
  };

  const logout = async () => {
    await userStore.logout();
  };

  return {
    currentUser,
    isAuthenticated,
    isLoading,
    login,
    logout
  };
}
