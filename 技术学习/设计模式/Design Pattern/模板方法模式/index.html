    模板方法模式（Template Method）¶

    • 模板方法模式（Template Method）
      • 1.简介
      • 2.使用场景
      • 3.优缺点
      • 4.实现示例
        • Python 实现
        • Go 实现
        • Rust 实现

    1.简介¶

    模板方法模式是一种行为型设计模式，它在超类中定义了一个算法的框架，允许子类在不改变算法整体结构的情况下重写算法的特定步骤。模板方法模式通过把不变的行为搬移到超类，去除了子类中的重复代码。

    2.使用场景¶

    • 多个类有相同的方法，逻辑相同，细节不同
    • 重要、复杂的算法，可以把核心算法设计为模板方法
    • 重构时，把相同的代码抽取到父类中
    • 数据挖掘算法、Web框架、游戏AI

    3.优缺点¶

    优点：

    • 提取了公共代码部分，易于维护
    • 行为由父类控制，子类实现
    • 符合开闭原则

    缺点：

    • 每一个不同的实现都需要一个子类来实现，导致类的个数增加
    • 类数目的增加，间接地增加了系统实现的复杂度

    4.实现示例¶

    Python 实现¶

    from abc import ABC, abstractmethod
import time
from typing import List, Optional

# 抽象模板类
class DataProcessor(ABC):
    """数据处理模板方法"""

    def process_data(self, data_source: str) -> dict:
        """模板方法：定义数据处理的整体流程"""
        print("=== 开始数据处理流程 ===")

        # 1. 读取数据
        raw_data = self.read_data(data_source)
        print(f"读取到 {len(raw_data)} 条原始数据")

        # 2. 验证数据
        if not self.validate_data(raw_data):
            raise ValueError("数据验证失败")
        print("数据验证通过")

        # 3. 处理数据
        processed_data = self.transform_data(raw_data)
        print(f"处理后得到 {len(processed_data)} 条数据")

        # 4. 保存数据（可选步骤）
        if self.should_save_data():
            self.save_data(processed_data)
            print("数据已保存")

        # 5. 生成报告
        report = self.generate_report(processed_data)

        print("=== 数据处理流程完成 ===")
        return report

    # 抽象方法：子类必须实现
    @abstractmethod
    def read_data(self, source: str) -> List[dict]:
        """读取数据"""
        pass

    @abstractmethod
    def transform_data(self, data: List[dict]) -> List[dict]:
        """转换数据"""
        pass

    # 具体方法：提供默认实现
    def validate_data(self, data: List[dict]) -> bool:
        """验证数据（默认实现）"""
        return len(data) > 0

    def generate_report(self, data: List[dict]) -> dict:
        """生成报告（默认实现）"""
        return {
            "total_records": len(data),
            "processing_time": time.time(),
            "status": "success"
        }

    # 钩子方法：子类可以选择性重写
    def should_save_data(self) -> bool:
        """是否保存数据的钩子方法"""
        return True

    def save_data(self, data: List[dict]) -> None:
        """保存数据的钩子方法"""
        print(f"保存了 {len(data)} 条数据到默认存储")

# 具体实现类 - CSV数据处理器
class CSVDataProcessor(DataProcessor):
    def read_data(self, source: str) -> List[dict]:
        """从CSV文件读取数据"""
        print(f"从CSV文件读取数据: {source}")
        # 模拟CSV数据
        return [
            {"id": 1, "name": "Alice", "age": 25, "salary": 50000},
            {"id": 2, "name": "Bob", "age": 30, "salary": 60000},
            {"id": 3, "name": "Charlie", "age": 35, "salary": 70000},
        ]

    def transform_data(self, data: List[dict]) -> List[dict]:
        """转换CSV数据：添加年薪字段"""
        transformed = []
        for record in data:
            new_record = record.copy()
            new_record["annual_salary"] = record["salary"] * 12
            new_record["age_group"] = "young" if record["age"] < 30 else "senior"
            transformed.append(new_record)
        return transformed

    def validate_data(self, data: List[dict]) -> bool:
        """CSV特定的数据验证"""
        if not super().validate_data(data):
            return False

        # 检查必需字段
        required_fields = ["id", "name", "age", "salary"]
        for record in data:
            for field in required_fields:
                if field not in record:
                    return False
        return True

# 具体实现类 - JSON数据处理器
class JSONDataProcessor(DataProcessor):
    def read_data(self, source: str) -> List[dict]:
        """从JSON API读取数据"""
        print(f"从JSON API读取数据: {source}")
        # 模拟JSON API数据
        return [
            {"user_id": "u001", "username": "alice", "profile": {"age": 25, "department": "IT"}},
            {"user_id": "u002", "username": "bob", "profile": {"age": 30, "department": "HR"}},
            {"user_id": "u003", "username": "charlie", "profile": {"age": 35, "department": "Finance"}},
        ]

    def transform_data(self, data: List[dict]) -> List[dict]:
        """转换JSON数据：扁平化结构"""
        transformed = []
        for record in data:
            new_record = {
                "id": record["user_id"],
                "name": record["username"],
                "age": record["profile"]["age"],
                "department": record["profile"]["department"]
            }
            transformed.append(new_record)
        return transformed

    def should_save_data(self) -> bool:
        """JSON处理器选择不保存数据"""
        return False

    def generate_report(self, data: List[dict]) -> dict:
        """生成JSON特定的报告"""
        base_report = super().generate_report(data)

        # 添加部门统计
        departments = {}
        for record in data:
            dept = record.get("department", "Unknown")
            departments[dept] = departments.get(dept, 0) + 1

        base_report["department_stats"] = departments
        return base_report

    Go 实现¶

    package main

import (
    "fmt"
    "time"
)

// DataRecord 表示数据记录
type DataRecord map[string]interface{}

// DataProcessor 抽象数据处理器接口
type DataProcessor interface {
    ReadData(source string) []DataRecord
    TransformData(data []DataRecord) []DataRecord
    ValidateData(data []DataRecord) bool
    GenerateReport(data []DataRecord) map[string]interface{}
    ShouldSaveData() bool
    SaveData(data []DataRecord)
}

// BaseDataProcessor 基础数据处理器，实现模板方法
type BaseDataProcessor struct {
    processor DataProcessor
}

// NewBaseDataProcessor 创建基础数据处理器
func NewBaseDataProcessor(processor DataProcessor) *BaseDataProcessor {
    return &BaseDataProcessor{processor: processor}
}

// ProcessData 模板方法：定义数据处理的整体流程
func (b *BaseDataProcessor) ProcessData(dataSource string) map[string]interface{} {
    fmt.Println("=== 开始数据处理流程 ===")

    // 1. 读取数据
    rawData := b.processor.ReadData(dataSource)
    fmt.Printf("读取到 %d 条原始数据\n", len(rawData))

    // 2. 验证数据
    if !b.processor.ValidateData(rawData) {
        panic("数据验证失败")
    }
    fmt.Println("数据验证通过")

    // 3. 处理数据
    processedData := b.processor.TransformData(rawData)
    fmt.Printf("处理后得到 %d 条数据\n", len(processedData))

    // 4. 保存数据（可选步骤）
    if b.processor.ShouldSaveData() {
        b.processor.SaveData(processedData)
        fmt.Println("数据已保存")
    }

    // 5. 生成报告
    report := b.processor.GenerateReport(processedData)

    fmt.Println("=== 数据处理流程完成 ===")
    return report
}

// CSVDataProcessor CSV数据处理器
type CSVDataProcessor struct{}

func NewCSVDataProcessor() *CSVDataProcessor {
    return &CSVDataProcessor{}
}

func (c *CSVDataProcessor) ReadData(source string) []DataRecord {
    fmt.Printf("从CSV文件读取数据: %s\n", source)
    // 模拟CSV数据
    return []DataRecord{
        {"id": 1, "name": "Alice", "age": 25, "salary": 50000},
        {"id": 2, "name": "Bob", "age": 30, "salary": 60000},
        {"id": 3, "name": "Charlie", "age": 35, "salary": 70000},
    }
}

func (c *CSVDataProcessor) TransformData(data []DataRecord) []DataRecord {
    transformed := make([]DataRecord, 0, len(data))
    for _, record := range data {
        newRecord := make(DataRecord)
        for k, v := range record {
            newRecord[k] = v
        }

        if salary, ok := record["salary"].(int); ok {
            newRecord["annual_salary"] = salary * 12
        }

        if age, ok := record["age"].(int); ok {
            if age < 30 {
                newRecord["age_group"] = "young"
            } else {
                newRecord["age_group"] = "senior"
            }
        }

        transformed = append(transformed, newRecord)
    }
    return transformed
}

func (c *CSVDataProcessor) ValidateData(data []DataRecord) bool {
    if len(data) == 0 {
        return false
    }

    requiredFields := []string{"id", "name", "age", "salary"}
    for _, record := range data {
        for _, field := range requiredFields {
            if _, exists := record[field]; !exists {
                return false
            }
        }
    }
    return true
}

func (c *CSVDataProcessor) GenerateReport(data []DataRecord) map[string]interface{} {
    return map[string]interface{}{
        "total_records":   len(data),
        "processing_time": time.Now().Unix(),
        "status":         "success",
    }
}

func (c *CSVDataProcessor) ShouldSaveData() bool {
    return true
}

func (c *CSVDataProcessor) SaveData(data []DataRecord) {
    fmt.Printf("保存了 %d 条数据到默认存储\n", len(data))
}

}

    use std::collections::HashMap;

// 数据记录类型
pub type DataRecord = HashMap<String, serde_json::Value>;

// 数据处理器trait
pub trait DataProcessor {
    fn read_data(&self, source: &str) -> Vec<DataRecord>;
    fn transform_data(&self, data: Vec<DataRecord>) -> Vec<DataRecord>;
    fn validate_data(&self, data: &[DataRecord]) -> bool {
        !data.is_empty()
    }
    fn generate_report(&self, data: &[DataRecord]) -> HashMap<String, serde_json::Value> {
        let mut report = HashMap::new();
        report.insert("total_records".to_string(), serde_json::Value::Number(serde_json::Number::from(data.len())));
        report.insert("processing_time".to_string(), serde_json::Value::Number(serde_json::Number::from(
            SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs()
        )));
        report.insert("status".to_string(), serde_json::Value::String("success".to_string()));
        report
    }
    fn should_save_data(&self) -> bool {
        true
    }
    fn save_data(&self, data: &[DataRecord]) {
        println!("保存了 {} 条数据到默认存储", data.len());
    }
}

// 基础数据处理器，实现模板方法
pub struct BaseDataProcessor<T: DataProcessor> {
    processor: T,
}

impl<T: DataProcessor> BaseDataProcessor<T> {
    pub fn new(processor: T) -> Self {
        BaseDataProcessor { processor }
    }

    // 模板方法：定义数据处理的整体流程
    pub fn process_data(&self, data_source: &str) -> HashMap<String, serde_json::Value> {
        println!("=== 开始数据处理流程 ===");

        // 1. 读取数据
        let raw_data = self.processor.read_data(data_source);
        println!("读取到 {} 条原始数据", raw_data.len());

        // 2. 验证数据
        if !self.processor.validate_data(&raw_data) {
            panic!("数据验证失败");
        }
        println!("数据验证通过");

        // 3. 处理数据
        let processed_data = self.processor.transform_data(raw_data);
        println!("处理后得到 {} 条数据", processed_data.len());

        // 4. 保存数据（可选步骤）
        if self.processor.should_save_data() {
            self.processor.save_data(&processed_data);
            println!("数据已保存");
        }

        // 5. 生成报告
        let report = self.processor.generate_report(&processed_data);

        println!("=== 数据处理流程完成 ===");
        report
    }
}

    fn read_data(&self, source: &str) -> Vec<DataRecord> {

        // 模拟CSV数据
        vec![
            {

            },

                record.insert("name".to_string(), serde_json::Value::String("Bob".to_string()));
                record.insert("age".to_string(), serde_json::Value::Number(serde_json::Number::from(30)));
                record.insert("salary".to_string(), serde_json::Value::Number(serde_json::Number::from(60000)));
                record
            },
            {
                let mut record = HashMap::new();
                record.insert("id".to_string(), serde_json::Value::Number(serde_json::Number::from(3)));
                record.insert("name".to_string(), serde_json::Value::String("Charlie".to_string()));
                record.insert("age".to_string(), serde_json::Value::Number(serde_json::Number::from(35)));
                record.insert("salary".to_string(), serde_json::Value::Number(serde_json::Number::from(70000)));
                record
            },
        ]
    }

    fn transform_data(&self, data: Vec<DataRecord>) -> Vec<DataRecord> {
        data.into_iter().map(|mut record| {
            // 添加年薪字段
            if let Some(serde_json::Value::Number(salary)) = record.get("salary") {
                if let Some(salary_val) = salary.as_i64() {
                    record.insert("annual_salary".to_string(),
                        serde_json::Value::Number(serde_json::Number::from(salary_val * 12)));
                }
            }

            // 添加年龄组字段
            if let Some(serde_json::Value::Number(age)) = record.get("age") {
                if let Some(age_val) = age.as_i64() {
                    let age_group = if age_val < 30 { "young" } else { "senior" };
                    record.insert("age_group".to_string(),
                        serde_json::Value::String(age_group.to_string()));
                }
            }

            record
        }).collect()
    }

    fn validate_data(&self, data: &[DataRecord]) -> bool {
        if data.is_empty() {
            return false;
        }

        let required_fields = ["id", "name", "age", "salary"];
        for record in data {
            for field in &required_fields {
                if !record.contains_key(*field) {
                    return false;
                }
            }
        }
        true
    }
}

                profile.insert("department".to_string(), serde_json::Value::String("IT".to_string()));
                record.insert("profile".to_string(), serde_json::Value::Object(
                    profile.into_iter().collect()
                ));
                record
            },
            {
                let mut record = HashMap::new();
                record.insert("user_id".to_string(), serde_json::Value::String("u002".to_string()));
                record.insert("username".to_string(), serde_json::Value::String("bob".to_string()));

                let mut profile = HashMap::new();
                profile.insert("age".to_string(), serde_json::Value::Number(serde_json::Number::from(30)));
                profile.insert("department".to_string(), serde_json::Value::String("HR".to_string()));
                record.insert("profile".to_string(), serde_json::Value::Object(
                    profile.into_iter().collect()
                ));
                record
            },
            {
                let mut record = HashMap::new();
                record.insert("user_id".to_string(), serde_json::Value::String("u003".to_string()));
                record.insert("username".to_string(), serde_json::Value::String("charlie".to_string()));

                let mut profile = HashMap::new();
                profile.insert("age".to_string(), serde_json::Value::Number(serde_json::Number::from(35)));
                profile.insert("department".to_string(), serde_json::Value::String("Finance".to_string()));
                record.insert("profile".to_string(), serde_json::Value::Object(
                    profile.into_iter().collect()
                ));
                record
            },
        ]
    }

    fn transform_data(&self, data: Vec<DataRecord>) -> Vec<DataRecord> {
        data.into_iter().map(|record| {
            let mut new_record = HashMap::new();

            if let Some(user_id) = record.get("user_id") {
                new_record.insert("id".to_string(), user_id.clone());
            }

            if let Some(username) = record.get("username") {
                new_record.insert("name".to_string(), username.clone());
            }

            if let Some(serde_json::Value::Object(profile)) = record.get("profile") {
                if let Some(age) = profile.get("age") {
                    new_record.insert("age".to_string(), age.clone());
                }
                if let Some(department) = profile.get("department") {
                    new_record.insert("department".to_string(), department.clone());
                }
            }

            new_record
        }).collect()
    }

    fn should_save_data(&self) -> bool {
        false
    }

    fn generate_report(&self, data: &[DataRecord]) -> HashMap<String, serde_json::Value> {
        let mut base_report = self.validate_data(data).then(|| {
            let mut report = HashMap::new();
            report.insert("total_records".to_string(), serde_json::Value::Number(serde_json::Number::from(data.len())));
            report.insert("processing_time".to_string(), serde_json::Value::Number(serde_json::Number::from(
                SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs()
            )));
            report.insert("status".to_string(), serde_json::Value::String("success".to_string()));
            report
        }).unwrap_or_default();

        // 添加部门统计
        let mut departments: HashMap<String, i32> = HashMap::new();
        for record in data {
            let dept = record.get("department")
                .and_then(|v| v.as_str())
                .unwrap_or("Unknown");
            *departments.entry(dept.to_string()).or_insert(0) += 1;
        }

        let dept_stats: serde_json::Map<String, serde_json::Value> = departments
            .into_iter()
            .map(|(k, v)| (k, serde_json::Value::Number(serde_json::Number::from(v))))
            .collect();

        base_report.insert("department_stats".to_string(),
            serde_json::Value::Object(dept_stats));

        base_report
    }
}
