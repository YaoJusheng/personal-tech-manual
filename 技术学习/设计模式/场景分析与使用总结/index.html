    一、概述¶

    本文档基于《设计模式.md》中的23种设计模式实现，总结分析各种实际场景下应优先选择的设计模式，并提供详细的分析推导过程。

    设计模式三大分类¶

    • 创建型模式（Creational Patterns）：关注对象的创建过程
    • 结构型模式（Structural Patterns）：关注类和对象的组合
    • 行为型模式（Behavioral Patterns）：关注对象间的通信和职责分配

    二、设计模式分类与选择指南¶

    创建型模式适用场景¶

    核心问题：如何创建对象？何时创建？创建什么样的对象？

    适用情况：

    • 需要控制对象的创建过程
    • 对象创建逻辑复杂
    • 需要根据条件创建不同类型的对象
    • 需要延迟对象创建
    • 需要确保对象的唯一性

    结构型模式适用场景¶

    核心问题：如何组合现有的类和对象？如何让不兼容的接口协同工作？

    适用情况：

    • 需要适配不兼容的接口
    • 需要为对象添加新功能而不修改原有代码
    • 需要简化复杂子系统的使用
    • 需要在运行时组合对象
    • 需要共享对象以节省内存

    行为型模式适用场景¶

    核心问题：对象间如何通信？如何分配职责？如何处理算法变化？

    适用情况：

    • 需要在对象间传递请求
    • 需要根据状态改变行为
    • 需要支持多种算法切换
    • 需要遍历集合对象
    • 需要实现撤销/重做功能
    • 需要观察者通知机制

class DatabaseAccess(DataAccess):

        self.data = {}

    def read_data(self, user_id, data_id):
        return f"从数据库读取数据 {data_id}"

    def write_data(self, user_id, data_id, data):

        return f"向数据库写入数据 {data_id}"

class PermissionProxy(DataAccess):
    def __init__(self, real_subject: DataAccess):
        self._real_subject = real_subject
        self._permissions = {
            'admin': ['read', 'write'],
            'manager': ['read', 'write'],
            'employee': ['read'],
            'guest': []
        }

    def _check_permission(self, user_role, operation):
        return operation in self._permissions.get(user_role, [])

    def read_data(self, user_id, data_id):
        user_role = self._get_user_role(user_id)
        if self._check_permission(user_role, 'read'):
            return self._real_subject.read_data(user_id, data_id)
        else:
            return f"用户 {user_id} 没有读取权限"

    def write_data(self, user_id, data_id, data):
        user_role = self._get_user_role(user_id)
        if self._check_permission(user_role, 'write'):
            return self._real_subject.write_data(user_id, data_id, data)
        else:
            return f"用户 {user_id} 没有写入权限"

    def _get_user_role(self, user_id):
        # 模拟根据用户ID获取角色
        role_map = {
            'user001': 'admin',
            'user002': 'manager',
            'user003': 'employee',
            'user004': 'guest'
        }
        return role_map.get(user_id, 'guest')

    2. 观察者模式 - 数据同步:

    from abc import ABC, abstractmethod

class Observer(ABC):
    @abstractmethod
    def update(self, data):
        pass

class Subject(ABC):
    def __init__(self):
        self._observers = []

    def attach(self, observer):
        self._observers.append(observer)

    def detach(self, observer):
        self._observers.remove(observer)

    def notify(self, data):
        for observer in self._observers:
            observer.update(data)

# 具体主题 - 数据同步器
class DataSyncManager(Subject):
    def __init__(self):
        super().__init__()
        self._data = {}
        self._is_online = True

    def set_online_status(self, is_online):
        self._is_online = is_online
        self.notify({'status': 'online' if is_online else 'offline'})

    def sync_data(self, key, value):
        self._data[key] = value
        self.notify({'action': 'sync', 'key': key, 'value': value})

# 具体观察者
class CacheManager(Observer):
    def __init__(self):
        self.cache = {}

    def update(self, data):
        if data.get('action') == 'sync':
            self.cache[data['key']] = data['value']
            print(f"缓存更新: {data['key']} = {data['value']}")

class UIManager(Observer):
    def update(self, data):
        if 'status' in data:
            print(f"UI状态更新: {data['status']}")
        elif 'action' in data:
            print(f"UI数据更新: {data['key']} = {data['value']}")

class NetworkManager(Observer):
    def update(self, data):
        if 'status' in data and data['status'] == 'offline':
            print("网络离线，启用离线模式")

# 使用示例
sync_manager = DataSyncManager()

# 注册观察者

network_manager = NetworkManager()

sync_manager.attach(ui_manager)

class ApplicationContext:
    _instance = None
    _initialized = False

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

    def __init__(self):
        if not self._initialized:
            self._beans = {}
            self._initialized = True

    def register_bean(self, name, bean):
        self._beans[name] = bean

    def get_bean(self, name):
        return self._beans.get(name)

# 使用示例
context = ApplicationContext()
context.register_bean('userService', 'UserService实例')
print(context.get_bean('userService'))

    2. 工厂模式 - BeanFactory:

    from abc import ABC, abstractmethod

class BeanFactory(ABC):
    @abstractmethod
    def create_bean(self, bean_type):
        pass

class AnnotationBeanFactory(BeanFactory):
    def create_bean(self, bean_type):
        if bean_type == 'service':
            return ServiceBean()
        elif bean_type == 'repository':
            return RepositoryBean()
        elif bean_type == 'controller':
            return ControllerBean()
        else:
            raise ValueError(f"未知的Bean类型: {bean_type}")

class ServiceBean:
    def __init__(self):
        self.type = 'Service'

class RepositoryBean:
    def __init__(self):
        self.type = 'Repository'

class ControllerBean:
    def __init__(self):
        self.type = 'Controller'

# 使用示例
factory = AnnotationBeanFactory()
service = factory.create_bean('service')
print(f"创建的Bean类型: {service.type}")

    Django框架中的设计模式¶

    1. 模板方法模式 - 视图基类:

    from abc import ABC, abstractmethod

class BaseView(ABC):
    def process_request(self, request):
        # 模板方法定义处理流程
        self.before_process(request)
        result = self.handle_request(request)
        self.after_process(request)
        return result

    def before_process(self, request):
        print("请求处理前：权限检查")

    @abstractmethod
    def handle_request(self, request):
        pass

    def after_process(self, request):
        print("请求处理后：日志记录")

class UserView(BaseView):
    def handle_request(self, request):
        return f"处理用户请求: {request}"

class ProductView(BaseView):
    def handle_request(self, request):
        return f"处理产品请求: {request}"

# 使用示例
user_view = UserView()
result = user_view.process_request("GET /users/")
print(result)

# 2. 工厂模式 - 服务创建
class ServiceFactory:
    @staticmethod
    def create_service(service_type):
        if service_type == 'user':
            return UserService()
        elif service_type == 'product':
            return ProductService()
        elif service_type == 'order':
            return OrderService()
        else:
            raise ValueError(f"未知服务类型: {service_type}")

# 3. 外观模式 - 统一服务接口
class ECommerceFacade:
    def __init__(self):
        self.user_service = ServiceFactory.create_service('user')
        self.product_service = ServiceFactory.create_service('product')
        self.order_service = ServiceFactory.create_service('order')

    def create_order(self, user_id, product_ids):
        # 验证用户
        user = self.user_service.get_user(user_id)
        if not user:
            return "用户不存在"

        # 验证商品
        products = []
        for product_id in product_ids:
            product = self.product_service.get_product(product_id)
            if not product:
                return f"商品 {product_id} 不存在"
            products.append(product)

        # 创建订单
        order = self.order_service.create_order(user, products)
        return f"订单创建成功: {order}"

# 使用示例
facade = ECommerceFacade()

    9.3 实施建议¶

    渐进式应用策略¶

    1. 第一阶段：识别核心问题，应用单一模式
    2. 第二阶段：发现模式不足，考虑模式组合
    3. 第三阶段：优化模式组合，提升系统质量

    常见陷阱与避免¶

    • 过度设计：不要为了使用模式而使用模式
    • 模式滥用：简单问题不要用复杂模式
    • 忽视性能：某些模式可能影响性能
    • 维护困难：过多模式组合可能增加复杂度

    十、总结与展望¶

    10.1 核心要点回顾¶

    1. 问题导向：始终以解决实际问题为目标
    2. 分类思维：先确定模式分类，再选择具体模式
    3. 渐进优化：从简单开始，逐步完善
    4. 组合应用：复杂问题往往需要多个模式协作
    5. 实践验证：理论结合实践，在项目中验证效果

    10.2 学习建议¶

    1. 掌握分类：深入理解三大分类的核心思想
    2. 熟练常用：重点掌握高频使用的模式
    3. 理解组合：学会多个模式的协作使用
    4. 项目实践：在实际项目中应用和验证
    5. 持续改进：根据项目经验不断优化选择

    10.3 扩展方向¶

    • 领域特定模式：针对特定领域的模式应用
    • 架构模式：更高层次的架构设计模式
    • 并发模式：多线程和并发编程中的模式
    • 分布式模式：分布式系统中的设计模式
